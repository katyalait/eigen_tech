21@python_2_unicode_compatible 

22class LazyCorpusLoader(object): 

23 """ 

24 To see the API documentation for this lazily loaded corpus, first 

25 run corpus.ensure_loaded(), and then run help(this_corpus). 

26 

27 LazyCorpusLoader is a proxy object which is used to stand in for a 

28 corpus object before the corpus is loaded. This allows NLTK to 

29 create an object for each corpus, but defer the costs associated 

30 with loading those corpora until the first time that they're 

31 actually accessed. 

32 

33 The first time this object is accessed in any way, it will load 

34 the corresponding corpus, and transform itself into that corpus 

35 (by modifying its own ``__class__`` and ``__dict__`` attributes). 

36 

37 If the corpus can not be found, then accessing this object will 

38 raise an exception, displaying installation instructions for the 

39 NLTK data package. Once they've properly installed the data 

40 package (or modified ``nltk.data.path`` to point to its location), 

41 they can then use the corpus object without restarting python. 

42 

43 :param name: The name of the corpus 

44 :type name: str 

45 :param reader_cls: The specific CorpusReader class, e.g. PlaintextCorpusReader, WordListCorpusReader 

46 :type reader: nltk.corpus.reader.api.CorpusReader 

47 :param nltk_data_subdir: The subdirectory where the corpus is stored. 

48 :type nltk_data_subdir: str 

49 :param *args: Any other non-keywords arguments that `reader_cls` might need. 

50 :param *kargs: Any other keywords arguments that `reader_cls` might need. 

51 """ 

52 

53 def __init__(self, name, reader_cls, *args, **kwargs): 

54 from nltk.corpus.reader.api import CorpusReader 

55 

56 assert issubclass(reader_cls, CorpusReader) 

57 self.__name = self.__name__ = name 

58 self.__reader_cls = reader_cls 

59 # If nltk_data_subdir is set explicitly 

60 if 'nltk_data_subdir' in kwargs: 

61 # Use the specified subdirectory path 

62 self.subdir = kwargs['nltk_data_subdir'] 

63 # Pops the `nltk_data_subdir` argument, we don't need it anymore. 

64 kwargs.pop('nltk_data_subdir', None) 

65 else: # Otherwise use 'nltk_data/corpora' 

66 self.subdir = 'corpora' 

67 self.__args = args 

68 self.__kwargs = kwargs 

69 

70 def __load(self): 

71 # Find the corpus root directory. 

72 zip_name = re.sub(r'(([^/]+)(/.*)?)', r'\2.zip/\1/', self.__name) 

73 if TRY_ZIPFILE_FIRST: 

74 try: 

75 root = nltk.data.find('{}/{}'.format(self.subdir, zip_name)) 

76 except LookupError as e: 

77 try: 

78 root = nltk.data.find('{}/{}'.format(self.subdir, self.__name)) 

79 except LookupError: 

80 raise e 

81 else: 

82 try: 

83 root = nltk.data.find('{}/{}'.format(self.subdir, self.__name)) 

84 except LookupError as e: 

85 try: 

86 root = nltk.data.find('{}/{}'.format(self.subdir, zip_name)) 

87 except LookupError: 

88 raise e 

89 

90 # Load the corpus. 

91 corpus = self.__reader_cls(root, *self.__args, **self.__kwargs) 

92 

93 # This is where the magic happens! Transform ourselves into 

94 # the corpus by modifying our own __dict__ and __class__ to 

95 # match that of the corpus. 

96 

97 args, kwargs = self.__args, self.__kwargs 

98 name, reader_cls = self.__name, self.__reader_cls 

99 

100 self.__dict__ = corpus.__dict__ 

101 self.__class__ = corpus.__class__ 

102 

103 # _unload support: assign __dict__ and __class__ back, then do GC. 

104 # after reassigning __dict__ there shouldn't be any references to 

105 # corpus data so the memory should be deallocated after gc.collect() 

106 def _unload(self): 

107 lazy_reader = LazyCorpusLoader(name, reader_cls, *args, **kwargs) 

108 self.__dict__ = lazy_reader.__dict__ 

109 self.__class__ = lazy_reader.__class__ 

110 gc.collect() 

111 

112 self._unload = _make_bound_method(_unload, self) 

113 

114 def __getattr__(self, attr): 

115 

116 # Fix for inspect.isclass under Python 2.6 

117 # (see http://bugs.python.org/issue1225107). 

118 # Without this fix tests may take extra 1.5GB RAM 

119 # because all corpora gets loaded during test collection. 

120 if attr == '__bases__': 

121 raise AttributeError("LazyCorpusLoader object has no attribute '__bases__'") 

122 

123 self.__load() 

124 # This looks circular, but its not, since __load() changes our 

125 # __class__ to something new: 

126 return getattr(self, attr) 

127 

128 def __repr__(self): 

129 return '<%s in %r (not loaded yet)>' % ( 

130 self.__reader_cls.__name__, 

131 '.../corpora/' + self.__name, 

132 )

135 # If an exception occures during corpus loading then

137 # we shouldn't trigger corpus loading again in this case. 

138 pass
