10r""" 

11Regular-Expression Tokenizers 

12 

13A ``RegexpTokenizer`` splits a string into substrings using a regular expression. 

14For example, the following tokenizer forms tokens out of alphabetic sequences, 

15money expressions, and any other non-whitespace sequences: 

16 

17 >>> from nltk.tokenize import RegexpTokenizer 

18 >>> s = "Good muffins cost $3.88\nin New York. Please buy me\ntwo of them.\n\nThanks." 

19 >>> tokenizer = RegexpTokenizer('\w+|\$[\d\.]+|\S+') 

20 >>> tokenizer.tokenize(s) 

21 ['Good', 'muffins', 'cost', '$3.88', 'in', 'New', 'York', '.', 

22 'Please', 'buy', 'me', 'two', 'of', 'them', '.', 'Thanks', '.'] 

23 

24A ``RegexpTokenizer`` can use its regexp to match delimiters instead: 

25 

26 >>> tokenizer = RegexpTokenizer('\s+', gaps=True) 

27 >>> tokenizer.tokenize(s) 

28 ['Good', 'muffins', 'cost', '$3.88', 'in', 'New', 'York.', 

29 'Please', 'buy', 'me', 'two', 'of', 'them.', 'Thanks.'] 

30 

31Note that empty tokens are not returned when the delimiter appears at 

32the start or end of the string. 

33 

34The material between the tokens is discarded. For example, 

35the following tokenizer selects just the capitalized words: 

36 

37 >>> capword_tokenizer = RegexpTokenizer('[A-Z]\w+') 

38 >>> capword_tokenizer.tokenize(s) 

39 ['Good', 'New', 'York', 'Please', 'Thanks'] 

40 

41This module contains several subclasses of ``RegexpTokenizer`` 

42that use pre-defined regular expressions. 

43 

44 >>> from nltk.tokenize import BlanklineTokenizer 

45 >>> # Uses '\s*\n\s*\n\s*': 

46 >>> BlanklineTokenizer().tokenize(s) 

47 ['Good muffins cost $3.88\nin New York. Please buy me\ntwo of them.', 

48 'Thanks.'] 

49 

50All of the regular expression tokenizers are also available as functions: 

51 

52 >>> from nltk.tokenize import regexp_tokenize, wordpunct_tokenize, blankline_tokenize 

53 >>> regexp_tokenize(s, pattern='\w+|\$[\d\.]+|\S+') 

54 ['Good', 'muffins', 'cost', '$3.88', 'in', 'New', 'York', '.', 

55 'Please', 'buy', 'me', 'two', 'of', 'them', '.', 'Thanks', '.'] 

56 >>> wordpunct_tokenize(s) 

57 ['Good', 'muffins', 'cost', '$', '3', '.', '88', 'in', 'New', 'York', 

58 '.', 'Please', 'buy', 'me', 'two', 'of', 'them', '.', 'Thanks', '.'] 

59 >>> blankline_tokenize(s) 

60 ['Good muffins cost $3.88\nin New York. Please buy me\ntwo of them.', 'Thanks.'] 

61 

62Caution: The function ``regexp_tokenize()`` takes the text as its 

63first argument, and the regular expression pattern as its second 

64argument. This differs from the conventions used by Python's 

65``re`` functions, where the pattern is always the first argument. 

66(This is for consistency with the other NLTK tokenizers.) 

67"""

77@python_2_unicode_compatible 

78class RegexpTokenizer(TokenizerI): 

79 """ 

80 A tokenizer that splits a string using a regular expression, which 

81 matches either the tokens or the separators between tokens. 

82 

83 >>> tokenizer = RegexpTokenizer('\w+|\$[\d\.]+|\S+') 

84 

85 :type pattern: str 

86 :param pattern: The pattern used to build this tokenizer. 

87 (This pattern must not contain capturing parentheses; 

88 Use non-capturing parentheses, e.g. (?:...), instead) 

89 :type gaps: bool 

90 :param gaps: True if this tokenizer's pattern should be used 

91 to find separators between tokens; False if this 

92 tokenizer's pattern should be used to find the tokens 

93 themselves. 

94 :type discard_empty: bool 

95 :param discard_empty: True if any empty tokens `''` 

96 generated by the tokenizer should be discarded. Empty 

97 tokens can only be generated if `_gaps == True`. 

98 :type flags: int 

99 :param flags: The regexp flags used to compile this 

100 tokenizer's pattern. By default, the following flags are 

101 used: `re.UNICODE | re.MULTILINE | re.DOTALL`. 

102 

103 """ 

104 

105 def __init__( 

106 self, 

107 pattern, 

108 gaps=False, 

109 discard_empty=True, 

110 flags=re.UNICODE | re.MULTILINE | re.DOTALL, 

111 ): 

112 # If they gave us a regexp object, extract the pattern. 

113 pattern = getattr(pattern, 'pattern', pattern) 

114 

115 self._pattern = pattern 

116 self._gaps = gaps 

117 self._discard_empty = discard_empty 

118 self._flags = flags 

119 self._regexp = None 

120 

121 def _check_regexp(self): 

122 if self._regexp is None: 

123 self._regexp = re.compile(self._pattern, self._flags) 

124 

125 def tokenize(self, text): 

126 self._check_regexp() 

127 # If our regexp matches gaps, use re.split: 

128 if self._gaps: 

129 if self._discard_empty: 

130 return [tok for tok in self._regexp.split(text) if tok] 

131 else: 

132 return self._regexp.split(text) 

133 

134 # If our regexp matches tokens, use re.findall: 

135 else: 

136 return self._regexp.findall(text) 

137 

138 def span_tokenize(self, text): 

139 self._check_regexp() 

140 

141 if self._gaps: 

142 for left, right in regexp_span_tokenize(text, self._regexp): 

143 if not (self._discard_empty and left == right): 

144 yield left, right 

145 else: 

146 for m in re.finditer(self._regexp, text): 

147 yield m.span() 

148 

149 def __repr__(self): 

150 return '%s(pattern=%r, gaps=%r, discard_empty=%r, flags=%r)' % ( 

151 self.__class__.__name__, 

152 self._pattern, 

153 self._gaps, 

154 self._discard_empty, 

155 self._flags, 

156 )

159class WhitespaceTokenizer(RegexpTokenizer): 

160 r""" 

161 Tokenize a string on whitespace (space, tab, newline). 

162 In general, users should use the string ``split()`` method instead. 

163 

164 >>> from nltk.tokenize import WhitespaceTokenizer 

165 >>> s = "Good muffins cost $3.88\nin New York. Please buy me\ntwo of them.\n\nThanks." 

166 >>> WhitespaceTokenizer().tokenize(s) 

167 ['Good', 'muffins', 'cost', '$3.88', 'in', 'New', 'York.', 

168 'Please', 'buy', 'me', 'two', 'of', 'them.', 'Thanks.'] 

169 """ 

170 

171 def __init__(self): 

172 RegexpTokenizer.__init__(self, r'\s+', gaps=True)

175class BlanklineTokenizer(RegexpTokenizer): 

176 """ 

177 Tokenize a string, treating any sequence of blank lines as a delimiter. 

178 Blank lines are defined as lines containing no characters, except for 

179 space or tab characters.

182 def __init__(self):
