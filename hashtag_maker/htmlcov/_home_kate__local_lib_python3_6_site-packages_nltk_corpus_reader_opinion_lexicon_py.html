51class OpinionLexiconCorpusReader(WordListCorpusReader): 

52 """ 

53 Reader for Liu and Hu opinion lexicon. Blank lines and readme are ignored. 

54 

55 >>> from nltk.corpus import opinion_lexicon 

56 >>> opinion_lexicon.words() 

57 ['2-faced', '2-faces', 'abnormal', 'abolish', ...] 

58 

59 The OpinionLexiconCorpusReader provides shortcuts to retrieve positive/negative 

60 words: 

61 

62 >>> opinion_lexicon.negative() 

63 ['2-faced', '2-faces', 'abnormal', 'abolish', ...] 

64 

65 Note that words from `words()` method are sorted by file id, not alphabetically: 

66 

67 >>> opinion_lexicon.words()[0:10] 

68 ['2-faced', '2-faces', 'abnormal', 'abolish', 'abominable', 'abominably', 

69 'abominate', 'abomination', 'abort', 'aborted'] 

70 >>> sorted(opinion_lexicon.words())[0:10] 

71 ['2-faced', '2-faces', 'a+', 'abnormal', 'abolish', 'abominable', 'abominably', 

72 'abominate', 'abomination', 'abort'] 

73 """ 

74 

75 CorpusView = IgnoreReadmeCorpusView

77 def words(self, fileids=None): 

78 """ 

79 Return all words in the opinion lexicon. Note that these words are not 

80 sorted in alphabetical order. 

81 

82 :param fileids: a list or regexp specifying the ids of the files whose 

83 words have to be returned. 

84 :return: the given file(s) as a list of words and punctuation symbols. 

85 :rtype: list(str) 

86 """ 

87 if fileids is None: 

88 fileids = self._fileids 

89 elif isinstance(fileids, string_types): 

90 fileids = [fileids] 

91 return concat( 

92 [ 

93 self.CorpusView(path, self._read_word_block, encoding=enc) 

94 for (path, enc, fileid) in self.abspaths(fileids, True, True) 

95 ] 

96 ) 

97 

98 def positive(self): 

99 """ 

100 Return all positive words in alphabetical order. 

101 

102 :return: a list of positive words. 

103 :rtype: list(str) 

104 """ 

105 return self.words('positive-words.txt')

107 def negative(self): 

108 """ 

109 Return all negative words in alphabetical order. 

110 

111 :return: a list of negative words. 

112 :rtype: list(str) 

113 """ 

114 return self.words('negative-words.txt')

116 def _read_word_block(self, stream): 

117 words = [] 

118 for i in range(20): # Read 20 lines at a time. 

119 line = stream.readline() 

120 if not line: 

121 continue 

122 words.append(line.strip()) 

123 return words
