15@add_metaclass(ABCMeta) 

16class ClusterI(object): 

17 """ 

18 Interface covering basic clustering functionality.

21 @abstractmethod 

22 def cluster(self, vectors, assign_clusters=False): 

23 """ 

24 Assigns the vectors to clusters, learning the clustering parameters 

25 from the data. Returns a cluster identifier for each vector. 

26 """ 

27 

28 @abstractmethod 

29 def classify(self, token): 

30 """ 

31 Classifies the token into a cluster, setting the token's CLUSTER 

32 parameter to that cluster identifier. 

33 """ 

34 

35 def likelihood(self, vector, label): 

36 """ 

37 Returns the likelihood (a float) of the token having the 

38 corresponding cluster. 

39 """ 

40 if self.classify(vector) == label: 

41 return 1.0 

42 else: 

43 return 0.0 

44 

45 def classification_probdist(self, vector): 

46 """ 

47 Classifies the token into a cluster, returning 

48 a probability distribution over the cluster identifiers. 

49 """ 

50 likelihoods = {} 

51 sum = 0.0 

52 for cluster in self.cluster_names(): 

53 likelihoods[cluster] = self.likelihood(vector, cluster) 

54 sum += likelihoods[cluster] 

55 for cluster in self.cluster_names(): 

56 likelihoods[cluster] /= sum 

57 return DictionaryProbDist(likelihoods)
