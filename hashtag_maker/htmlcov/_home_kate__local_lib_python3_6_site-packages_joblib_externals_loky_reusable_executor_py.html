27def _get_next_executor_id(): 

28 """Ensure that each successive executor instance has a unique, monotonic id. 

29 

30 The purpose of this monotonic id is to help debug and test automated 

31 instance creation. 

32 """ 

33 global _next_executor_id 

34 with _executor_lock: 

35 executor_id = _next_executor_id 

36 _next_executor_id += 1 

37 return executor_id

40def get_reusable_executor(max_workers=None, context=None, timeout=10, 

41 kill_workers=False, reuse="auto", 

42 job_reducers=None, result_reducers=None, 

43 initializer=None, initargs=(), env=None): 

44 """Return the current ReusableExectutor instance. 

45 

46 Start a new instance if it has not been started already or if the previous 

47 instance was left in a broken state. 

48 

49 If the previous instance does not have the requested number of workers, the 

50 executor is dynamically resized to adjust the number of workers prior to 

51 returning. 

52 

53 Reusing a singleton instance spares the overhead of starting new worker 

54 processes and importing common python packages each time. 

55 

56 ``max_workers`` controls the maximum number of tasks that can be running in 

57 parallel in worker processes. By default this is set to the number of 

58 CPUs on the host. 

59 

60 Setting ``timeout`` (in seconds) makes idle workers automatically shutdown 

61 so as to release system resources. New workers are respawn upon submission 

62 of new tasks so that ``max_workers`` are available to accept the newly 

63 submitted tasks. Setting ``timeout`` to around 100 times the time required 

64 to spawn new processes and import packages in them (on the order of 100ms) 

65 ensures that the overhead of spawning workers is negligible. 

66 

67 Setting ``kill_workers=True`` makes it possible to forcibly interrupt 

68 previously spawned jobs to get a new instance of the reusable executor 

69 with new constructor argument values. 

70 

71 The ``job_reducers`` and ``result_reducers`` are used to customize the 

72 pickling of tasks and results send to the executor. 

73 

74 When provided, the ``initializer`` is run first in newly spawned 

75 processes with argument ``initargs``. 

76 

77 The environment variable in the child process are a copy of the values in 

78 the main process. One can provide a dict ``{ENV: VAL}`` where ``ENV`` and 

79 ``VAR`` are string literals to overwrite the environment variable ``ENV`` 

80 in the child processes to value ``VAL``. The environment variables are set 

81 in the children before any module is loaded. This only works with with the 

82 ``loky`` context and it is unreliable on Windows with Python < 3.6. 

83 """ 

84 with _executor_lock: 

85 global _executor, _executor_kwargs 

86 executor = _executor 

87 

88 if max_workers is None: 

89 if reuse is True and executor is not None: 

90 max_workers = executor._max_workers 

91 else: 

92 max_workers = cpu_count() 

93 elif max_workers <= 0: 

94 raise ValueError( 

95 "max_workers must be greater than 0, got {}." 

96 .format(max_workers)) 

97 

98 if isinstance(context, STRING_TYPE): 

99 context = get_context(context) 

100 if context is not None and context.get_start_method() == "fork": 

101 raise ValueError("Cannot use reusable executor with the 'fork' " 

102 "context") 

103 

104 kwargs = dict(context=context, timeout=timeout, 

105 job_reducers=job_reducers, 

106 result_reducers=result_reducers, 

107 initializer=initializer, initargs=initargs, 

108 env=env) 

109 if executor is None: 

110 mp.util.debug("Create a executor with max_workers={}." 

111 .format(max_workers)) 

112 executor_id = _get_next_executor_id() 

113 _executor_kwargs = kwargs 

114 _executor = executor = _ReusablePoolExecutor( 

115 _executor_lock, max_workers=max_workers, 

116 executor_id=executor_id, **kwargs) 

117 else: 

118 if reuse == 'auto': 

119 reuse = kwargs == _executor_kwargs 

120 if (executor._flags.broken or executor._flags.shutdown 

121 or not reuse): 

122 if executor._flags.broken: 

123 reason = "broken" 

124 elif executor._flags.shutdown: 

125 reason = "shutdown" 

126 else: 

127 reason = "arguments have changed" 

128 mp.util.debug( 

129 "Creating a new executor with max_workers={} as the " 

130 "previous instance cannot be reused ({})." 

131 .format(max_workers, reason)) 

132 executor.shutdown(wait=True, kill_workers=kill_workers) 

133 _executor = executor = _executor_kwargs = None 

134 # Recursive call to build a new instance 

135 return get_reusable_executor(max_workers=max_workers, 

136 **kwargs) 

137 else: 

138 mp.util.debug("Reusing existing executor with max_workers={}." 

139 .format(executor._max_workers)) 

140 executor._resize(max_workers) 

141 

142 return executor

162 def _resize(self, max_workers): 

163 with self._submit_resize_lock: 

164 if max_workers is None: 

165 raise ValueError("Trying to resize with max_workers=None") 

166 elif max_workers == self._max_workers: 

167 return 

168 

169 if self._queue_management_thread is None: 

170 # If the queue_management_thread has not been started 

171 # then no processes have been spawned and we can just 

172 # update _max_workers and return 

173 self._max_workers = max_workers 

174 return 

175 

176 self._wait_job_completion() 

177 

178 # Some process might have returned due to timeout so check how many 

179 # children are still alive. Use the _process_management_lock to 

180 # ensure that no process are spawned or timeout during the resize. 

181 with self._processes_management_lock: 

182 processes = list(self._processes.values()) 

183 nb_children_alive = sum(p.is_alive() for p in processes) 

184 self._max_workers = max_workers 

185 for _ in range(max_workers, nb_children_alive): 

186 self._call_queue.put(None) 

187 while (len(self._processes) > max_workers 

188 and not self._flags.broken): 

189 time.sleep(1e-3) 

190 

191 self._adjust_process_count() 

192 processes = list(self._processes.values()) 

193 while not all([p.is_alive() for p in processes]): 

194 time.sleep(1e-3)

196 def _wait_job_completion(self): 

197 """Wait for the cache to be empty before resizing the pool.""" 

198 # Issue a warning to the user about the bad effect of this usage. 

199 if len(self._pending_work_items) > 0: 

200 warnings.warn("Trying to resize an executor with running jobs: " 

201 "waiting for jobs completion before resizing.", 

202 UserWarning) 

203 mp.util.debug("Executor {} waiting for jobs completion before" 

204 " resizing".format(self.executor_id)) 

205 # Wait for the completion of the jobs 

206 while len(self._pending_work_items) > 0: 

207 time.sleep(1e-3)
