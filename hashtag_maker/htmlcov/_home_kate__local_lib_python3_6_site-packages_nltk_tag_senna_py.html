46@python_2_unicode_compatible 

47class SennaTagger(Senna): 

48 def __init__(self, path, encoding='utf-8'): 

49 super(SennaTagger, self).__init__(path, ['pos'], encoding) 

50 

51 def tag_sents(self, sentences): 

52 """ 

53 Applies the tag method over a list of sentences. This method will return 

54 for each sentence a list of tuples of (word, tag). 

55 """ 

56 tagged_sents = super(SennaTagger, self).tag_sents(sentences) 

57 for i in range(len(tagged_sents)): 

58 for j in range(len(tagged_sents[i])): 

59 annotations = tagged_sents[i][j] 

60 tagged_sents[i][j] = (annotations['word'], annotations['pos']) 

61 return tagged_sents

81 def bio_to_chunks(self, tagged_sent, chunk_type): 

82 """ 

83 Extracts the chunks in a BIO chunk-tagged sentence. 

84 

85 >>> from nltk.tag import SennaChunkTagger 

86 >>> chktagger = SennaChunkTagger('/usr/share/senna-v3.0') 

87 >>> sent = 'What is the airspeed of an unladen swallow ?'.split() 

88 >>> tagged_sent = chktagger.tag(sent) # doctest: +SKIP 

89 >>> tagged_sent # doctest: +SKIP 

90 [('What', 'B-NP'), ('is', 'B-VP'), ('the', 'B-NP'), ('airspeed', 'I-NP'), 

91 ('of', 'B-PP'), ('an', 'B-NP'), ('unladen', 'I-NP'), ('swallow', 'I-NP'), 

92 ('?', 'O')] 

93 >>> list(chktagger.bio_to_chunks(tagged_sent, chunk_type='NP')) # doctest: +SKIP 

94 [('What', '0'), ('the airspeed', '2-3'), ('an unladen swallow', '5-6-7')] 

95 

96 :param tagged_sent: A list of tuples of word and BIO chunk tag. 

97 :type tagged_sent: list(tuple) 

98 :param tagged_sent: The chunk tag that users want to extract, e.g. 'NP' or 'VP' 

99 :type tagged_sent: str 

100 

101 :return: An iterable of tuples of chunks that users want to extract 

102 and their corresponding indices. 

103 :rtype: iter(tuple(str)) 

104 """ 

105 current_chunk = [] 

106 current_chunk_position = [] 

107 for idx, word_pos in enumerate(tagged_sent): 

108 word, pos = word_pos 

109 if '-' + chunk_type in pos: # Append the word to the current_chunk. 

110 current_chunk.append((word)) 

111 current_chunk_position.append((idx)) 

112 else: 

113 if current_chunk: # Flush the full chunk when out of an NP. 

114 _chunk_str = ' '.join(current_chunk) 

115 _chunk_pos_str = '-'.join(map(str, current_chunk_position)) 

116 yield _chunk_str, _chunk_pos_str 

117 current_chunk = [] 

118 current_chunk_position = [] 

119 if current_chunk: # Flush the last chunk. 

120 yield ' '.join(current_chunk), '-'.join(map(str, current_chunk_position))
