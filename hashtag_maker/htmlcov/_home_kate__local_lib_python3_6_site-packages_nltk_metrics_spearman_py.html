14def _rank_dists(ranks1, ranks2): 

15 """Finds the difference between the values in ranks1 and ranks2 for keys 

16 present in both dicts. If the arguments are not dicts, they are converted 

17 from (key, rank) sequences. 

18 """ 

19 ranks1 = dict(ranks1) 

20 ranks2 = dict(ranks2) 

21 for k in ranks1: 

22 try: 

23 yield k, ranks1[k] - ranks2[k] 

24 except KeyError: 

25 pass

28def spearman_correlation(ranks1, ranks2): 

29 """Returns the Spearman correlation coefficient for two rankings, which 

30 should be dicts or sequences of (key, rank). The coefficient ranges from 

31 -1.0 (ranks are opposite) to 1.0 (ranks are identical), and is only 

32 calculated for keys in both rankings (for meaningful results, remove keys 

33 present in only one list before ranking).""" 

34 n = 0 

35 res = 0 

36 for k, d in _rank_dists(ranks1, ranks2): 

37 res += d * d 

38 n += 1 

39 try: 

40 return 1 - (6 * res / (n * (n * n - 1))) 

41 except ZeroDivisionError: 

42 # Result is undefined if only one item is ranked 

43 return 0.0

46def ranks_from_sequence(seq): 

47 """Given a sequence, yields each element with an increasing rank, suitable 

48 for use as an argument to ``spearman_correlation``. 

49 """ 

50 return ((k, i) for i, k in enumerate(seq))

53def ranks_from_scores(scores, rank_gap=1e-15): 

54 """Given a sequence of (key, score) tuples, yields each key with an 

55 increasing rank, tying with previous key's rank if the difference between 

56 their scores is less than rank_gap. Suitable for use as an argument to 

57 ``spearman_correlation``. 

58 """ 

59 prev_score = None 

60 rank = 0 

61 for i, (key, score) in enumerate(scores): 

62 try: 

63 if abs(score - prev_score) > rank_gap: 

64 rank = i 

65 except TypeError: 

66 pass 

67 

68 yield key, rank 

69 prev_score = score
