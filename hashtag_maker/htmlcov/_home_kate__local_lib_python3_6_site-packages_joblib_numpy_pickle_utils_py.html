52def _get_prefixes_max_len(): 

53 # Compute the max prefix len of registered compressors. 

54 prefixes = [len(compressor.prefix) for compressor in _COMPRESSORS.values()] 

55 prefixes += [len(_ZFILE_PREFIX)] 

56 return max(prefixes)

61def _detect_compressor(fileobj): 

62 """Return the compressor matching fileobj. 

63 

64 Parameters 

65 ---------- 

66 fileobj: file object 

67 

68 Returns 

69 ------- 

70 str in {'zlib', 'gzip', 'bz2', 'lzma', 'xz', 'compat', 'not-compressed'} 

71 """ 

72 # Read the magic number in the first bytes of the file. 

73 max_prefix_len = _get_prefixes_max_len() 

74 if hasattr(fileobj, 'peek'): 

75 # Peek allows to read those bytes without moving the cursor in the 

76 # file whic. 

77 first_bytes = fileobj.peek(max_prefix_len) 

78 else: 

79 # Fallback to seek if the fileobject is not peekable. 

80 first_bytes = fileobj.read(max_prefix_len) 

81 fileobj.seek(0) 

82 

83 if first_bytes.startswith(_ZFILE_PREFIX): 

84 return "compat" 

85 else: 

86 for name, compressor in _COMPRESSORS.items(): 

87 if first_bytes.startswith(compressor.prefix): 

88 return name 

89 

90 return "not-compressed"

93def _buffered_read_file(fobj): 

94 """Return a buffered version of a read file object.""" 

95 if PY27 and bz2 is not None and isinstance(fobj, bz2.BZ2File): 

96 # Python 2.7 doesn't work with BZ2File through a buffer: "no 

97 # attribute 'readable'" error. 

98 return fobj 

99 else: 

100 return io.BufferedReader(fobj, buffer_size=_IO_BUFFER_SIZE)

103def _buffered_write_file(fobj): 

104 """Return a buffered version of a write file object.""" 

105 if PY27 and bz2 is not None and isinstance(fobj, bz2.BZ2File): 

106 # Python 2.7 doesn't work with BZ2File through a buffer: no attribute 

107 # 'writable'. 

108 # BZ2File doesn't implement the file object context manager in python 2 

109 # so we wrap the fileobj using `closing`. 

110 return closing(fobj) 

111 else: 

112 return io.BufferedWriter(fobj, buffer_size=_IO_BUFFER_SIZE)

115@contextlib.contextmanager 

116def _read_fileobject(fileobj, filename, mmap_mode=None): 

117 """Utility function opening the right fileobject from a filename. 

118 

119 The magic number is used to choose between the type of file object to open: 

120 * regular file object (default) 

121 * zlib file object 

122 * gzip file object 

123 * bz2 file object 

124 * lzma file object (for xz and lzma compressor) 

125 

126 Parameters 

127 ---------- 

128 fileobj: file object 

129 compressor: str in {'zlib', 'gzip', 'bz2', 'lzma', 'xz', 'compat', 

130 'not-compressed'} 

131 filename: str 

132 filename path corresponding to the fileobj parameter. 

133 mmap_mode: str 

134 memory map mode that should be used to open the pickle file. This 

135 parameter is useful to verify that the user is not trying to one with 

136 compression. Default: None. 

137 

138 Returns 

139 ------- 

140 a file like object 

141 

142 """ 

143 # Detect if the fileobj contains compressed data. 

144 compressor = _detect_compressor(fileobj) 

145 

146 if compressor == 'compat': 

147 # Compatibility with old pickle mode: simply return the input 

148 # filename "as-is" and let the compatibility function be called by the 

149 # caller. 

150 warnings.warn("The file '%s' has been generated with a joblib " 

151 "version less than 0.10. " 

152 "Please regenerate this pickle file." % filename, 

153 DeprecationWarning, stacklevel=2) 

154 yield filename 

155 else: 

156 if compressor in _COMPRESSORS: 

157 # based on the compressor detected in the file, we open the 

158 # correct decompressor file object, wrapped in a buffer. 

159 compressor_wrapper = _COMPRESSORS[compressor] 

160 inst = compressor_wrapper.decompressor_file(fileobj) 

161 fileobj = _buffered_read_file(inst) 

162 

163 # Checking if incompatible load parameters with the type of file: 

164 # mmap_mode cannot be used with compressed file or in memory buffers 

165 # such as io.BytesIO. 

166 if mmap_mode is not None: 

167 if isinstance(fileobj, io.BytesIO): 

168 warnings.warn('In memory persistence is not compatible with ' 

169 'mmap_mode "%(mmap_mode)s" flag passed. ' 

170 'mmap_mode option will be ignored.' 

171 % locals(), stacklevel=2) 

172 elif compressor != 'not-compressed': 

173 warnings.warn('mmap_mode "%(mmap_mode)s" is not compatible ' 

174 'with compressed file %(filename)s. ' 

175 '"%(mmap_mode)s" flag will be ignored.' 

176 % locals(), stacklevel=2) 

177 elif not _is_raw_file(fileobj): 

178 warnings.warn('"%(fileobj)r" is not a raw file, mmap_mode ' 

179 '"%(mmap_mode)s" flag will be ignored.' 

180 % locals(), stacklevel=2) 

181 

182 yield fileobj

185def _write_fileobject(filename, compress=("zlib", 3)): 

186 """Return the right compressor file object in write mode.""" 

187 compressmethod = compress[0] 

188 compresslevel = compress[1] 

189 

190 if compressmethod in _COMPRESSORS.keys(): 

191 file_instance = _COMPRESSORS[compressmethod].compressor_file( 

192 filename, compresslevel=compresslevel) 

193 return _buffered_write_file(file_instance) 

194 else: 

195 file_instance = _COMPRESSORS['zlib'].compressor_file( 

196 filename, compresslevel=compresslevel) 

197 return _buffered_write_file(file_instance)

206def _read_bytes(fp, size, error_template="ran out of data"): 

207 """Read from file-like object until size bytes are read. 

208 

209 Raises ValueError if not EOF is encountered before size bytes are read. 

210 Non-blocking objects only supported if they derive from io objects. 

211 

212 Required as e.g. ZipExtFile in python 2.6 can return less data than 

213 requested. 

214 

215 This function was taken from numpy/lib/format.py in version 1.10.2. 

216 

217 Parameters 

218 ---------- 

219 fp: file-like object 

220 size: int 

221 error_template: str 

222 

223 Returns 

224 ------- 

225 a bytes object 

226 The data read in bytes. 

227 

228 """ 

229 data = bytes() 

230 while True: 

231 # io files (default in python3) return None or raise on 

232 # would-block, python2 file will truncate, probably nothing can be 

233 # done about that. note that regular files can't be non-blocking 

234 try: 

235 r = fp.read(size - len(data)) 

236 data += r 

237 if len(r) == 0 or len(data) == size: 

238 break 

239 except io.BlockingIOError: 

240 pass 

241 if len(data) != size: 

242 msg = "EOF: reading %s, expected %d bytes got %d" 

243 raise ValueError(msg % (error_template, size, len(data))) 

244 else: 

245 return data
