28class FeatureHasher(BaseEstimator, TransformerMixin): 

29 """Implements feature hashing, aka the hashing trick. 

30 

31 This class turns sequences of symbolic feature names (strings) into 

32 scipy.sparse matrices, using a hash function to compute the matrix column 

33 corresponding to a name. The hash function employed is the signed 32-bit 

34 version of Murmurhash3. 

35 

36 Feature names of type byte string are used as-is. Unicode strings are 

37 converted to UTF-8 first, but no Unicode normalization is done. 

38 Feature values must be (finite) numbers. 

39 

40 This class is a low-memory alternative to DictVectorizer and 

41 CountVectorizer, intended for large-scale (online) learning and situations 

42 where memory is tight, e.g. when running prediction code on embedded 

43 devices. 

44 

45 Read more in the :ref:`User Guide <feature_hashing>`. 

46 

47 Parameters 

48 ---------- 

49 n_features : integer, optional 

50 The number of features (columns) in the output matrices. Small numbers 

51 of features are likely to cause hash collisions, but large numbers 

52 will cause larger coefficient dimensions in linear learners. 

53 input_type : string, optional, default "dict" 

54 Either "dict" (the default) to accept dictionaries over 

55 (feature_name, value); "pair" to accept pairs of (feature_name, value); 

56 or "string" to accept single strings. 

57 feature_name should be a string, while value should be a number. 

58 In the case of "string", a value of 1 is implied. 

59 The feature_name is hashed to find the appropriate column for the 

60 feature. The value's sign might be flipped in the output (but see 

61 non_negative, below). 

62 dtype : numpy type, optional, default np.float64 

63 The type of feature values. Passed to scipy.sparse matrix constructors 

64 as the dtype argument. Do not set this to bool, np.boolean or any 

65 unsigned integer type. 

66 alternate_sign : boolean, optional, default True 

67 When True, an alternating sign is added to the features as to 

68 approximately conserve the inner product in the hashed space even for 

69 small n_features. This approach is similar to sparse random projection. 

70 

71 Examples 

72 -------- 

73 >>> from sklearn.feature_extraction import FeatureHasher 

74 >>> h = FeatureHasher(n_features=10) 

75 >>> D = [{'dog': 1, 'cat':2, 'elephant':4},{'dog': 2, 'run': 5}] 

76 >>> f = h.transform(D) 

77 >>> f.toarray() 

78 array([[ 0., 0., -4., -1., 0., 0., 0., 0., 0., 2.], 

79 [ 0., 0., 0., -2., -5., 0., 0., 0., 0., 0.]]) 

80 

81 See also 

82 -------- 

83 DictVectorizer : vectorizes string-valued features using a hash table. 

84 sklearn.preprocessing.OneHotEncoder : handles nominal/categorical features. 

85 """

129 def transform(self, raw_X): 

130 """Transform a sequence of instances to a scipy.sparse matrix. 

131 

132 Parameters 

133 ---------- 

134 raw_X : iterable over iterable over raw features, length = n_samples 

135 Samples. Each sample must be iterable an (e.g., a list or tuple) 

136 containing/generating feature names (and optionally values, see 

137 the input_type constructor argument) which will be hashed. 

138 raw_X need not support the len function, so it can be the result 

139 of a generator; n_samples is determined on the fly. 

140 

141 Returns 

142 ------- 

143 X : scipy.sparse matrix, shape = (n_samples, self.n_features) 

144 Feature matrix, for use with estimators or further transformers. 

145 

146 """ 

147 raw_X = iter(raw_X) 

148 if self.input_type == "dict": 

149 raw_X = (_iteritems(d) for d in raw_X) 

150 elif self.input_type == "string": 

151 raw_X = (((f, 1) for f in x) for x in raw_X) 

152 indices, indptr, values = \ 

153 _hashing_transform(raw_X, self.n_features, self.dtype, 

154 self.alternate_sign) 

155 n_samples = indptr.shape[0] - 1 

156 

157 if n_samples == 0: 

158 raise ValueError("Cannot vectorize empty sequence.") 

159 

160 X = sp.csr_matrix((values, indices, indptr), dtype=self.dtype, 

161 shape=(n_samples, self.n_features)) 

162 X.sum_duplicates() # also sorts the indices 

163 

164 return X
