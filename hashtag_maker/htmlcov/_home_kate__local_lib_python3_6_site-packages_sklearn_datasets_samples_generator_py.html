23def _generate_hypercube(samples, dimensions, rng): 

24 """Returns distinct binary samples of length dimensions 

25 """ 

26 if dimensions > 30: 

27 return np.hstack([rng.randint(2, size=(samples, dimensions - 30)), 

28 _generate_hypercube(samples, 30, rng)]) 

29 out = sample_without_replacement(2 ** dimensions, samples, 

30 random_state=rng).astype(dtype='>u4', 

31 copy=False) 

32 out = np.unpackbits(out.view('>u1')).reshape((-1, 32))[:, -dimensions:] 

33 return out

36def make_classification(n_samples=100, n_features=20, n_informative=2, 

37 n_redundant=2, n_repeated=0, n_classes=2, 

38 n_clusters_per_class=2, weights=None, flip_y=0.01, 

39 class_sep=1.0, hypercube=True, shift=0.0, scale=1.0, 

40 shuffle=True, random_state=None): 

41 """Generate a random n-class classification problem. 

42 

43 This initially creates clusters of points normally distributed (std=1) 

44 about vertices of an ``n_informative``-dimensional hypercube with sides of 

45 length ``2*class_sep`` and assigns an equal number of clusters to each 

46 class. It introduces interdependence between these features and adds 

47 various types of further noise to the data. 

48 

49 Without shuffling, ``X`` horizontally stacks features in the following 

50 order: the primary ``n_informative`` features, followed by ``n_redundant`` 

51 linear combinations of the informative features, followed by ``n_repeated`` 

52 duplicates, drawn randomly with replacement from the informative and 

53 redundant features. The remaining features are filled with random noise. 

54 Thus, without shuffling, all useful features are contained in the columns 

55 ``X[:, :n_informative + n_redundant + n_repeated]``. 

56 

57 Read more in the :ref:`User Guide <sample_generators>`. 

58 

59 Parameters 

60 ---------- 

61 n_samples : int, optional (default=100) 

62 The number of samples. 

63 

64 n_features : int, optional (default=20) 

65 The total number of features. These comprise ``n_informative`` 

66 informative features, ``n_redundant`` redundant features, 

67 ``n_repeated`` duplicated features and 

68 ``n_features-n_informative-n_redundant-n_repeated`` useless features 

69 drawn at random. 

70 

71 n_informative : int, optional (default=2) 

72 The number of informative features. Each class is composed of a number 

73 of gaussian clusters each located around the vertices of a hypercube 

74 in a subspace of dimension ``n_informative``. For each cluster, 

75 informative features are drawn independently from N(0, 1) and then 

76 randomly linearly combined within each cluster in order to add 

77 covariance. The clusters are then placed on the vertices of the 

78 hypercube. 

79 

80 n_redundant : int, optional (default=2) 

81 The number of redundant features. These features are generated as 

82 random linear combinations of the informative features. 

83 

84 n_repeated : int, optional (default=0) 

85 The number of duplicated features, drawn randomly from the informative 

86 and the redundant features. 

87 

88 n_classes : int, optional (default=2) 

89 The number of classes (or labels) of the classification problem. 

90 

91 n_clusters_per_class : int, optional (default=2) 

92 The number of clusters per class. 

93 

94 weights : list of floats or None (default=None) 

95 The proportions of samples assigned to each class. If None, then 

96 classes are balanced. Note that if ``len(weights) == n_classes - 1``, 

97 then the last class weight is automatically inferred. 

98 More than ``n_samples`` samples may be returned if the sum of 

99 ``weights`` exceeds 1. 

100 

101 flip_y : float, optional (default=0.01) 

102 The fraction of samples whose class are randomly exchanged. Larger 

103 values introduce noise in the labels and make the classification 

104 task harder. 

105 

106 class_sep : float, optional (default=1.0) 

107 The factor multiplying the hypercube size. Larger values spread 

108 out the clusters/classes and make the classification task easier. 

109 

110 hypercube : boolean, optional (default=True) 

111 If True, the clusters are put on the vertices of a hypercube. If 

112 False, the clusters are put on the vertices of a random polytope. 

113 

114 shift : float, array of shape [n_features] or None, optional (default=0.0) 

115 Shift features by the specified value. If None, then features 

116 are shifted by a random value drawn in [-class_sep, class_sep]. 

117 

118 scale : float, array of shape [n_features] or None, optional (default=1.0) 

119 Multiply features by the specified value. If None, then features 

120 are scaled by a random value drawn in [1, 100]. Note that scaling 

121 happens after shifting. 

122 

123 shuffle : boolean, optional (default=True) 

124 Shuffle the samples and the features. 

125 

126 random_state : int, RandomState instance or None (default) 

127 Determines random number generation for dataset creation. Pass an int 

128 for reproducible output across multiple function calls. 

129 See :term:`Glossary <random_state>`. 

130 

131 Returns 

132 ------- 

133 X : array of shape [n_samples, n_features] 

134 The generated samples. 

135 

136 y : array of shape [n_samples] 

137 The integer labels for class membership of each sample. 

138 

139 Notes 

140 ----- 

141 The algorithm is adapted from Guyon [1] and was designed to generate 

142 the "Madelon" dataset. 

143 

144 References 

145 ---------- 

146 .. [1] I. Guyon, "Design of experiments for the NIPS 2003 variable 

147 selection benchmark", 2003. 

148 

149 See also 

150 -------- 

151 make_blobs: simplified variant 

152 make_multilabel_classification: unrelated generator for multilabel tasks 

153 """ 

154 generator = check_random_state(random_state) 

155 

156 # Count features, clusters and samples 

157 if n_informative + n_redundant + n_repeated > n_features: 

158 raise ValueError("Number of informative, redundant and repeated " 

159 "features must sum to less than the number of total" 

160 " features") 

161 # Use log2 to avoid overflow errors 

162 if n_informative < np.log2(n_classes * n_clusters_per_class): 

163 raise ValueError("n_classes * n_clusters_per_class must" 

164 " be smaller or equal 2 ** n_informative") 

165 if weights and len(weights) not in [n_classes, n_classes - 1]: 

166 raise ValueError("Weights specified but incompatible with number " 

167 "of classes.") 

168 

169 n_useless = n_features - n_informative - n_redundant - n_repeated 

170 n_clusters = n_classes * n_clusters_per_class 

171 

172 if weights and len(weights) == (n_classes - 1): 

173 weights = weights + [1.0 - sum(weights)] 

174 

175 if weights is None: 

176 weights = [1.0 / n_classes] * n_classes 

177 weights[-1] = 1.0 - sum(weights[:-1]) 

178 

179 # Distribute samples among clusters by weight 

180 n_samples_per_cluster = [ 

181 int(n_samples * weights[k % n_classes] / n_clusters_per_class) 

182 for k in range(n_clusters)] 

183 

184 for i in range(n_samples - sum(n_samples_per_cluster)): 

185 n_samples_per_cluster[i % n_clusters] += 1 

186 

187 # Initialize X and y 

188 X = np.zeros((n_samples, n_features)) 

189 y = np.zeros(n_samples, dtype=np.int) 

190 

191 # Build the polytope whose vertices become cluster centroids 

192 centroids = _generate_hypercube(n_clusters, n_informative, 

193 generator).astype(float, copy=False) 

194 centroids *= 2 * class_sep 

195 centroids -= class_sep 

196 if not hypercube: 

197 centroids *= generator.rand(n_clusters, 1) 

198 centroids *= generator.rand(1, n_informative) 

199 

200 # Initially draw informative features from the standard normal 

201 X[:, :n_informative] = generator.randn(n_samples, n_informative) 

202 

203 # Create each cluster; a variant of make_blobs 

204 stop = 0 

205 for k, centroid in enumerate(centroids): 

206 start, stop = stop, stop + n_samples_per_cluster[k] 

207 y[start:stop] = k % n_classes # assign labels 

208 X_k = X[start:stop, :n_informative] # slice a view of the cluster 

209 

210 A = 2 * generator.rand(n_informative, n_informative) - 1 

211 X_k[...] = np.dot(X_k, A) # introduce random covariance 

212 

213 X_k += centroid # shift the cluster to a vertex 

214 

215 # Create redundant features 

216 if n_redundant > 0: 

217 B = 2 * generator.rand(n_informative, n_redundant) - 1 

218 X[:, n_informative:n_informative + n_redundant] = \ 

219 np.dot(X[:, :n_informative], B) 

220 

221 # Repeat some features 

222 if n_repeated > 0: 

223 n = n_informative + n_redundant 

224 indices = ((n - 1) * generator.rand(n_repeated) + 0.5).astype(np.intp) 

225 X[:, n:n + n_repeated] = X[:, indices] 

226 

227 # Fill useless features 

228 if n_useless > 0: 

229 X[:, -n_useless:] = generator.randn(n_samples, n_useless) 

230 

231 # Randomly replace labels 

232 if flip_y >= 0.0: 

233 flip_mask = generator.rand(n_samples) < flip_y 

234 y[flip_mask] = generator.randint(n_classes, size=flip_mask.sum()) 

235 

236 # Randomly shift and scale 

237 if shift is None: 

238 shift = (2 * generator.rand(n_features) - 1) * class_sep 

239 X += shift 

240 

241 if scale is None: 

242 scale = 1 + 100 * generator.rand(n_features) 

243 X *= scale 

244 

245 if shuffle: 

246 # Randomly permute samples 

247 X, y = util_shuffle(X, y, random_state=generator) 

248 

249 # Randomly permute features 

250 indices = np.arange(n_features) 

251 generator.shuffle(indices) 

252 X[:, :] = X[:, indices] 

253 

254 return X, y

257def make_multilabel_classification(n_samples=100, n_features=20, n_classes=5, 

258 n_labels=2, length=50, allow_unlabeled=True, 

259 sparse=False, return_indicator='dense', 

260 return_distributions=False, 

261 random_state=None): 

262 """Generate a random multilabel classification problem. 

263 

264 For each sample, the generative process is: 

265 - pick the number of labels: n ~ Poisson(n_labels) 

266 - n times, choose a class c: c ~ Multinomial(theta) 

267 - pick the document length: k ~ Poisson(length) 

268 - k times, choose a word: w ~ Multinomial(theta_c) 

269 

270 In the above process, rejection sampling is used to make sure that 

271 n is never zero or more than `n_classes`, and that the document length 

272 is never zero. Likewise, we reject classes which have already been chosen. 

273 

274 Read more in the :ref:`User Guide <sample_generators>`. 

275 

276 Parameters 

277 ---------- 

278 n_samples : int, optional (default=100) 

279 The number of samples. 

280 

281 n_features : int, optional (default=20) 

282 The total number of features. 

283 

284 n_classes : int, optional (default=5) 

285 The number of classes of the classification problem. 

286 

287 n_labels : int, optional (default=2) 

288 The average number of labels per instance. More precisely, the number 

289 of labels per sample is drawn from a Poisson distribution with 

290 ``n_labels`` as its expected value, but samples are bounded (using 

291 rejection sampling) by ``n_classes``, and must be nonzero if 

292 ``allow_unlabeled`` is False. 

293 

294 length : int, optional (default=50) 

295 The sum of the features (number of words if documents) is drawn from 

296 a Poisson distribution with this expected value. 

297 

298 allow_unlabeled : bool, optional (default=True) 

299 If ``True``, some instances might not belong to any class. 

300 

301 sparse : bool, optional (default=False) 

302 If ``True``, return a sparse feature matrix 

303 

304 .. versionadded:: 0.17 

305 parameter to allow *sparse* output. 

306 

307 return_indicator : 'dense' (default) | 'sparse' | False 

308 If ``dense`` return ``Y`` in the dense binary indicator format. If 

309 ``'sparse'`` return ``Y`` in the sparse binary indicator format. 

310 ``False`` returns a list of lists of labels. 

311 

312 return_distributions : bool, optional (default=False) 

313 If ``True``, return the prior class probability and conditional 

314 probabilities of features given classes, from which the data was 

315 drawn. 

316 

317 random_state : int, RandomState instance or None (default) 

318 Determines random number generation for dataset creation. Pass an int 

319 for reproducible output across multiple function calls. 

320 See :term:`Glossary <random_state>`. 

321 

322 Returns 

323 ------- 

324 X : array of shape [n_samples, n_features] 

325 The generated samples. 

326 

327 Y : array or sparse CSR matrix of shape [n_samples, n_classes] 

328 The label sets. 

329 

330 p_c : array, shape [n_classes] 

331 The probability of each class being drawn. Only returned if 

332 ``return_distributions=True``. 

333 

334 p_w_c : array, shape [n_features, n_classes] 

335 The probability of each feature being drawn given each class. 

336 Only returned if ``return_distributions=True``. 

337 

338 """ 

339 generator = check_random_state(random_state) 

340 p_c = generator.rand(n_classes) 

341 p_c /= p_c.sum() 

342 cumulative_p_c = np.cumsum(p_c) 

343 p_w_c = generator.rand(n_features, n_classes) 

344 p_w_c /= np.sum(p_w_c, axis=0) 

345 

346 def sample_example(): 

347 _, n_classes = p_w_c.shape 

348 

349 # pick a nonzero number of labels per document by rejection sampling 

350 y_size = n_classes + 1 

351 while (not allow_unlabeled and y_size == 0) or y_size > n_classes: 

352 y_size = generator.poisson(n_labels) 

353 

354 # pick n classes 

355 y = set() 

356 while len(y) != y_size: 

357 # pick a class with probability P(c) 

358 c = np.searchsorted(cumulative_p_c, 

359 generator.rand(y_size - len(y))) 

360 y.update(c) 

361 y = list(y) 

362 

363 # pick a non-zero document length by rejection sampling 

364 n_words = 0 

365 while n_words == 0: 

366 n_words = generator.poisson(length) 

367 

368 # generate a document of length n_words 

369 if len(y) == 0: 

370 # if sample does not belong to any class, generate noise word 

371 words = generator.randint(n_features, size=n_words) 

372 return words, y 

373 

374 # sample words with replacement from selected classes 

375 cumulative_p_w_sample = p_w_c.take(y, axis=1).sum(axis=1).cumsum() 

376 cumulative_p_w_sample /= cumulative_p_w_sample[-1] 

377 words = np.searchsorted(cumulative_p_w_sample, generator.rand(n_words)) 

378 return words, y 

379 

380 X_indices = array.array('i') 

381 X_indptr = array.array('i', [0]) 

382 Y = [] 

383 for i in range(n_samples): 

384 words, y = sample_example() 

385 X_indices.extend(words) 

386 X_indptr.append(len(X_indices)) 

387 Y.append(y) 

388 X_data = np.ones(len(X_indices), dtype=np.float64) 

389 X = sp.csr_matrix((X_data, X_indices, X_indptr), 

390 shape=(n_samples, n_features)) 

391 X.sum_duplicates() 

392 if not sparse: 

393 X = X.toarray() 

394 

395 # return_indicator can be True due to backward compatibility 

396 if return_indicator in (True, 'sparse', 'dense'): 

397 lb = MultiLabelBinarizer(sparse_output=(return_indicator == 'sparse')) 

398 Y = lb.fit([range(n_classes)]).transform(Y) 

399 elif return_indicator is not False: 

400 raise ValueError("return_indicator must be either 'sparse', 'dense' " 

401 'or False.') 

402 if return_distributions: 

403 return X, Y, p_c, p_w_c 

404 return X, Y

407def make_hastie_10_2(n_samples=12000, random_state=None): 

408 """Generates data for binary classification used in 

409 Hastie et al. 2009, Example 10.2. 

410 

411 The ten features are standard independent Gaussian and 

412 the target ``y`` is defined by:: 

413 

414 y[i] = 1 if np.sum(X[i] ** 2) > 9.34 else -1 

415 

416 Read more in the :ref:`User Guide <sample_generators>`. 

417 

418 Parameters 

419 ---------- 

420 n_samples : int, optional (default=12000) 

421 The number of samples. 

422 

423 random_state : int, RandomState instance or None (default) 

424 Determines random number generation for dataset creation. Pass an int 

425 for reproducible output across multiple function calls. 

426 See :term:`Glossary <random_state>`. 

427 

428 Returns 

429 ------- 

430 X : array of shape [n_samples, 10] 

431 The input samples. 

432 

433 y : array of shape [n_samples] 

434 The output values. 

435 

436 References 

437 ---------- 

438 .. [1] T. Hastie, R. Tibshirani and J. Friedman, "Elements of Statistical 

439 Learning Ed. 2", Springer, 2009. 

440 

441 See also 

442 -------- 

443 make_gaussian_quantiles: a generalization of this dataset approach 

444 """

446

455def make_regression(n_samples=100, n_features=100, n_informative=10, 

456 n_targets=1, bias=0.0, effective_rank=None, 

457 tail_strength=0.5, noise=0.0, shuffle=True, coef=False, 

458 random_state=None): 

459 """Generate a random regression problem. 

460 

461 The input set can either be well conditioned (by default) or have a low 

462 rank-fat tail singular profile. See :func:`make_low_rank_matrix` for 

463 more details. 

464 

465 The output is generated by applying a (potentially biased) random linear 

466 regression model with `n_informative` nonzero regressors to the previously 

467 generated input and some gaussian centered noise with some adjustable 

468 scale. 

469 

470 Read more in the :ref:`User Guide <sample_generators>`. 

471 

472 Parameters 

473 ---------- 

474 n_samples : int, optional (default=100) 

475 The number of samples. 

476 

477 n_features : int, optional (default=100) 

478 The number of features. 

479 

480 n_informative : int, optional (default=10) 

481 The number of informative features, i.e., the number of features used 

482 to build the linear model used to generate the output. 

483 

484 n_targets : int, optional (default=1) 

485 The number of regression targets, i.e., the dimension of the y output 

486 vector associated with a sample. By default, the output is a scalar. 

487 

488 bias : float, optional (default=0.0) 

489 The bias term in the underlying linear model. 

490 

491 effective_rank : int or None, optional (default=None) 

492 if not None: 

493 The approximate number of singular vectors required to explain most 

494 of the input data by linear combinations. Using this kind of 

495 singular spectrum in the input allows the generator to reproduce 

496 the correlations often observed in practice. 

497 if None: 

498 The input set is well conditioned, centered and gaussian with 

499 unit variance. 

500 

501 tail_strength : float between 0.0 and 1.0, optional (default=0.5) 

502 The relative importance of the fat noisy tail of the singular values 

503 profile if `effective_rank` is not None. 

504 

505 noise : float, optional (default=0.0) 

506 The standard deviation of the gaussian noise applied to the output. 

507 

508 shuffle : boolean, optional (default=True) 

509 Shuffle the samples and the features. 

510 

511 coef : boolean, optional (default=False) 

512 If True, the coefficients of the underlying linear model are returned. 

513 

514 random_state : int, RandomState instance or None (default) 

515 Determines random number generation for dataset creation. Pass an int 

516 for reproducible output across multiple function calls. 

517 See :term:`Glossary <random_state>`. 

518 

519 Returns 

520 ------- 

521 X : array of shape [n_samples, n_features] 

522 The input samples. 

523 

524 y : array of shape [n_samples] or [n_samples, n_targets] 

525 The output values. 

526 

527 coef : array of shape [n_features] or [n_features, n_targets], optional 

528 The coefficient of the underlying linear model. It is returned only if 

529 coef is True. 

530 """ 

531 n_informative = min(n_features, n_informative) 

532 generator = check_random_state(random_state) 

533 

534 if effective_rank is None: 

535 # Randomly generate a well conditioned input set 

536 X = generator.randn(n_samples, n_features) 

537 

538 else: 

539 # Randomly generate a low rank, fat tail input set 

540 X = make_low_rank_matrix(n_samples=n_samples, 

541 n_features=n_features, 

542 effective_rank=effective_rank, 

543 tail_strength=tail_strength, 

544 random_state=generator) 

545 

546 # Generate a ground truth model with only n_informative features being non 

547 # zeros (the other features are not correlated to y and should be ignored 

548 # by a sparsifying regularizers such as L1 or elastic net) 

549 ground_truth = np.zeros((n_features, n_targets)) 

550 ground_truth[:n_informative, :] = 100 * generator.rand(n_informative, 

551 n_targets) 

552 

553 y = np.dot(X, ground_truth) + bias 

554 

555 # Add noise 

556 if noise > 0.0: 

557 y += generator.normal(scale=noise, size=y.shape) 

558 

559 # Randomly permute samples and features 

560 if shuffle: 

561 X, y = util_shuffle(X, y, random_state=generator) 

562 

563 indices = np.arange(n_features) 

564 generator.shuffle(indices) 

565 X[:, :] = X[:, indices] 

566 ground_truth = ground_truth[indices] 

567 

568 y = np.squeeze(y) 

569 

570 if coef: 

571 return X, y, np.squeeze(ground_truth) 

572 

573 else: 

574 return X, y

577def make_circles(n_samples=100, shuffle=True, noise=None, random_state=None, 

578 factor=.8): 

579 """Make a large circle containing a smaller circle in 2d. 

580 

581 A simple toy dataset to visualize clustering and classification 

582 algorithms. 

583 

584 Read more in the :ref:`User Guide <sample_generators>`. 

585 

586 Parameters 

587 ---------- 

588 n_samples : int, optional (default=100) 

589 The total number of points generated. If odd, the inner circle will 

590 have one point more than the outer circle. 

591 

592 shuffle : bool, optional (default=True) 

593 Whether to shuffle the samples. 

594 

595 noise : double or None (default=None) 

596 Standard deviation of Gaussian noise added to the data. 

597 

598 random_state : int, RandomState instance or None (default) 

599 Determines random number generation for dataset shuffling and noise. 

600 Pass an int for reproducible output across multiple function calls. 

601 See :term:`Glossary <random_state>`. 

602 

603 factor : 0 < double < 1 (default=.8) 

604 Scale factor between inner and outer circle. 

605 

606 Returns 

607 ------- 

608 X : array of shape [n_samples, 2] 

609 The generated samples. 

610 

611 y : array of shape [n_samples] 

612 The integer labels (0 or 1) for class membership of each sample. 

613 """ 

614 

615 if factor >= 1 or factor < 0: 

616 raise ValueError("'factor' has to be between 0 and 1.") 

617 

618 n_samples_out = n_samples // 2 

619 n_samples_in = n_samples - n_samples_out 

620 

621 generator = check_random_state(random_state) 

622 # so as not to have the first point = last point, we set endpoint=False 

623 linspace_out = np.linspace(0, 2 * np.pi, n_samples_out, endpoint=False) 

624 linspace_in = np.linspace(0, 2 * np.pi, n_samples_in, endpoint=False) 

625 outer_circ_x = np.cos(linspace_out) 

626 outer_circ_y = np.sin(linspace_out) 

627 inner_circ_x = np.cos(linspace_in) * factor 

628 inner_circ_y = np.sin(linspace_in) * factor 

629

698def make_blobs(n_samples=100, n_features=2, centers=None, cluster_std=1.0, 

699 center_box=(-10.0, 10.0), shuffle=True, random_state=None): 

700 """Generate isotropic Gaussian blobs for clustering. 

701 

702 Read more in the :ref:`User Guide <sample_generators>`. 

703 

704 Parameters 

705 ---------- 

706 n_samples : int or array-like, optional (default=100) 

707 If int, it is the total number of points equally divided among 

708 clusters. 

709 If array-like, each element of the sequence indicates 

710 the number of samples per cluster. 

711 

712 n_features : int, optional (default=2) 

713 The number of features for each sample. 

714 

715 centers : int or array of shape [n_centers, n_features], optional 

716 (default=None) 

717 The number of centers to generate, or the fixed center locations. 

718 If n_samples is an int and centers is None, 3 centers are generated. 

719 If n_samples is array-like, centers must be 

720 either None or an array of length equal to the length of n_samples. 

721 

722 cluster_std : float or sequence of floats, optional (default=1.0) 

723 The standard deviation of the clusters. 

724 

725 center_box : pair of floats (min, max), optional (default=(-10.0, 10.0)) 

726 The bounding box for each cluster center when centers are 

727 generated at random. 

728 

729 shuffle : boolean, optional (default=True) 

730 Shuffle the samples. 

731 

732 random_state : int, RandomState instance or None (default) 

733 Determines random number generation for dataset creation. Pass an int 

734 for reproducible output across multiple function calls. 

735 See :term:`Glossary <random_state>`. 

736 

737 Returns 

738 ------- 

739 X : array of shape [n_samples, n_features] 

740 The generated samples. 

741 

742 y : array of shape [n_samples] 

743 The integer labels for cluster membership of each sample. 

744 

745 Examples 

746 -------- 

747 >>> from sklearn.datasets.samples_generator import make_blobs 

748 >>> X, y = make_blobs(n_samples=10, centers=3, n_features=2, 

749 ... random_state=0) 

750 >>> print(X.shape) 

751 (10, 2) 

752 >>> y 

753 array([0, 0, 1, 0, 2, 2, 2, 1, 1, 0]) 

754 >>> X, y = make_blobs(n_samples=[3, 3, 4], centers=None, n_features=2, 

755 ... random_state=0) 

756 >>> print(X.shape) 

757 (10, 2) 

758 >>> y 

759 array([0, 1, 2, 0, 2, 2, 2, 1, 1, 0]) 

760 

761 See also 

762 -------- 

763 make_classification: a more intricate variant 

764 """ 

765 generator = check_random_state(random_state) 

766 

767 if isinstance(n_samples, numbers.Integral): 

768 # Set n_centers by looking at centers arg 

769 if centers is None: 

770 centers = 3 

771 

772 if isinstance(centers, numbers.Integral): 

773 n_centers = centers 

774 centers = generator.uniform(center_box[0], center_box[1], 

775 size=(n_centers, n_features)) 

776 

777 else: 

778 centers = check_array(centers) 

779 n_features = centers.shape[1] 

780 n_centers = centers.shape[0] 

781 

782 else: 

783 # Set n_centers by looking at [n_samples] arg 

784 n_centers = len(n_samples) 

785 if centers is None: 

786 centers = generator.uniform(center_box[0], center_box[1], 

787 size=(n_centers, n_features)) 

788 try: 

789 assert len(centers) == n_centers 

790 except TypeError: 

791 raise ValueError("Parameter `centers` must be array-like. " 

792 "Got {!r} instead".format(centers)) 

793 except AssertionError: 

794 raise ValueError("Length of `n_samples` not consistent" 

795 " with number of centers. Got n_samples = {} " 

796 "and centers = {}".format(n_samples, centers)) 

797 else: 

798 centers = check_array(centers) 

799 n_features = centers.shape[1] 

800 

801 # stds: if cluster_std is given as list, it must be consistent 

802 # with the n_centers 

803 if (hasattr(cluster_std, "__len__") and len(cluster_std) != n_centers): 

804 raise ValueError("Length of `clusters_std` not consistent with " 

805 "number of centers. Got centers = {} " 

806 "and cluster_std = {}".format(centers, cluster_std)) 

807 

808 if isinstance(cluster_std, numbers.Real): 

809 cluster_std = np.full(len(centers), cluster_std) 

810 

811 X = [] 

812 y = [] 

813 

814 if isinstance(n_samples, Iterable): 

815 n_samples_per_center = n_samples 

816 else: 

817 n_samples_per_center = [int(n_samples // n_centers)] * n_centers 

818 

819 for i in range(n_samples % n_centers): 

820 n_samples_per_center[i] += 1 

821 

822 for i, (n, std) in enumerate(zip(n_samples_per_center, cluster_std)): 

823 X.append(generator.normal(loc=centers[i], scale=std, 

824 size=(n, n_features))) 

825 y += [i] * n 

826 

827 X = np.concatenate(X) 

828 y = np.array(y) 

829 

830 if shuffle: 

831 total_n_samples = np.sum(n_samples) 

832 indices = np.arange(total_n_samples) 

833 generator.shuffle(indices) 

834 X = X[indices] 

835 y = y[indices] 

836 

837 return X, y

1031def make_low_rank_matrix(n_samples=100, n_features=100, effective_rank=10, 

1032 tail_strength=0.5, random_state=None): 

1033 """Generate a mostly low rank matrix with bell-shaped singular values 

1034 

1035 Most of the variance can be explained by a bell-shaped curve of width 

1036 effective_rank: the low rank part of the singular values profile is:: 

1037 

1038 (1 - tail_strength) * exp(-1.0 * (i / effective_rank) ** 2) 

1039 

1040 The remaining singular values' tail is fat, decreasing as:: 

1041 

1042 tail_strength * exp(-0.1 * i / effective_rank). 

1043 

1044 The low rank part of the profile can be considered the structured 

1045 signal part of the data while the tail can be considered the noisy 

1046 part of the data that cannot be summarized by a low number of linear 

1047 components (singular vectors). 

1048 

1049 This kind of singular profiles is often seen in practice, for instance: 

1050 - gray level pictures of faces 

1051 - TF-IDF vectors of text documents crawled from the web 

1052 

1053 Read more in the :ref:`User Guide <sample_generators>`. 

1054 

1055 Parameters 

1056 ---------- 

1057 n_samples : int, optional (default=100) 

1058 The number of samples. 

1059 

1060 n_features : int, optional (default=100) 

1061 The number of features. 

1062 

1063 effective_rank : int, optional (default=10) 

1064 The approximate number of singular vectors required to explain most of 

1065 the data by linear combinations. 

1066 

1067 tail_strength : float between 0.0 and 1.0, optional (default=0.5) 

1068 The relative importance of the fat noisy tail of the singular values 

1069 profile. 

1070 

1071 random_state : int, RandomState instance or None (default) 

1072 Determines random number generation for dataset creation. Pass an int 

1073 for reproducible output across multiple function calls. 

1074 See :term:`Glossary <random_state>`. 

1075 

1076 Returns 

1077 ------- 

1078 X : array of shape [n_samples, n_features] 

1079 The matrix. 

1080 """ 

1081 generator = check_random_state(random_state) 

1082 n = min(n_samples, n_features) 

1083 

1084 # Random (ortho normal) vectors 

1085 u, _ = linalg.qr(generator.randn(n_samples, n), mode='economic') 

1086 v, _ = linalg.qr(generator.randn(n_features, n), mode='economic') 

1087 

1088 # Index of the singular values 

1089 singular_ind = np.arange(n, dtype=np.float64) 

1090 

1091 # Build the singular profile by assembling signal and noise components 

1092 low_rank = ((1 - tail_strength) * 

1093 np.exp(-1.0 * (singular_ind / effective_rank) ** 2)) 

1094 tail = tail_strength * np.exp(-0.1 * singular_ind / effective_rank) 

1095 s = np.identity(n) * (low_rank + tail) 

1096 

1097 return np.dot(np.dot(u, s), v.T)

1100def make_sparse_coded_signal(n_samples, n_components, n_features, 

1101 n_nonzero_coefs, random_state=None): 

1102 """Generate a signal as a sparse combination of dictionary elements. 

1103 

1104 Returns a matrix Y = DX, such as D is (n_features, n_components), 

1105 X is (n_components, n_samples) and each column of X has exactly 

1106 n_nonzero_coefs non-zero elements. 

1107 

1108 Read more in the :ref:`User Guide <sample_generators>`. 

1109 

1110 Parameters 

1111 ---------- 

1112 n_samples : int 

1113 number of samples to generate 

1114 

1115 n_components : int, 

1116 number of components in the dictionary 

1117 

1118 n_features : int 

1119 number of features of the dataset to generate 

1120 

1121 n_nonzero_coefs : int 

1122 number of active (non-zero) coefficients in each sample 

1123 

1124 random_state : int, RandomState instance or None (default) 

1125 Determines random number generation for dataset creation. Pass an int 

1126 for reproducible output across multiple function calls. 

1127 See :term:`Glossary <random_state>`. 

1128 

1129 Returns 

1130 ------- 

1131 data : array of shape [n_features, n_samples] 

1132 The encoded signal (Y). 

1133 

1134 dictionary : array of shape [n_features, n_components] 

1135 The dictionary with normalized components (D). 

1136 

1137 code : array of shape [n_components, n_samples]

1140 

1141 """ 

1142 generator = check_random_state(random_state) 

1143 

1144 # generate dictionary

1155 

1156 # encode signal 

1157 Y = np.dot(D, X) 

1158 

1159 return map(np.squeeze, (Y, D, X))

1270 The value of the largest coefficient. 

1271 

1272 random_state : int, RandomState instance or None (default) 

1273 Determines random number generation for dataset creation. Pass an int 

1274 for reproducible output across multiple function calls. 

1275 See :term:`Glossary <random_state>`. 

1276 

1277 Returns 

1278 ------- 

1279 prec : sparse matrix of shape (dim, dim) 

1280 The generated matrix. 

1281 

1282 Notes 

1283 ----- 

1284 The sparsity is actually imposed on the cholesky factor of the matrix. 

1285 Thus alpha does not translate directly into the filling fraction of 

1286 the matrix itself. 

1287 

1288 See also 

1289 -------- 

1290 make_spd_matrix 

1291 """ 

1292 random_state = check_random_state(random_state) 

1293 

1294 chol = -np.eye(dim) 

1295 aux = random_state.rand(dim, dim) 

1296 aux[aux < alpha] = 0 

1297 aux[aux > alpha] = (smallest_coef 

1298 + (largest_coef - smallest_coef) 

1299 * random_state.rand(np.sum(aux > alpha))) 

1300 aux = np.tril(aux, k=-1) 

1301 

1302 # Permute the lines: we don't want to have asymmetries in the final 

1303 # SPD matrix 

1304 permutation = random_state.permutation(dim) 

1305 aux = aux[permutation].T[permutation] 

1306 chol += aux 

1307 prec = np.dot(chol.T, chol) 

1308 

1309 if norm_diag: 

1310 # Form the diagonal vector into a row matrix 

1311 d = np.diag(prec).reshape(1, prec.shape[0]) 

1312 d = 1. / np.sqrt(d) 

1313 

1314 prec *= d 

1315 prec *= d.T 

1316 

1317 return prec

1414def make_gaussian_quantiles(mean=None, cov=1., n_samples=100, 

1415 n_features=2, n_classes=3, 

1416 shuffle=True, random_state=None): 

1417 r"""Generate isotropic Gaussian and label samples by quantile 

1418 

1419 This classification dataset is constructed by taking a multi-dimensional 

1420 standard normal distribution and defining classes separated by nested 

1421 concentric multi-dimensional spheres such that roughly equal numbers of 

1422 samples are in each class (quantiles of the :math:`\chi^2` distribution). 

1423 

1424 Read more in the :ref:`User Guide <sample_generators>`. 

1425 

1426 Parameters 

1427 ---------- 

1428 mean : array of shape [n_features], optional (default=None) 

1429 The mean of the multi-dimensional normal distribution. 

1430 If None then use the origin (0, 0, ...). 

1431 

1432 cov : float, optional (default=1.) 

1433 The covariance matrix will be this value times the unit matrix. This 

1434 dataset only produces symmetric normal distributions. 

1435 

1436 n_samples : int, optional (default=100) 

1437 The total number of points equally divided among classes. 

1438 

1439 n_features : int, optional (default=2) 

1440 The number of features for each sample. 

1441 

1442 n_classes : int, optional (default=3) 

1443 The number of classes 

1444 

1445 shuffle : boolean, optional (default=True) 

1446 Shuffle the samples. 

1447 

1448 random_state : int, RandomState instance or None (default) 

1449 Determines random number generation for dataset creation. Pass an int 

1450 for reproducible output across multiple function calls. 

1451 See :term:`Glossary <random_state>`. 

1452 

1453 Returns 

1454 ------- 

1455 X : array of shape [n_samples, n_features] 

1456 The generated samples. 

1457 

1458 y : array of shape [n_samples] 

1459 The integer labels for quantile membership of each sample. 

1460 

1461 Notes 

1462 ----- 

1463 The dataset is from Zhu et al [1]. 

1464 

1465 References 

1466 ---------- 

1467 .. [1] J. Zhu, H. Zou, S. Rosset, T. Hastie, "Multi-class AdaBoost", 2009. 

1468 

1469 """ 

1470 if n_samples < n_classes: 

1471 raise ValueError("n_samples must be at least n_classes") 

1472 

1473 generator = check_random_state(random_state) 

1474 

1475 if mean is None: 

1476 mean = np.zeros(n_features) 

1477 else: 

1478 mean = np.array(mean) 

1479 

1480 # Build multivariate normal distribution 

1481 X = generator.multivariate_normal(mean, cov * np.identity(n_features), 

1482 (n_samples,)) 

1483 

1484 # Sort by distance from origin 

1485 idx = np.argsort(np.sum((X - mean[np.newaxis, :]) ** 2, axis=1)) 

1486 X = X[idx, :] 

1487 

1488 # Label by quantile 

1489 step = n_samples // n_classes 

1490 

1491 y = np.hstack([np.repeat(np.arange(n_classes), step), 

1492 np.repeat(n_classes - 1, n_samples - step * n_classes)]) 

1493 

1494 if shuffle: 

1495 X, y = util_shuffle(X, y, random_state=generator) 

1496 

1497 return X, y
