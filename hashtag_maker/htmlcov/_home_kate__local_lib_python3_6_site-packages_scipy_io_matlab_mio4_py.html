108 def read_header(self): 

109 ''' Read and return header for variable ''' 

110 data = read_dtype(self.mat_stream, self.dtypes['header']) 

111 name = self.mat_stream.read(int(data['namlen'])).strip(b'\x00') 

112 if data['mopt'] < 0 or data['mopt'] > 5000: 

113 raise ValueError('Mat 4 mopt wrong format, byteswapping problem?') 

114 M, rest = divmod(data['mopt'], 1000) # order code 

115 if M not in (0, 1): 

116 warnings.warn("We do not support byte ordering '%s'; returned " 

117 "data may be corrupt" % order_codes[M], 

118 UserWarning) 

119 O, rest = divmod(rest, 100) # unused, should be 0 

120 if O != 0: 

121 raise ValueError('O in MOPT integer should be 0, wrong format?') 

122 P, rest = divmod(rest, 10) # data type code e.g miDOUBLE (see above) 

123 T = rest # matrix type code e.g. mxFULL_CLASS (see above) 

124 dims = (data['mrows'], data['ncols']) 

125 is_complex = data['imagf'] == 1 

126 dtype = self.dtypes[P] 

127 return VarHeader4( 

128 name, 

129 dtype, 

130 T, 

131 dims, 

132 is_complex)

134 def array_from_header(self, hdr, process=True): 

135 mclass = hdr.mclass 

136 if mclass == mxFULL_CLASS: 

137 arr = self.read_full_array(hdr) 

138 elif mclass == mxCHAR_CLASS: 

139 arr = self.read_char_array(hdr) 

140 if process and self.chars_as_strings: 

141 arr = chars_to_strings(arr) 

142 elif mclass == mxSPARSE_CLASS: 

143 # no current processing (below) makes sense for sparse 

144 return self.read_sparse_array(hdr) 

145 else: 

146 raise TypeError('No reader for class code %s' % mclass) 

147 if process and self.squeeze_me: 

148 return squeeze_element(arr) 

149 return arr

151 def read_sub_array(self, hdr, copy=True): 

152 ''' Mat4 read using header `hdr` dtype and dims 

153 

154 Parameters 

155 ---------- 

156 hdr : object 

157 object with attributes ``dtype``, ``dims``. dtype is assumed to be 

158 the correct endianness 

159 copy : bool, optional 

160 copies array before return if True (default True) 

161 (buffer is usually read only) 

162 

163 Returns 

164 ------- 

165 arr : ndarray 

166 of dtype givem by `hdr` ``dtype`` and shape givem by `hdr` ``dims`` 

167 ''' 

168 dt = hdr.dtype 

169 dims = hdr.dims 

170 num_bytes = dt.itemsize 

171 for d in dims: 

172 num_bytes *= d 

173 buffer = self.mat_stream.read(int(num_bytes)) 

174 if len(buffer) != num_bytes: 

175 raise ValueError("Not enough bytes to read matrix '%s'; is this " 

176 "a badly-formed file? Consider listing matrices " 

177 "with `whosmat` and loading named matrices with " 

178 "`variable_names` kwarg to `loadmat`" % hdr.name) 

179 arr = np.ndarray(shape=dims, 

180 dtype=dt, 

181 buffer=buffer, 

182 order='F') 

183 if copy: 

184 arr = arr.copy() 

185 return arr

187 def read_full_array(self, hdr): 

188 ''' Full (rather than sparse) matrix getter 

189 

190 Read matrix (array) can be real or complex 

191 

192 Parameters 

193 ---------- 

194 hdr : ``VarHeader4`` instance 

195 

196 Returns 

197 ------- 

198 arr : ndarray 

199 complex array if ``hdr.is_complex`` is True, otherwise a real 

200 numeric array 

201 ''' 

202 if hdr.is_complex: 

203 # avoid array copy to save memory 

204 res = self.read_sub_array(hdr, copy=False) 

205 res_j = self.read_sub_array(hdr, copy=False) 

206 return res + (res_j * 1j) 

207 return self.read_sub_array(hdr)

209 def read_char_array(self, hdr): 

210 ''' latin-1 text matrix (char matrix) reader 

211

227 def read_sparse_array(self, hdr): 

228 ''' Read and return sparse matrix type 

229 

230 Parameters 

231 ---------- 

232 hdr : ``VarHeader4`` instance 

233 

234 Returns 

235 ------- 

236 arr : ``scipy.sparse.coo_matrix`` 

237 with dtype ``float`` and shape read from the sparse matrix data 

238 

239 Notes 

240 ----- 

241 MATLAB 4 real sparse arrays are saved in a N+1 by 3 array format, where 

242 N is the number of non-zero values. Column 1 values [0:N] are the 

243 (1-based) row indices of the each non-zero value, column 2 [0:N] are the 

244 column indices, column 3 [0:N] are the (real) values. The last values 

245 [-1,0:2] of the rows, column indices are shape[0] and shape[1] 

246 respectively of the output matrix. The last value for the values column 

247 is a padding 0. mrows and ncols values from the header give the shape of 

248 the stored matrix, here [N+1, 3]. Complex data is saved as a 4 column 

249 matrix, where the fourth column contains the imaginary component; the 

250 last value is again 0. Complex sparse data do *not* have the header 

251 ``imagf`` field set to True; the fact that the data are complex is only 

252 detectable because there are 4 storage columns 

253 ''' 

254 res = self.read_sub_array(hdr) 

255 tmp = res[:-1,:] 

256 # All numbers are float64 in Matlab, but SciPy sparse expects int shape 

257 dims = (int(res[-1,0]), int(res[-1,1])) 

258 I = np.ascontiguousarray(tmp[:,0],dtype='intc') # fixes byte order also 

259 J = np.ascontiguousarray(tmp[:,1],dtype='intc') 

260 I -= 1 # for 1-based indexing 

261 J -= 1 

262 if res.shape[1] == 3: 

263 V = np.ascontiguousarray(tmp[:,2],dtype='float') 

264 else: 

265 V = np.ascontiguousarray(tmp[:,2],dtype='complex') 

266 V.imag = tmp[:,3] 

267 return scipy.sparse.coo_matrix((V,(I,J)), dims)

269 def shape_from_header(self, hdr): 

270 '''Read the shape of the array described by the header. 

271 The file position after this call is unspecified. 

272 ''' 

273 mclass = hdr.mclass 

274 if mclass == mxFULL_CLASS: 

275 shape = tuple(map(int, hdr.dims)) 

276 elif mclass == mxCHAR_CLASS: 

277 shape = tuple(map(int, hdr.dims)) 

278 if self.chars_as_strings: 

279 shape = shape[:-1] 

280 elif mclass == mxSPARSE_CLASS: 

281 dt = hdr.dtype 

282 dims = hdr.dims 

283 

284 if not (len(dims) == 2 and dims[0] >= 1 and dims[1] >= 1): 

285 return () 

286 

287 # Read only the row and column counts 

288 self.mat_stream.seek(dt.itemsize * (dims[0] - 1), 1) 

289 rows = np.ndarray(shape=(1,), dtype=dt, 

290 buffer=self.mat_stream.read(dt.itemsize)) 

291 self.mat_stream.seek(dt.itemsize * (dims[0] - 1), 1) 

292 cols = np.ndarray(shape=(1,), dtype=dt, 

293 buffer=self.mat_stream.read(dt.itemsize)) 

294 

295 shape = (int(rows), int(cols)) 

296 else: 

297 raise TypeError('No reader for class code %s' % mclass) 

298 

299 if self.squeeze_me: 

300 shape = tuple([x for x in shape if x != 1]) 

301 return shape

336 def read_var_header(self): 

337 ''' Read and return header, next position 

338 

339 Parameters 

340 ---------- 

341 None 

342 

343 Returns 

344 ------- 

345 header : object 

346 object that can be passed to self.read_var_array, and that 

347 has attributes ``name`` and ``is_global`` 

348 next_position : int 

349 position in stream of next variable 

350 ''' 

351 hdr = self._matrix_reader.read_header() 

352 n = reduce(lambda x, y: x*y, hdr.dims, 1) # fast product 

353 remaining_bytes = hdr.dtype.itemsize * n 

354 if hdr.is_complex and not hdr.mclass == mxSPARSE_CLASS: 

355 remaining_bytes *= 2 

356 next_position = self.mat_stream.tell() + remaining_bytes 

357 return hdr, next_position

377 def get_variables(self, variable_names=None): 

378 ''' get variables from stream as dictionary 

379 

380 Parameters 

381 ---------- 

382 variable_names : None or str or sequence of str, optional 

383 variable name, or sequence of variable names to get from Mat file / 

384 file stream. If None, then get all variables in file 

385 ''' 

386 if isinstance(variable_names, string_types): 

387 variable_names = [variable_names] 

388 elif variable_names is not None: 

389 variable_names = list(variable_names) 

390 self.mat_stream.seek(0) 

391 # set up variable reader 

392 self.initialize_read() 

393 mdict = {} 

394 while not self.end_of_stream(): 

395 hdr, next_position = self.read_var_header() 

396 name = asstr(hdr.name) 

397 if variable_names is not None and name not in variable_names: 

398 self.mat_stream.seek(next_position) 

399 continue 

400 mdict[name] = self.read_var_array(hdr) 

401 self.mat_stream.seek(next_position) 

402 if variable_names is not None: 

403 variable_names.remove(name) 

404 if len(variable_names) == 0: 

405 break 

406 return mdict

408 def list_variables(self): 

409 ''' list variables from stream ''' 

410 self.mat_stream.seek(0) 

411 # set up variable reader 

412 self.initialize_read() 

413 vars = [] 

414 while not self.end_of_stream(): 

415 hdr, next_position = self.read_var_header() 

416 name = asstr(hdr.name) 

417 shape = self._matrix_reader.shape_from_header(hdr) 

418 info = mclass_info.get(hdr.mclass, 'unknown') 

419 vars.append((name, shape, info))

421 self.mat_stream.seek(next_position) 

422 return vars

460 def write_header(self, name, shape, P=miDOUBLE, T=mxFULL_CLASS, imagf=0): 

461 ''' Write header for given data options 

462 

463 Parameters 

464 ---------- 

465 name : str 

466 name of variable 

467 shape : sequence 

468 Shape of array as it will be read in matlab 

469 P : int, optional 

470 code for mat4 data type, one of ``miDOUBLE, miSINGLE, miINT32, 

471 miINT16, miUINT16, miUINT8`` 

472 T : int, optional 

473 code for mat4 matrix class, one of ``mxFULL_CLASS, mxCHAR_CLASS, 

474 mxSPARSE_CLASS`` 

475 imagf : int, optional 

476 flag indicating complex 

477 ''' 

478 header = np.empty((), mdtypes_template['header']) 

479 M = not SYS_LITTLE_ENDIAN 

480 O = 0 

481 header['mopt'] = (M * 1000 + 

482 O * 100 + 

483 P * 10 + 

484 T) 

485 header['mrows'] = shape[0] 

486 header['ncols'] = shape[1] 

487 header['imagf'] = imagf 

488 header['namlen'] = len(name) + 1 

489 self.write_bytes(header) 

490 self.write_string(asbytes(name + '\0'))

492 def write(self, arr, name): 

493 ''' Write matrix `arr`, with name `name` 

494 

495 Parameters 

496 ---------- 

497 arr : array_like 

498 array to write 

499 name : str 

500 name in matlab workspace 

501 ''' 

502 # we need to catch sparse first, because np.asarray returns an 

503 # an object array for scipy.sparse 

504 if scipy.sparse.issparse(arr): 

505 self.write_sparse(arr, name) 

506 return 

507 arr = np.asarray(arr) 

508 dt = arr.dtype 

509 if not dt.isnative: 

510 arr = arr.astype(dt.newbyteorder('=')) 

511 dtt = dt.type 

512 if dtt is np.object_: 

513 raise TypeError('Cannot save object arrays in Mat4') 

514 elif dtt is np.void: 

515 raise TypeError('Cannot save void type arrays') 

516 elif dtt in (np.unicode_, np.string_): 

517 self.write_char(arr, name) 

518 return 

519 self.write_numeric(arr, name)

521 def write_numeric(self, arr, name): 

522 arr = arr_to_2d(arr, self.oned_as) 

523 imagf = arr.dtype.kind == 'c' 

524 try: 

525 P = np_to_mtypes[arr.dtype.str[1:]] 

526 except KeyError: 

527 if imagf: 

528 arr = arr.astype('c128') 

529 else: 

530 arr = arr.astype('f8') 

531 P = miDOUBLE 

532 self.write_header(name, 

533 arr.shape, 

534 P=P, 

535 T=mxFULL_CLASS, 

536 imagf=imagf) 

537 if imagf: 

538 self.write_bytes(arr.real) 

539 self.write_bytes(arr.imag) 

540 else: 

541 self.write_bytes(arr)

543 def write_char(self, arr, name): 

544 arr = arr_to_chars(arr) 

545 arr = arr_to_2d(arr, self.oned_as) 

546 dims = arr.shape 

547 self.write_header( 

548 name, 

549 dims, 

550 P=miUINT8, 

551 T=mxCHAR_CLASS) 

552 if arr.dtype.kind == 'U': 

553 # Recode unicode to latin1 

554 n_chars = np.prod(dims)

557 buffer=arr)

559 arr = np.ndarray(shape=dims, dtype='S1', buffer=st)

562 def write_sparse(self, arr, name): 

563 ''' Sparse matrices are 2D 

564 

565 See docstring for VarReader4.read_sparse_array 

566 ''' 

567 A = arr.tocoo() # convert to sparse COO format (ijv) 

568 imagf = A.dtype.kind == 'c' 

569 ijv = np.zeros((A.nnz + 1, 3+imagf), dtype='f8') 

570 ijv[:-1,0] = A.row 

571 ijv[:-1,1] = A.col 

572 ijv[:-1,0:2] += 1 # 1 based indexing 

573 if imagf: 

574 ijv[:-1,2] = A.data.real 

575 ijv[:-1,3] = A.data.imag 

576 else: 

577 ijv[:-1,2] = A.data 

578 ijv[-1,0:2] = A.shape 

579 self.write_header( 

580 name, 

581 ijv.shape, 

582 P=miDOUBLE, 

583 T=mxSPARSE_CLASS) 

584 self.write_bytes(ijv)
