27def _line_search_wolfe12(f, fprime, xk, pk, gfk, old_fval, old_old_fval, 

28 **kwargs): 

29 """ 

30 Same as line_search_wolfe1, but fall back to line_search_wolfe2 if 

31 suitable step length is not found, and raise an exception if a 

32 suitable step length is not found. 

33 

34 Raises 

35 ------ 

36 _LineSearchError 

37 If no suitable step size is found 

38 

39 """ 

40 ret = line_search_wolfe1(f, fprime, xk, pk, gfk, 

41 old_fval, old_old_fval, 

42 **kwargs)

44 if ret[0] is None:

46 ret = line_search_wolfe2(f, fprime, xk, pk, gfk,

55def _cg(fhess_p, fgrad, maxiter, tol): 

56 """ 

57 Solve iteratively the linear system 'fhess_p . xsupi = fgrad' 

58 with a conjugate gradient descent. 

59 

60 Parameters 

61 ---------- 

62 fhess_p : callable 

63 Function that takes the gradient as a parameter and returns the 

64 matrix product of the Hessian and gradient 

65 

66 fgrad : ndarray, shape (n_features,) or (n_features + 1,) 

67 Gradient vector 

68 

69 maxiter : int 

70 Number of CG iterations. 

71 

72 tol : float 

73 Stopping criterion. 

74 

75 Returns 

76 ------- 

77 xsupi : ndarray, shape (n_features,) or (n_features + 1,) 

78 Estimated solution 

79 """ 

80 xsupi = np.zeros(len(fgrad), dtype=fgrad.dtype) 

81 ri = fgrad 

82 psupi = -ri 

83 i = 0 

84 dri0 = np.dot(ri, ri) 

85 

86 while i <= maxiter: 

87 if np.sum(np.abs(ri)) <= tol: 

88 break 

89 

90 Ap = fhess_p(psupi) 

91 # check curvature 

92 curv = np.dot(psupi, Ap) 

93 if 0 <= curv <= 3 * np.finfo(np.float64).eps: 

94 break 

95 elif curv < 0: 

96 if i > 0: 

97 break 

98 else: 

99 # fall back to steepest descent direction 

100 xsupi += dri0 / curv * psupi 

101 break 

102 alphai = dri0 / curv 

103 xsupi += alphai * psupi 

104 ri = ri + alphai * Ap 

105 dri1 = np.dot(ri, ri) 

106 betai = dri1 / dri0 

107 psupi = -ri + betai * psupi 

108 i = i + 1 

109 dri0 = dri1 # update np.dot(ri,ri) for next time. 

110 

111 return xsupi

114def newton_cg(grad_hess, func, grad, x0, args=(), tol=1e-4, 

115 maxiter=100, maxinner=200, line_search=True, warn=True): 

116 """ 

117 Minimization of scalar function of one or more variables using the 

118 Newton-CG algorithm. 

119 

120 Parameters 

121 ---------- 

122 grad_hess : callable 

123 Should return the gradient and a callable returning the matvec product 

124 of the Hessian. 

125 

126 func : callable 

127 Should return the value of the function. 

128 

129 grad : callable 

130 Should return the function value and the gradient. This is used 

131 by the linesearch functions. 

132 

133 x0 : array of float 

134 Initial guess. 

135 

136 args : tuple, optional 

137 Arguments passed to func_grad_hess, func and grad. 

138 

139 tol : float 

140 Stopping criterion. The iteration will stop when 

141 ``max{|g_i | i = 1, ..., n} <= tol`` 

142 where ``g_i`` is the i-th component of the gradient. 

143 

144 maxiter : int 

145 Number of Newton iterations. 

146 

147 maxinner : int 

148 Number of CG iterations. 

149 

150 line_search : boolean 

151 Whether to use a line search or not. 

152 

153 warn : boolean 

154 Whether to warn when didn't converge. 

155 

156 Returns 

157 ------- 

158 xk : ndarray of float 

159 Estimated minimum. 

160 """ 

161 x0 = np.asarray(x0).flatten() 

162 xk = x0 

163 k = 0 

164 

165 if line_search: 

166 old_fval = func(x0, *args) 

167 old_old_fval = None 

168 

169 # Outer loop: our Newton iteration 

170 while k < maxiter: 

171 # Compute a search direction pk by applying the CG method to 

172 # del2 f(xk) p = - fgrad f(xk) starting from 0. 

173 fgrad, fhess_p = grad_hess(xk, *args) 

174 

175 absgrad = np.abs(fgrad) 

176 if np.max(absgrad) < tol: 

177 break 

178 

179 maggrad = np.sum(absgrad) 

180 eta = min([0.5, np.sqrt(maggrad)]) 

181 termcond = eta * maggrad 

182 

183 # Inner loop: solve the Newton update by conjugate gradient, to 

184 # avoid inverting the Hessian 

185 xsupi = _cg(fhess_p, fgrad, maxiter=maxinner, tol=termcond) 

186 

187 alphak = 1.0 

188 

189 if line_search: 

190 try: 

191 alphak, fc, gc, old_fval, old_old_fval, gfkp1 = \ 

192 _line_search_wolfe12(func, grad, xk, xsupi, fgrad, 

193 old_fval, old_old_fval, args=args) 

194 except _LineSearchError: 

195 warnings.warn('Line Search failed') 

196 break 

197 

198 xk = xk + alphak * xsupi # upcast if necessary 

199 k += 1 

200 

201 if warn and k >= maxiter: 

202 warnings.warn("newton-cg failed to converge. Increase the " 

203 "number of iterations.", ConvergenceWarning) 

204 return xk, k
