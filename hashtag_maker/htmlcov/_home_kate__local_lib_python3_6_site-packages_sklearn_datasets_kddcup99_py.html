48def fetch_kddcup99(subset=None, data_home=None, shuffle=False, 

49 random_state=None, 

50 percent10=True, download_if_missing=True, return_X_y=False): 

51 """Load the kddcup99 dataset (classification). 

52 

53 Download it if necessary. 

54 

55 ================= ==================================== 

56 Classes 23 

57 Samples total 4898431 

58 Dimensionality 41 

59 Features discrete (int) or continuous (float) 

60 ================= ==================================== 

61 

62 Read more in the :ref:`User Guide <kddcup99_dataset>`. 

63 

64 .. versionadded:: 0.18 

65 

66 Parameters 

67 ---------- 

68 subset : None, 'SA', 'SF', 'http', 'smtp' 

69 To return the corresponding classical subsets of kddcup 99. 

70 If None, return the entire kddcup 99 dataset. 

71 

72 data_home : string, optional 

73 Specify another download and cache folder for the datasets. By default 

74 all scikit-learn data is stored in '~/scikit_learn_data' subfolders. 

75 .. versionadded:: 0.19 

76 

77 shuffle : bool, default=False 

78 Whether to shuffle dataset. 

79 

80 random_state : int, RandomState instance or None (default) 

81 Determines random number generation for dataset shuffling and for 

82 selection of abnormal samples if `subset='SA'`. Pass an int for 

83 reproducible output across multiple function calls. 

84 See :term:`Glossary <random_state>`. 

85 

86 percent10 : bool, default=True 

87 Whether to load only 10 percent of the data. 

88 

89 download_if_missing : bool, default=True 

90 If False, raise a IOError if the data is not locally available 

91 instead of trying to download the data from the source site. 

92 

93 return_X_y : boolean, default=False. 

94 If True, returns ``(data, target)`` instead of a Bunch object. See 

95 below for more information about the `data` and `target` object. 

96 

97 .. versionadded:: 0.20 

98 

99 Returns 

100 ------- 

101 data : Bunch 

102 Dictionary-like object, the interesting attributes are: 

103 - 'data', the data to learn. 

104 - 'target', the regression target for each sample. 

105 - 'DESCR', a description of the dataset. 

106 

107 (data, target) : tuple if ``return_X_y`` is True 

108 

109 .. versionadded:: 0.20 

110 """ 

111 data_home = get_data_home(data_home=data_home) 

112 kddcup99 = _fetch_brute_kddcup99(data_home=data_home, 

113 percent10=percent10, 

114 download_if_missing=download_if_missing) 

115 

116 data = kddcup99.data 

117 target = kddcup99.target 

118 

119 if subset == 'SA': 

120 s = target == b'normal.' 

121 t = np.logical_not(s) 

122 normal_samples = data[s, :] 

123 normal_targets = target[s] 

124 abnormal_samples = data[t, :] 

125 abnormal_targets = target[t] 

126 

127 n_samples_abnormal = abnormal_samples.shape[0] 

128 # selected abnormal samples: 

129 random_state = check_random_state(random_state) 

130 r = random_state.randint(0, n_samples_abnormal, 3377) 

131 abnormal_samples = abnormal_samples[r] 

132 abnormal_targets = abnormal_targets[r] 

133 

134 data = np.r_[normal_samples, abnormal_samples] 

135 target = np.r_[normal_targets, abnormal_targets] 

136 

137 if subset == 'SF' or subset == 'http' or subset == 'smtp': 

138 # select all samples with positive logged_in attribute: 

139 s = data[:, 11] == 1 

140 data = np.c_[data[s, :11], data[s, 12:]] 

141 target = target[s] 

142 

143 data[:, 0] = np.log((data[:, 0] + 0.1).astype(float, copy=False)) 

144 data[:, 4] = np.log((data[:, 4] + 0.1).astype(float, copy=False)) 

145 data[:, 5] = np.log((data[:, 5] + 0.1).astype(float, copy=False)) 

146 

147 if subset == 'http': 

148 s = data[:, 2] == b'http' 

149 data = data[s] 

150 target = target[s] 

151 data = np.c_[data[:, 0], data[:, 4], data[:, 5]] 

152 

153 if subset == 'smtp': 

154 s = data[:, 2] == b'smtp' 

155 data = data[s] 

156 target = target[s] 

157 data = np.c_[data[:, 0], data[:, 4], data[:, 5]] 

158 

159 if subset == 'SF': 

160 data = np.c_[data[:, 0], data[:, 2], data[:, 4], data[:, 5]] 

161 

162 if shuffle: 

163 data, target = shuffle_method(data, target, random_state=random_state) 

164 

165 module_path = dirname(__file__) 

166 with open(join(module_path, 'descr', 'kddcup99.rst')) as rst_file: 

167 fdescr = rst_file.read() 

168 

169 if return_X_y: 

170 return data, target 

171 

172 return Bunch(data=data, target=target, DESCR=fdescr)

178 """Load the kddcup99 dataset, downloading it if necessary. 

179 

180 Parameters 

181 ---------- 

182 data_home : string, optional 

183 Specify another download and cache folder for the datasets. By default 

184 all scikit-learn data is stored in '~/scikit_learn_data' subfolders. 

185 

186 download_if_missing : boolean, default=True 

187 If False, raise a IOError if the data is not locally available 

188 instead of trying to download the data from the source site. 

189 

190 percent10 : bool, default=True 

191 Whether to load only 10 percent of the data. 

192 

193 Returns 

194 ------- 

195 dataset : dict-like object with the following attributes: 

196 dataset.data : numpy array of shape (494021, 41) 

197 Each row corresponds to the 41 features in the dataset. 

198 dataset.target : numpy array of shape (494021,) 

199 Each value corresponds to one of the 21 attack types or to the 

200 label 'normal.'. 

201 dataset.DESCR : string 

202 Description of the kddcup99 dataset.

205 

206 data_home = get_data_home(data_home=data_home) 

207 dir_suffix = "-py3" 

208 

209 if percent10: 

210 kddcup_dir = join(data_home, "kddcup99_10" + dir_suffix) 

211 archive = ARCHIVE_10_PERCENT 

212 else: 

213 kddcup_dir = join(data_home, "kddcup99" + dir_suffix) 

214 archive = ARCHIVE 

215 

216 samples_path = join(kddcup_dir, "samples") 

217 targets_path = join(kddcup_dir, "targets") 

218 available = exists(samples_path) 

219 

220 if download_if_missing and not available: 

221 _mkdirp(kddcup_dir) 

222 logger.info("Downloading %s" % archive.url) 

223 _fetch_remote(archive, dirname=kddcup_dir) 

224 dt = [('duration', int), 

225 ('protocol_type', 'S4'), 

226 ('service', 'S11'), 

227 ('flag', 'S6'), 

228 ('src_bytes', int), 

229 ('dst_bytes', int),

240 ('num_file_creations', int), 

241 ('num_shells', int), 

242 ('num_access_files', int), 

243 ('num_outbound_cmds', int), 

244 ('is_host_login', int), 

245 ('is_guest_login', int), 

246 ('count', int), 

247 ('srv_count', int), 

248 ('serror_rate', float), 

249 ('srv_serror_rate', float), 

250 ('rerror_rate', float), 

251 ('srv_rerror_rate', float), 

252 ('same_srv_rate', float), 

253 ('diff_srv_rate', float), 

254 ('srv_diff_host_rate', float), 

255 ('dst_host_count', int),

260 ('dst_host_srv_diff_host_rate', float), 

261 ('dst_host_serror_rate', float), 

262 ('dst_host_srv_serror_rate', float), 

263 ('dst_host_rerror_rate', float), 

264 ('dst_host_srv_rerror_rate', float), 

265 ('labels', 'S16')] 

266 DT = np.dtype(dt) 

267 logger.debug("extracting archive") 

268 archive_path = join(kddcup_dir, archive.filename) 

269 file_ = GzipFile(filename=archive_path, mode='r') 

270 Xy = [] 

271 for line in file_.readlines(): 

272 line = line.decode() 

273 Xy.append(line.replace('\n', '').split(',')) 

274 file_.close() 

275 logger.debug('extraction done') 

276 os.remove(archive_path) 

277 

278 Xy = np.asarray(Xy, dtype=object) 

279 for j in range(42): 

280 Xy[:, j] = Xy[:, j].astype(DT[j]) 

281 

282 X = Xy[:, :-1] 

283 y = Xy[:, -1] 

284 # XXX bug when compress!=0: 

285 # (error: 'Incorrect data length while decompressing[...] the file 

286 # could be corrupted.') 

287 

288 _joblib.dump(X, samples_path, compress=0) 

289 _joblib.dump(y, targets_path, compress=0) 

290 elif not available: 

291 if not download_if_missing: 

292 raise IOError("Data not found and `download_if_missing` is False") 

293 

294 try: 

295 X, y 

296 except NameError: 

297 X, y = _refresh_cache([samples_path, targets_path], 0) 

298 # TODO: Revert to the following two lines in v0.23 

299 # X = joblib.load(samples_path) 

300 # y = joblib.load(targets_path) 

301 

302 return Bunch(data=X, target=y)
