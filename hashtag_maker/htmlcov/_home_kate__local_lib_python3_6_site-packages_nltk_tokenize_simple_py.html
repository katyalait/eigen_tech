9r""" 

10Simple Tokenizers 

11 

12These tokenizers divide strings into substrings using the string 

13``split()`` method. 

14When tokenizing using a particular delimiter string, use 

15the string ``split()`` method directly, as this is more efficient. 

16 

17The simple tokenizers are *not* available as separate functions; 

18instead, you should just use the string ``split()`` method directly: 

19 

20 >>> s = "Good muffins cost $3.88\nin New York. Please buy me\ntwo of them.\n\nThanks." 

21 >>> s.split() 

22 ['Good', 'muffins', 'cost', '$3.88', 'in', 'New', 'York.', 

23 'Please', 'buy', 'me', 'two', 'of', 'them.', 'Thanks.'] 

24 >>> s.split(' ') 

25 ['Good', 'muffins', 'cost', '$3.88\nin', 'New', 'York.', '', 

26 'Please', 'buy', 'me\ntwo', 'of', 'them.\n\nThanks.'] 

27 >>> s.split('\n') 

28 ['Good muffins cost $3.88', 'in New York. Please buy me', 

29 'two of them.', '', 'Thanks.'] 

30 

31The simple tokenizers are mainly useful because they follow the 

32standard ``TokenizerI`` interface, and so can be used with any code 

33that expects a tokenizer. For example, these tokenizers can be used 

34to specify the tokenization conventions when building a `CorpusReader`. 

35 

36"""

81class LineTokenizer(TokenizerI): 

82 r"""Tokenize a string into its lines, optionally discarding blank lines. 

83 This is similar to ``s.split('\n')``. 

84 

85 >>> from nltk.tokenize import LineTokenizer 

86 >>> s = "Good muffins cost $3.88\nin New York. Please buy me\ntwo of them.\n\nThanks." 

87 >>> LineTokenizer(blanklines='keep').tokenize(s) 

88 ['Good muffins cost $3.88', 'in New York. Please buy me', 

89 'two of them.', '', 'Thanks.'] 

90 >>> # same as [l for l in s.split('\n') if l.strip()]: 

91 >>> LineTokenizer(blanklines='discard').tokenize(s) 

92 ['Good muffins cost $3.88', 'in New York. Please buy me', 

93 'two of them.', 'Thanks.'] 

94 

95 :param blanklines: Indicates how blank lines should be handled. Valid values are: 

96 

97 - ``discard``: strip blank lines out of the token list before returning it. 

98 A line is considered blank if it contains only whitespace characters. 

99 - ``keep``: leave all blank lines in the token list. 

100 - ``discard-eof``: if the string ends with a newline, then do not generate 

101 a corresponding token ``''`` after that newline. 

102 """ 

103 

104 def __init__(self, blanklines='discard'): 

105 valid_blanklines = ('discard', 'keep', 'discard-eof') 

106 if blanklines not in valid_blanklines: 

107 raise ValueError( 

108 'Blank lines must be one of: %s' % ' '.join(valid_blanklines) 

109 ) 

110 

111 self._blanklines = blanklines 

112 

113 def tokenize(self, s): 

114 lines = s.splitlines() 

115 # If requested, strip off blank lines. 

116 if self._blanklines == 'discard': 

117 lines = [l for l in lines if l.rstrip()] 

118 elif self._blanklines == 'discard-eof': 

119 if lines and not lines[-1].strip(): 

120 lines.pop() 

121 return lines 

122 

123 # discard-eof not implemented 

124 def span_tokenize(self, s): 

125 if self._blanklines == 'keep': 

126 for span in string_span_tokenize(s, r'\n'): 

127 yield span 

128 else: 

129 for span in regexp_span_tokenize(s, r'\n(\s+\n)*'): 

130 yield span
