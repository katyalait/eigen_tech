25def disk_used(path): 

26 """ Return the disk usage in a directory.""" 

27 size = 0 

28 for file in os.listdir(path) + ['.']: 

29 stat = os.stat(os.path.join(path, file)) 

30 if hasattr(stat, 'st_blocks'): 

31 size += stat.st_blocks * 512 

32 else: 

33 # on some platform st_blocks is not available (e.g., Windows) 

34 # approximate by rounding to next multiple of 512 

35 size += (stat.st_size // 512 + 1) * 512 

36 # We need to convert to int to avoid having longs on some systems (we 

37 # don't want longs to avoid problems we SQLite) 

38 return int(size / 1024.)

41def memstr_to_bytes(text): 

42 """ Convert a memory text to its value in bytes. 

43 """ 

44 kilo = 1024 

45 units = dict(K=kilo, M=kilo ** 2, G=kilo ** 3) 

46 try: 

47 size = int(units[text[-1]] * float(text[:-1])) 

48 except (KeyError, ValueError): 

49 raise ValueError( 

50 "Invalid literal for size give: %s (type %s) should be " 

51 "alike '10G', '500M', '50K'." % (text, type(text))) 

52 return size

66# if a rmtree operation fails in rm_subdirs, wait for this much time (in secs), 

67# then retry up to RM_SUBDIRS_N_RETRY times. If it still fails, raise the 

68# exception. this mecanism ensures that the sub-process gc have the time to 

69# collect and close the memmaps before we fail. 

70RM_SUBDIRS_RETRY_TIME = 0.1

74def rm_subdirs(path, onerror=None): 

75 """Remove all subdirectories in this path. 

76 

77 The directory indicated by `path` is left in place, and its subdirectories 

78 are erased. 

79 

80 If onerror is set, it is called to handle the error with arguments (func, 

81 path, exc_info) where func is os.listdir, os.remove, or os.rmdir; 

82 path is the argument to that function that caused it to fail; and 

83 exc_info is a tuple returned by sys.exc_info(). If onerror is None, 

84 an exception is raised. 

85 """ 

86 

87 # NOTE this code is adapted from the one in shutil.rmtree, and is 

88 # just as fast 

89 

90 names = [] 

91 try: 

92 names = os.listdir(path) 

93 except os.error: 

94 if onerror is not None: 

95 onerror(os.listdir, path, sys.exc_info()) 

96 else: 

97 raise 

98 

99 for name in names: 

100 fullname = os.path.join(path, name) 

101 delete_folder(fullname, onerror=onerror)

104def delete_folder(folder_path, onerror=None): 

105 """Utility function to cleanup a temporary folder if it still exists.""" 

106 if os.path.isdir(folder_path): 

107 if onerror is not None: 

108 shutil.rmtree(folder_path, False, onerror) 

109 else: 

110 # allow the rmtree to fail once, wait and re-try. 

111 # if the error is raised again, fail 

112 err_count = 0 

113 while True: 

114 try: 

115 shutil.rmtree(folder_path, False, None) 

116 break 

117 except (OSError, WindowsError): 

118 err_count += 1 

119 if err_count > RM_SUBDIRS_N_RETRY: 

120 warnings.warn( 

121 "Unable to delete folder {} after {} tentatives." 

122 .format(folder_path, RM_SUBDIRS_N_RETRY)) 

123 raise 

124 time.sleep(RM_SUBDIRS_RETRY_TIME)
