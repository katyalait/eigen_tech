67 def _arg_min_or_max_axis(self, axis, op, compare): 

68 if self.shape[axis] == 0: 

69 raise ValueError("Can't apply the operation along a zero-sized " 

70 "dimension.") 

71 

72 if axis < 0: 

73 axis += 2 

74 

75 zero = self.dtype.type(0) 

76 

77 mat = self.tocsc() if axis == 0 else self.tocsr() 

78 mat.sum_duplicates() 

79 

80 ret_size, line_size = mat._swap(mat.shape) 

81 ret = np.zeros(ret_size, dtype=int) 

82 

83 nz_lines, = np.nonzero(np.diff(mat.indptr)) 

84 for i in nz_lines: 

85 p, q = mat.indptr[i:i + 2] 

86 data = mat.data[p:q] 

87 indices = mat.indices[p:q] 

88 am = op(data) 

89 m = data[am] 

90 if compare(m, zero) or q - p == line_size: 

91 ret[i] = indices[am] 

92 else: 

93 zero_ind = _find_missing_index(indices, line_size) 

94 if m == zero: 

95 ret[i] = min(am, zero_ind) 

96 else: 

97 ret[i] = zero_ind 

98 

99 if axis == 1: 

100 ret = ret.reshape(-1, 1) 

101 

102 return np.asmatrix(ret)

104 def _arg_min_or_max(self, axis, out, op, compare): 

105 if out is not None: 

106 raise ValueError("Sparse matrices do not support " 

107 "an 'out' parameter.") 

108 

109 # validateaxis(axis) 

110 

111 if axis is None: 

112 if 0 in self.shape: 

113 raise ValueError("Can't apply the operation to " 

114 "an empty matrix.") 

115 

116 if self.nnz == 0: 

117 return 0 

118 else: 

119 zero = self.dtype.type(0) 

120 mat = self.tocoo() 

121 mat.sum_duplicates() 

122 am = op(mat.data) 

123 m = mat.data[am] 

124 

125 if compare(m, zero): 

126 return mat.row[am] * mat.shape[1] + mat.col[am] 

127 else: 

128 size = np.product(mat.shape) 

129 if size == mat.nnz: 

130 return am 

131 else: 

132 ind = mat.row * mat.shape[1] + mat.col 

133 zero_ind = _find_missing_index(ind, size) 

134 if m == zero: 

135 return min(zero_ind, am) 

136 else: 

137 return zero_ind 

138 

139 return _arg_min_or_max_axis(self, axis, op, compare)

207def _joblib_parallel_args(**kwargs): 

208 """Set joblib.Parallel arguments in a compatible way for 0.11 and 0.12+ 

209 

210 For joblib 0.11 this maps both ``prefer`` and ``require`` parameters to 

211 a specific ``backend``. 

212 

213 Parameters 

214 ---------- 

215 

216 prefer : str in {'processes', 'threads'} or None 

217 Soft hint to choose the default backend if no specific backend 

218 was selected with the parallel_backend context manager. 

219 

220 require : 'sharedmem' or None 

221 Hard condstraint to select the backend. If set to 'sharedmem', 

222 the selected backend will be single-host and thread-based even 

223 if the user asked for a non-thread based backend with 

224 parallel_backend. 

225 

226 See joblib.Parallel documentation for more details 

227 """ 

228 from . import _joblib 

229 

230 if _joblib.__version__ >= LooseVersion('0.12'): 

231 return kwargs 

232 

233 extra_args = set(kwargs.keys()).difference({'prefer', 'require'}) 

234 if extra_args: 

235 raise NotImplementedError('unhandled arguments %s with joblib %s' 

236 % (list(extra_args), _joblib.__version__)) 

237 args = {} 

238 if 'prefer' in kwargs: 

239 prefer = kwargs['prefer'] 

240 if prefer not in ['threads', 'processes', None]: 

241 raise ValueError('prefer=%s is not supported' % prefer) 

242 args['backend'] = {'threads': 'threading', 

243 'processes': 'multiprocessing', 

244 None: None}[prefer] 

245 

246 if 'require' in kwargs: 

247 require = kwargs['require'] 

248 if require not in [None, 'sharedmem']: 

249 raise ValueError('require=%s is not supported' % require) 

250 if require == 'sharedmem': 

251 args['backend'] = 'threading' 

252 return args
