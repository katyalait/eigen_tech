29def malt_regex_tagger(): 

30 from nltk.tag import RegexpTagger 

31 

32 _tagger = RegexpTagger( 

33 [ 

34 (r'\.$', '.'), 

35 (r'\,$', ','), 

36 (r'\?$', '?'), # fullstop, comma, Qmark 

37 (r'\($', '('), 

38 (r'\)$', ')'), # round brackets 

39 (r'\[$', '['), 

40 (r'\]$', ']'), # square brackets 

41 (r'^-?[0-9]+(.[0-9]+)?$', 'CD'), # cardinal numbers 

42 (r'(The|the|A|a|An|an)$', 'DT'), # articles 

43 (r'(He|he|She|she|It|it|I|me|Me|You|you)$', 'PRP'), # pronouns 

44 (r'(His|his|Her|her|Its|its)$', 'PRP$'), # possesive 

45 (r'(my|Your|your|Yours|yours)$', 'PRP$'), # possesive 

46 (r'(on|On|in|In|at|At|since|Since)$', 'IN'), # time prepopsitions 

47 (r'(for|For|ago|Ago|before|Before)$', 'IN'), # time prepopsitions 

48 (r'(till|Till|until|Until)$', 'IN'), # time prepopsitions 

49 (r'(by|By|beside|Beside)$', 'IN'), # space prepopsitions 

50 (r'(under|Under|below|Below)$', 'IN'), # space prepopsitions 

51 (r'(over|Over|above|Above)$', 'IN'), # space prepopsitions 

52 (r'(across|Across|through|Through)$', 'IN'), # space prepopsitions 

53 (r'(into|Into|towards|Towards)$', 'IN'), # space prepopsitions 

54 (r'(onto|Onto|from|From)$', 'IN'), # space prepopsitions 

55 (r'.*able$', 'JJ'), # adjectives 

56 (r'.*ness$', 'NN'), # nouns formed from adjectives

67def find_maltparser(parser_dirname): 

68 """ 

69 A module to find MaltParser .jar file and its dependencies. 

70 """ 

71 if os.path.exists(parser_dirname): # If a full path is given. 

72 _malt_dir = parser_dirname 

73 else: # Try to find path to maltparser directory in environment variables. 

74 _malt_dir = find_dir(parser_dirname, env_vars=('MALT_PARSER',)) 

75 # Checks that that the found directory contains all the necessary .jar 

76 malt_dependencies = ['', '', ''] 

77 _malt_jars = set(find_jars_within_path(_malt_dir)) 

78 _jars = set(os.path.split(jar)[1] for jar in _malt_jars) 

79 malt_dependencies = set(['log4j.jar', 'libsvm.jar', 'liblinear-1.8.jar']) 

80 

81 assert malt_dependencies.issubset(_jars) 

82 assert any( 

83 filter(lambda i: i.startswith('maltparser-') and i.endswith('.jar'), _jars) 

84 ) 

85 return list(_malt_jars)

100class MaltParser(ParserI): 

101 """ 

102 A class for dependency parsing with MaltParser. The input is the paths to: 

103 - a maltparser directory 

104 - (optionally) the path to a pre-trained MaltParser .mco model file 

105 - (optionally) the tagger to use for POS tagging before parsing 

106 - (optionally) additional Java arguments 

107 

108 Example: 

109 >>> from nltk.parse import malt 

110 >>> # With MALT_PARSER and MALT_MODEL environment set. 

111 >>> mp = malt.MaltParser('maltparser-1.7.2', 'engmalt.linear-1.7.mco') # doctest: +SKIP 

112 >>> mp.parse_one('I shot an elephant in my pajamas .'.split()).tree() # doctest: +SKIP 

113 (shot I (elephant an) (in (pajamas my)) .) 

114 >>> # Without MALT_PARSER and MALT_MODEL environment. 

115 >>> mp = malt.MaltParser('/home/user/maltparser-1.7.2/', '/home/user/engmalt.linear-1.7.mco') # doctest: +SKIP 

116 >>> mp.parse_one('I shot an elephant in my pajamas .'.split()).tree() # doctest: +SKIP 

117 (shot I (elephant an) (in (pajamas my)) .) 

118 """

120 def __init__( 

121 self, 

122 parser_dirname, 

123 model_filename=None, 

124 tagger=None, 

125 additional_java_args=None, 

126 ): 

127 """ 

128 An interface for parsing with the Malt Parser. 

129 

130 :param parser_dirname: The path to the maltparser directory that 

131 contains the maltparser-1.x.jar 

132 :type parser_dirname: str 

133 :param model_filename: The name of the pre-trained model with .mco file 

134 extension. If provided, training will not be required. 

135 (see http://www.maltparser.org/mco/mco.html and 

136 see http://www.patful.com/chalk/node/185) 

137 :type model_filename: str 

138 :param tagger: The tagger used to POS tag the raw string before 

139 formatting to CONLL format. It should behave like `nltk.pos_tag` 

140 :type tagger: function 

141 :param additional_java_args: This is the additional Java arguments that 

142 one can use when calling Maltparser, usually this is the heapsize 

143 limits, e.g. `additional_java_args=['-Xmx1024m']` 

144 (see http://goo.gl/mpDBvQ) 

145 :type additional_java_args: list 

146 """ 

147 

148 # Find all the necessary jar files for MaltParser. 

149 self.malt_jars = find_maltparser(parser_dirname) 

150 # Initialize additional java arguments. 

151 self.additional_java_args = ( 

152 additional_java_args if additional_java_args is not None else [] 

153 ) 

154 # Initialize model. 

155 self.model = find_malt_model(model_filename) 

156 self._trained = self.model != 'malt_temp.mco' 

157 # Set the working_dir parameters i.e. `-w` from MaltParser's option. 

158 self.working_dir = tempfile.gettempdir() 

159 # Initialize POS tagger. 

160 self.tagger = tagger if tagger is not None else malt_regex_tagger()

162 def parse_tagged_sents(self, sentences, verbose=False, top_relation_label='null'): 

163 """ 

164 Use MaltParser to parse multiple POS tagged sentences. Takes multiple 

165 sentences where each sentence is a list of (word, tag) tuples. 

166 The sentences must have already been tokenized and tagged. 

167 

168 :param sentences: Input sentences to parse 

169 :type sentence: list(list(tuple(str, str))) 

170 :return: iter(iter(``DependencyGraph``)) the dependency graph 

171 representation of each sentence 

172 """ 

173 if not self._trained: 

174 raise Exception("Parser has not been trained. Call train() first.") 

175 

176 with tempfile.NamedTemporaryFile( 

177 prefix='malt_input.conll.', dir=self.working_dir, mode='w', delete=False 

178 ) as input_file: 

179 with tempfile.NamedTemporaryFile( 

180 prefix='malt_output.conll.', 

181 dir=self.working_dir, 

182 mode='w', 

183 delete=False, 

184 ) as output_file: 

185 # Convert list of sentences to CONLL format. 

186 for line in taggedsents_to_conll(sentences): 

187 input_file.write(text_type(line)) 

188 input_file.close() 

189 

190 # Generate command to run maltparser. 

191 cmd = self.generate_malt_command( 

192 input_file.name, output_file.name, mode="parse" 

193 ) 

194 

195 # This is a maltparser quirk, it needs to be run 

196 # where the model file is. otherwise it goes into an awkward 

197 # missing .jars or strange -w working_dir problem. 

198 _current_path = os.getcwd() # Remembers the current path. 

199 try: # Change to modelfile path 

200 os.chdir(os.path.split(self.model)[0]) 

201 except: 

202 pass 

203 ret = self._execute(cmd, verbose) # Run command. 

204 os.chdir(_current_path) # Change back to current path. 

205 

206 if ret is not 0: 

207 raise Exception( 

208 "MaltParser parsing (%s) failed with exit " 

209 "code %d" % (' '.join(cmd), ret) 

210 ) 

211 

212 # Must return iter(iter(Tree)) 

213 with open(output_file.name) as infile: 

214 for tree_str in infile.read().split('\n\n'): 

215 yield ( 

216 iter( 

217 [ 

218 DependencyGraph( 

219 tree_str, top_relation_label=top_relation_label 

220 ) 

221 ] 

222 ) 

223 ) 

224 

225 os.remove(input_file.name) 

226 os.remove(output_file.name)

228 def parse_sents(self, sentences, verbose=False, top_relation_label='null'): 

229 """ 

230 Use MaltParser to parse multiple sentences. 

231 Takes a list of sentences, where each sentence is a list of words. 

232 Each sentence will be automatically tagged with this 

233 MaltParser instance's tagger. 

234 

235 :param sentences: Input sentences to parse 

236 :type sentence: list(list(str)) 

237 :return: iter(DependencyGraph) 

238 """ 

239 tagged_sentences = (self.tagger(sentence) for sentence in sentences) 

240 return self.parse_tagged_sents( 

241 tagged_sentences, verbose, top_relation_label=top_relation_label 

242 )

244 def generate_malt_command(self, inputfilename, outputfilename=None, mode=None): 

245 """ 

246 This function generates the maltparser command use at the terminal. 

247 

248 :param inputfilename: path to the input file 

249 :type inputfilename: str 

250 :param outputfilename: path to the output file 

251 :type outputfilename: str 

252 """ 

253 

254 cmd = ['java'] 

255 cmd += self.additional_java_args # Adds additional java arguments 

256 # Joins classpaths with ";" if on Windows and on Linux/Mac use ":" 

257 classpaths_separator = ';' if sys.platform.startswith('win') else ':' 

258 cmd += [ 

259 '-cp', 

260 classpaths_separator.join(self.malt_jars), 

261 ] # Adds classpaths for jars 

262 cmd += ['org.maltparser.Malt'] # Adds the main function. 

263 

264 # Adds the model file. 

265 if os.path.exists(self.model): # when parsing 

266 cmd += ['-c', os.path.split(self.model)[-1]] 

267 else: # when learning 

268 cmd += ['-c', self.model] 

269 

270 cmd += ['-i', inputfilename] 

271 if mode == 'parse': 

272 cmd += ['-o', outputfilename] 

273 cmd += ['-m', mode] # mode use to generate parses. 

274 return cmd

276 @staticmethod 

277 def _execute(cmd, verbose=False): 

278 output = None if verbose else subprocess.PIPE 

279 p = subprocess.Popen(cmd, stdout=output, stderr=output) 

280 return p.wait()

282 def train(self, depgraphs, verbose=False): 

283 """ 

284 Train MaltParser from a list of ``DependencyGraph`` objects 

285 

286 :param depgraphs: list of ``DependencyGraph`` objects for training input data 

287 :type depgraphs: DependencyGraph 

288 """ 

289 

290 # Write the conll_str to malt_train.conll file in /tmp/ 

291 with tempfile.NamedTemporaryFile( 

292 prefix='malt_train.conll.', dir=self.working_dir, mode='w', delete=False 

293 ) as input_file: 

294 input_str = '\n'.join(dg.to_conll(10) for dg in depgraphs) 

295 input_file.write(text_type(input_str)) 

296 # Trains the model with the malt_train.conll 

297 self.train_from_file(input_file.name, verbose=verbose) 

298 # Removes the malt_train.conll once training finishes. 

299 os.remove(input_file.name)

301 def train_from_file(self, conll_file, verbose=False): 

302 """ 

303 Train MaltParser from a file 

304 :param conll_file: str for the filename of the training input data 

305 :type conll_file: str 

306 """ 

307 

308 # If conll_file is a ZipFilePathPointer, 

309 # then we need to do some extra massaging 

310 if isinstance(conll_file, ZipFilePathPointer): 

311 with tempfile.NamedTemporaryFile( 

312 prefix='malt_train.conll.', dir=self.working_dir, mode='w', delete=False 

313 ) as input_file: 

314 with conll_file.open() as conll_input_file: 

315 conll_str = conll_input_file.read() 

316 input_file.write(text_type(conll_str)) 

317 return self.train_from_file(input_file.name, verbose=verbose) 

318 

319 # Generate command to run maltparser. 

320 cmd = self.generate_malt_command(conll_file, mode="learn") 

321 ret = self._execute(cmd, verbose) 

322 if ret != 0: 

323 raise Exception( 

324 "MaltParser training (%s) failed with exit " 

325 "code %d" % (' '.join(cmd), ret) 

326 ) 

327 self._trained = True
