29 def __init__( 

30 self, initial_tagger, templates, trace=0, deterministic=None, ruleformat="str" 

31 ): 

32 """ 

33 Construct a Brill tagger from a baseline tagger and a 

34 set of templates 

35 

36 :param initial_tagger: the baseline tagger 

37 :type initial_tagger: Tagger 

38 :param templates: templates to be used in training 

39 :type templates: list of Templates 

40 :param trace: verbosity level 

41 :type trace: int 

42 :param deterministic: if True, adjudicate ties deterministically 

43 :type deterministic: bool 

44 :param ruleformat: format of reported Rules 

45 :type ruleformat: str 

46 :return: An untrained BrillTagger 

47 :rtype: BrillTagger 

48 """ 

49 

50 if deterministic is None: 

51 deterministic = trace > 0 

52 self._initial_tagger = initial_tagger 

53 self._templates = templates 

54 self._trace = trace 

55 self._deterministic = deterministic 

56 self._ruleformat = ruleformat 

57 

58 self._tag_positions = None 

59 """Mapping from tags to lists of positions that use that tag.""" 

60 

61 self._rules_by_position = None 

62 """Mapping from positions to the set of rules that are known 

63 to occur at that position. Position is (sentnum, wordnum). 

64 Initially, this will only contain positions where each rule 

65 applies in a helpful way; but when we examine a rule, we'll 

66 extend this list to also include positions where each rule 

67 applies in a harmful or neutral way.""" 

68 

69 self._positions_by_rule = None 

70 """Mapping from rule to position to effect, specifying the 

71 effect that each rule has on the overall score, at each 

72 position. Position is (sentnum, wordnum); and effect is 

73 -1, 0, or 1. As with _rules_by_position, this mapping starts 

74 out only containing rules with positive effects; but when 

75 we examine a rule, we'll extend this mapping to include 

76 the positions where the rule is harmful or neutral.""" 

77 

78 self._rules_by_score = None 

79 """Mapping from scores to the set of rules whose effect on the 

80 overall score is upper bounded by that score. Invariant: 

81 rulesByScore[s] will contain r iff the sum of 

82 _positions_by_rule[r] is s.""" 

83 

84 self._rule_scores = None 

85 """Mapping from rules to upper bounds on their effects on the 

86 overall score. This is the inverse mapping to _rules_by_score. 

87 Invariant: ruleScores[r] = sum(_positions_by_rule[r])""" 

88 

89 self._first_unknown_position = None 

90 """Mapping from rules to the first position where we're unsure 

91 if the rule applies. This records the next position we 

92 need to check to see if the rule messed anything up."""

242 # FIXME: tests in trainer.fast and trainer.brillorig are exact duplicates 

243 

244 # Basic idea: Keep track of the rules that apply at each position. 

245 # And keep track of the positions to which each rule applies. 

246 

247 # Create a new copy of the training corpus, and run the 

248 # initial tagger on it. We will progressively update this 

249 # test corpus to look more like the training corpus. 

250 test_sents = [ 

251 list(self._initial_tagger.tag(untag(sent))) for sent in train_sents 

252 ] 

253 

254 # Collect some statistics on the training process 

255 trainstats = {} 

256 trainstats['min_acc'] = min_acc 

257 trainstats['min_score'] = min_score 

258 trainstats['tokencount'] = sum(len(t) for t in test_sents) 

259 trainstats['sequencecount'] = len(test_sents) 

260 trainstats['templatecount'] = len(self._templates) 

261 trainstats['rulescores'] = [] 

262 trainstats['initialerrors'] = sum( 

263 tag[1] != truth[1] 

264 for paired in zip(test_sents, train_sents) 

265 for (tag, truth) in zip(*paired) 

266 ) 

267 trainstats['initialacc'] = ( 

268 1 - trainstats['initialerrors'] / trainstats['tokencount'] 

269 )

273 "tpls: {templatecount}; min score: {min_score}; min acc: {min_acc})".format( 

274 **trainstats 

275 ) 

276 ) 

277

279 # by the initial tagger, and use those to generate repair 

280 # rules, which are added to the rule mappings.

328 trainstats['rulescores']

331 1 - trainstats['finalerrors'] / trainstats['tokencount'] 

332 )

334 return BrillTagger(self._initial_tagger, rules, trainstats)

336 def _init_mappings(self, test_sents, train_sents): 

337 """ 

338 Initialize the tag position mapping & the rule related 

339 mappings. For each error in test_sents, find new rules that 

340 would correct them, and add them to the rule mappings. 

341 """ 

342 self._tag_positions = defaultdict(list) 

343 self._rules_by_position = defaultdict(set) 

344 self._positions_by_rule = defaultdict(dict) 

345 self._rules_by_score = defaultdict(set) 

346 self._rule_scores = defaultdict(int) 

347 self._first_unknown_position = defaultdict(int) 

348 # Scan through the corpus, initializing the tag_positions 

349 # mapping and all the rule-related mappings. 

350 for sentnum, sent in enumerate(test_sents): 

351 for wordnum, (word, tag) in enumerate(sent): 

352 

353 # Initialize tag_positions 

354 self._tag_positions[tag].append((sentnum, wordnum)) 

355 

356 # If it's an error token, update the rule-related mappings. 

357 correct_tag = train_sents[sentnum][wordnum][1] 

358 if tag != correct_tag: 

359 for rule in self._find_rules(sent, wordnum, correct_tag): 

360 self._update_rule_applies(rule, sentnum, wordnum, train_sents)

362 def _clean(self): 

363 self._tag_positions = None 

364 self._rules_by_position = None 

365 self._positions_by_rule = None 

366 self._rules_by_score = None 

367 self._rule_scores = None 

368 self._first_unknown_position = None

370 def _find_rules(self, sent, wordnum, new_tag): 

371 """ 

372 Use the templates to find rules that apply at index *wordnum* 

373 in the sentence *sent* and generate the tag *new_tag*. 

374 """ 

375 for template in self._templates: 

376 for rule in template.applicable_rules(sent, wordnum, new_tag): 

377 yield rule 

378 

379 def _update_rule_applies(self, rule, sentnum, wordnum, train_sents): 

380 """ 

381 Update the rule data tables to reflect the fact that 

382 *rule* applies at the position *(sentnum, wordnum)*. 

383 """ 

384 pos = sentnum, wordnum 

385 

386 # If the rule is already known to apply here, ignore. 

387 # (This only happens if the position's tag hasn't changed.) 

388 if pos in self._positions_by_rule[rule]: 

389 return 

390 

391 # Update self._positions_by_rule. 

392 correct_tag = train_sents[sentnum][wordnum][1] 

393 if rule.replacement_tag == correct_tag: 

394 self._positions_by_rule[rule][pos] = 1 

395 elif rule.original_tag == correct_tag: 

396 self._positions_by_rule[rule][pos] = -1 

397 else: # was wrong, remains wrong 

398 self._positions_by_rule[rule][pos] = 0 

399 

400 # Update _rules_by_position 

401 self._rules_by_position[pos].add(rule) 

402 

403 # Update _rule_scores. 

404 old_score = self._rule_scores[rule] 

405 self._rule_scores[rule] += self._positions_by_rule[rule][pos] 

406 

407 # Update _rules_by_score. 

408 self._rules_by_score[old_score].discard(rule) 

409 self._rules_by_score[self._rule_scores[rule]].add(rule) 

410 

411 def _update_rule_not_applies(self, rule, sentnum, wordnum): 

412 """ 

413 Update the rule data tables to reflect the fact that *rule* 

414 does not apply at the position *(sentnum, wordnum)*. 

415 """ 

416 pos = sentnum, wordnum 

417 

418 # Update _rule_scores. 

419 old_score = self._rule_scores[rule] 

420 self._rule_scores[rule] -= self._positions_by_rule[rule][pos] 

421 

422 # Update _rules_by_score. 

423 self._rules_by_score[old_score].discard(rule) 

424 self._rules_by_score[self._rule_scores[rule]].add(rule) 

425 

426 # Update _positions_by_rule 

427 del self._positions_by_rule[rule][pos] 

428 self._rules_by_position[pos].remove(rule) 

429 

430 # Optional addition: if the rule now applies nowhere, delete 

431 # all its dictionary entries.

433 def _best_rule(self, train_sents, test_sents, min_score, min_acc): 

434 """ 

435 Find the next best rule. This is done by repeatedly taking a 

436 rule with the highest score and stepping through the corpus to 

437 see where it applies. When it makes an error (decreasing its 

438 score) it's bumped down, and we try a new rule with the 

439 highest score. When we find a rule which has the highest 

440 score *and* which has been tested against the entire corpus, we 

441 can conclude that it's the next best rule. 

442 """ 

443 for max_score in sorted(self._rules_by_score.keys(), reverse=True): 

444 if len(self._rules_by_score) == 0: 

445 return None 

446 if max_score < min_score or max_score <= 0: 

447 return None 

448 best_rules = list(self._rules_by_score[max_score]) 

449 if self._deterministic: 

450 best_rules.sort(key=repr) 

451 for rule in best_rules: 

452 positions = self._tag_positions[rule.original_tag] 

453 

454 unk = self._first_unknown_position.get(rule, (0, -1)) 

455 start = bisect.bisect_left(positions, unk) 

456 

457 for i in range(start, len(positions)): 

458 sentnum, wordnum = positions[i] 

459 if rule.applies(test_sents[sentnum], wordnum): 

460 self._update_rule_applies(rule, sentnum, wordnum, train_sents) 

461 if self._rule_scores[rule] < max_score: 

462 self._first_unknown_position[rule] = (sentnum, wordnum + 1) 

463 break # The update demoted the rule. 

464 

465 if self._rule_scores[rule] == max_score: 

466 self._first_unknown_position[rule] = (len(train_sents) + 1, 0) 

467 # optimization: if no min_acc threshold given, don't bother computing accuracy 

468 if min_acc is None: 

469 return rule 

470 else: 

471 changes = self._positions_by_rule[rule].values() 

472 num_fixed = len([c for c in changes if c == 1]) 

473 num_broken = len([c for c in changes if c == -1]) 

474 # acc here is fixed/(fixed+broken); could also be 

475 # fixed/(fixed+broken+other) == num_fixed/len(changes) 

476 acc = num_fixed / (num_fixed + num_broken) 

477 if acc >= min_acc: 

478 return rule 

479 # else: rule too inaccurate, discard and try next 

480 

481 # We demoted (or skipped due to < min_acc, if that was given) 

482 # all the rules with score==max_score. 

483 

484 assert min_acc is not None or not self._rules_by_score[max_score] 

485 if not self._rules_by_score[max_score]: 

486 del self._rules_by_score[max_score]

488 def _apply_rule(self, rule, test_sents): 

489 """ 

490 Update *test_sents* by applying *rule* everywhere where its 

491 conditions are met. 

492 """ 

493 update_positions = set(self._positions_by_rule[rule]) 

494 new_tag = rule.replacement_tag 

495 

496 if self._trace > 3: 

497 self._trace_apply(len(update_positions)) 

498 

499 # Update test_sents. 

500 for (sentnum, wordnum) in update_positions: 

501 text = test_sents[sentnum][wordnum][0] 

502 test_sents[sentnum][wordnum] = (text, new_tag) 

503 

504 def _update_tag_positions(self, rule): 

505 """ 

506 Update _tag_positions to reflect the changes to tags that are 

507 made by *rule*. 

508 """ 

509 # Update the tag index. 

510 for pos in self._positions_by_rule[rule]: 

511 # Delete the old tag. 

512 old_tag_positions = self._tag_positions[rule.original_tag] 

513 old_index = bisect.bisect_left(old_tag_positions, pos) 

514 del old_tag_positions[old_index] 

515 # Insert the new tag. 

516 new_tag_positions = self._tag_positions[rule.replacement_tag] 

517 bisect.insort_left(new_tag_positions, pos)

519 def _update_rules(self, rule, train_sents, test_sents): 

520 """ 

521 Check if we should add or remove any rules from consideration, 

522 given the changes made by *rule*. 

523 """ 

524 # Collect a list of all positions that might be affected. 

525 neighbors = set() 

526 for sentnum, wordnum in self._positions_by_rule[rule]: 

527 for template in self._templates: 

528 n = template.get_neighborhood(test_sents[sentnum], wordnum) 

529 neighbors.update([(sentnum, i) for i in n]) 

530 

531 # Update the rules at each position. 

532 num_obsolete = num_new = num_unseen = 0 

533 for sentnum, wordnum in neighbors: 

534 test_sent = test_sents[sentnum] 

535 correct_tag = train_sents[sentnum][wordnum][1] 

536 

537 # Check if the change causes any rule at this position to 

538 # stop matching; if so, then update our rule mappings 

539 # accordingly. 

540 old_rules = set(self._rules_by_position[sentnum, wordnum]) 

541 for old_rule in old_rules: 

542 if not old_rule.applies(test_sent, wordnum): 

543 num_obsolete += 1 

544 self._update_rule_not_applies(old_rule, sentnum, wordnum) 

545 

546 # Check if the change causes our templates to propose any 

547 # new rules for this position. 

548 for template in self._templates: 

549 for new_rule in template.applicable_rules( 

550 test_sent, wordnum, correct_tag 

551 ): 

552 if new_rule not in old_rules: 

553 num_new += 1 

554 if new_rule not in self._rule_scores: 

555 num_unseen += 1 

556 old_rules.add(new_rule) 

557 self._update_rule_applies( 

558 new_rule, sentnum, wordnum, train_sents 

559 ) 

560 

561 # We may have caused other rules to match here, that are 

562 # not proposed by our templates -- in particular, rules 

563 # that are harmful or neutral. We therefore need to 

564 # update any rule whose first_unknown_position is past 

565 # this rule.
