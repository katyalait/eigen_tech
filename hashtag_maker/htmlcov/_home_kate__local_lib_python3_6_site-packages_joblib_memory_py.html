74def register_store_backend(backend_name, backend): 

75 """Extend available store backends. 

76 

77 The Memory, MemorizeResult and MemorizeFunc objects are designed to be 

78 agnostic to the type of store used behind. By default, the local file 

79 system is used but this function gives the possibility to extend joblib's 

80 memory pattern with other types of storage such as cloud storage (S3, GCS, 

81 OpenStack, HadoopFS, etc) or blob DBs. 

82 

83 Parameters 

84 ---------- 

85 backend_name: str 

86 The name identifying the store backend being registered. For example, 

87 'local' is used with FileSystemStoreBackend. 

88 backend: StoreBackendBase subclass 

89 The name of a class that implements the StoreBackendBase interface. 

90 

91 """ 

92 if not isinstance(backend_name, _basestring): 

93 raise ValueError("Store backend name should be a string, " 

94 "'{0}' given.".format(backend_name)) 

95 if backend is None or not issubclass(backend, StoreBackendBase): 

96 raise ValueError("Store backend should inherit " 

97 "StoreBackendBase, " 

98 "'{0}' given.".format(backend)) 

99 

100 _STORE_BACKENDS[backend_name] = backend

103def _store_backend_factory(backend, location, verbose=0, backend_options=None): 

104 """Return the correct store object for the given location.""" 

105 if backend_options is None: 

106 backend_options = {} 

107 

108 if (sys.version_info[:2] >= (3, 4) and isinstance(location, pathlib.Path)): 

109 location = str(location) 

110 

111 if isinstance(location, StoreBackendBase): 

112 return location 

113 elif isinstance(location, _basestring): 

114 obj = None 

115 location = os.path.expanduser(location) 

116 # The location is not a local file system, we look in the 

117 # registered backends if there's one matching the given backend 

118 # name. 

119 for backend_key, backend_obj in _STORE_BACKENDS.items(): 

120 if backend == backend_key: 

121 obj = backend_obj() 

122 

123 # By default, we assume the FileSystemStoreBackend can be used if no 

124 # matching backend could be found. 

125 if obj is None: 

126 raise TypeError('Unknown location {0} or backend {1}'.format( 

127 location, backend)) 

128 

129 # The store backend is configured with the extra named parameters, 

130 # some of them are specific to the underlying store backend. 

131 obj.configure(location, verbose=verbose, 

132 backend_options=backend_options) 

133 return obj 

134 elif location is not None: 

135 warnings.warn( 

136 "Instanciating a backend using a {} as a location is not " 

137 "supported by joblib. Returning None instead.".format( 

138 location.__class__.__name__), UserWarning) 

139 

140 

141 return None

455 def _cached_call(self, args, kwargs, shelving=False): 

456 """Call wrapped function and cache result, or read cache if available. 

457 

458 This function returns the wrapped function output and some metadata. 

459 

460 Arguments: 

461 ---------- 

462 

463 args, kwargs: list and dict 

464 input arguments for wrapped function 

465 

466 shelving: bool 

467 True when called via the call_and_shelve function. 

468 

469 

470 Returns 

471 ------- 

472 output: value or tuple or None 

473 Output of the wrapped function. 

474 If shelving is True and the call has been already cached, 

475 output is None. 

476 

477 argument_hash: string 

478 Hash of function arguments. 

479 

480 metadata: dict 

481 Some metadata about wrapped function call (see _persist_input()). 

482 """ 

483 func_id, args_id = self._get_output_identifiers(*args, **kwargs) 

484 metadata = None 

485 msg = None 

486 

487 # Wether or not the memorized function must be called 

488 must_call = False 

489 

490 # FIXME: The statements below should be try/excepted 

491 # Compare the function code with the previous to see if the 

492 # function code has changed 

493 if not (self._check_previous_func_code(stacklevel=4) and 

494 self.store_backend.contains_item([func_id, args_id])): 

495 if self._verbose > 10: 

496 _, name = get_func_name(self.func) 

497 self.warn('Computing func {0}, argument hash {1} ' 

498 'in location {2}' 

499 .format(name, args_id, 

500 self.store_backend. 

501 get_cached_func_info([func_id])['location'])) 

502 must_call = True 

503 else: 

504 try: 

505 t0 = time.time() 

506 if self._verbose: 

507 msg = _format_load_msg(func_id, args_id, 

508 timestamp=self.timestamp, 

509 metadata=metadata) 

510 

511 if not shelving: 

512 # When shelving, we do not need to load the output 

513 out = self.store_backend.load_item( 

514 [func_id, args_id], 

515 msg=msg, 

516 verbose=self._verbose) 

517 else: 

518 out = None 

519 

520 if self._verbose > 4: 

521 t = time.time() - t0 

522 _, name = get_func_name(self.func) 

523 msg = '%s cache loaded - %s' % (name, format_time(t)) 

524 print(max(0, (80 - len(msg))) * '_' + msg) 

525 except Exception: 

526 # XXX: Should use an exception logger 

527 _, signature = format_signature(self.func, *args, **kwargs) 

528 self.warn('Exception while loading results for ' 

529 '{}\n {}'.format(signature, traceback.format_exc())) 

530 

531 must_call = True 

532 

533 if must_call: 

534 out, metadata = self.call(*args, **kwargs) 

535 if self.mmap_mode is not None: 

536 # Memmap the output at the first call to be consistent with 

537 # later calls 

538 if self._verbose: 

539 msg = _format_load_msg(func_id, args_id, 

540 timestamp=self.timestamp, 

541 metadata=metadata) 

542 out = self.store_backend.load_item([func_id, args_id], msg=msg, 

543 verbose=self._verbose) 

544 

545 return (out, args_id, metadata)

548 """Call wrapped function, cache result and return a reference. 

549 

550 This method returns a reference to the cached result instead of the

597 def _write_func_code(self, func_code, first_line): 

598 """ Write the function code and the filename to a file. 

599 """ 

600 # We store the first line because the filename and the function 

601 # name is not always enough to identify a function: people 

602 # sometimes have several functions named the same way in a 

603 # file. This is bad practice, but joblib should be robust to bad 

604 # practice. 

605 func_id = _build_func_identifier(self.func) 

606 func_code = u'%s %i\n%s' % (FIRST_LINE_TEXT, first_line, func_code) 

607 self.store_backend.store_cached_func_code([func_id], func_code) 

608 

609 # Also store in the in-memory store of function hashes 

610 is_named_callable = False 

611 if PY3_OR_LATER: 

612 is_named_callable = (hasattr(self.func, '__name__') and 

613 self.func.__name__ != '<lambda>') 

614 else: 

615 is_named_callable = (hasattr(self.func, 'func_name') and 

616 self.func.func_name != '<lambda>') 

617 if is_named_callable: 

618 # Don't do this for lambda functions or strange callable 

619 # objects, as it ends up being too fragile 

620 func_hash = self._hash_func() 

621 try: 

622 _FUNCTION_HASHES[self.func] = func_hash 

623 except TypeError: 

624 # Some callable are not hashable 

625 pass

627 def _check_previous_func_code(self, stacklevel=2): 

628 """ 

629 stacklevel is the depth a which this function is called, to 

630 issue useful warnings to the user. 

631 """ 

632 # First check if our function is in the in-memory store. 

633 # Using the in-memory store not only makes things faster, but it 

634 # also renders us robust to variations of the files when the 

635 # in-memory version of the code does not vary 

636 try: 

637 if self.func in _FUNCTION_HASHES: 

638 # We use as an identifier the id of the function and its 

639 # hash. This is more likely to falsely change than have hash 

640 # collisions, thus we are on the safe side. 

641 func_hash = self._hash_func() 

642 if func_hash == _FUNCTION_HASHES[self.func]: 

643 return True 

644 except TypeError: 

645 # Some callables are not hashable 

646 pass 

647 

648 # Here, we go through some effort to be robust to dynamically 

649 # changing code and collision. We cannot inspect.getsource 

650 # because it is not reliable when using IPython's magic "%run". 

651 func_code, source_file, first_line = get_func_code(self.func) 

652 func_id = _build_func_identifier(self.func) 

653 

654 try: 

655 old_func_code, old_first_line =\ 

656 extract_first_line( 

657 self.store_backend.get_cached_func_code([func_id])) 

658 except (IOError, OSError): # some backend can also raise OSError 

659 self._write_func_code(func_code, first_line) 

660 return False 

661 if old_func_code == func_code: 

662 return True 

663 

664 # We have differing code, is this because we are referring to 

665 # different functions, or because the function we are referring to has 

666 # changed? 

667 

668 _, func_name = get_func_name(self.func, resolv_alias=False, 

669 win_characters=False) 

670 if old_first_line == first_line == -1 or func_name == '<lambda>': 

671 if not first_line == -1: 

672 func_description = ("{0} ({1}:{2})" 

673 .format(func_name, source_file, 

674 first_line)) 

675 else: 

676 func_description = func_name 

677 warnings.warn(JobLibCollisionWarning( 

678 "Cannot detect name collisions for function '{0}'" 

679 .format(func_description)), stacklevel=stacklevel) 

680 

681 # Fetch the code at the old location and compare it. If it is the 

682 # same than the code store, we have a collision: the code in the 

683 # file has not changed, but the name we have is pointing to a new 

684 # code block. 

685 if not old_first_line == first_line and source_file is not None: 

686 possible_collision = False 

687 if os.path.exists(source_file): 

688 _, func_name = get_func_name(self.func, resolv_alias=False) 

689 num_lines = len(func_code.split('\n')) 

690 with open_py_source(source_file) as f: 

691 on_disk_func_code = f.readlines()[ 

692 old_first_line - 1:old_first_line - 1 + num_lines - 1] 

693 on_disk_func_code = ''.join(on_disk_func_code) 

694 possible_collision = (on_disk_func_code.rstrip() == 

695 old_func_code.rstrip()) 

696 else: 

697 possible_collision = source_file.startswith('<doctest ') 

698 if possible_collision: 

699 warnings.warn(JobLibCollisionWarning( 

700 'Possible name collisions between functions ' 

701 "'%s' (%s:%i) and '%s' (%s:%i)" % 

702 (func_name, source_file, old_first_line, 

703 func_name, source_file, first_line)), 

704 stacklevel=stacklevel) 

705 

706 # The function has changed, wipe the cache directory. 

707 # XXX: Should be using warnings, and giving stacklevel 

708 if self._verbose > 10: 

709 _, func_name = get_func_name(self.func, resolv_alias=False) 

710 self.warn("Function {0} (identified by {1}) has changed" 

711 ".".format(func_name, func_id)) 

712 self.clear(warn=True) 

713 return False 

714 

715 def clear(self, warn=True): 

716 """Empty the function's cache.""" 

717 func_id = _build_func_identifier(self.func) 

718 

719 if self._verbose > 0 and warn: 

720 self.warn("Clearing function cache identified by %s" % func_id) 

721 self.store_backend.clear_path([func_id, ]) 

722 

723 func_code, _, first_line = get_func_code(self.func) 

724 self._write_func_code(func_code, first_line)

726 def call(self, *args, **kwargs): 

727 """ Force the execution of the function with the given arguments and 

728 persist the output values. 

729 """ 

730 start_time = time.time() 

731 func_id, args_id = self._get_output_identifiers(*args, **kwargs) 

732 if self._verbose > 0: 

733 print(format_call(self.func, args, kwargs)) 

734 output = self.func(*args, **kwargs) 

735 self.store_backend.dump_item( 

736 [func_id, args_id], output, verbose=self._verbose) 

737 

738 duration = time.time() - start_time 

739 metadata = self._persist_input(duration, args, kwargs) 

740 

741 if self._verbose > 0: 

742 _, name = get_func_name(self.func) 

743 msg = '%s - %s' % (name, format_time(duration)) 

744 print(max(0, (80 - len(msg))) * '_' + msg) 

745 return output, metadata 

746 

747 def _persist_input(self, duration, args, kwargs, this_duration_limit=0.5): 

748 """ Save a small summary of the call using json format in the 

749 output directory. 

750 

751 output_dir: string 

752 directory where to write metadata. 

753 

754 duration: float 

755 time taken by hashing input arguments, calling the wrapped 

756 function and persisting its output. 

757 

758 args, kwargs: list and dict 

759 input arguments for wrapped function 

760 

761 this_duration_limit: float 

762 Max execution time for this function before issuing a warning. 

763 """ 

764 start_time = time.time() 

765 argument_dict = filter_args(self.func, self.ignore, 

766 args, kwargs) 

767 

768 input_repr = dict((k, repr(v)) for k, v in argument_dict.items()) 

769 # This can fail due to race-conditions with multiple 

770 # concurrent joblibs removing the file or the directory 

771 metadata = {"duration": duration, "input_args": input_repr} 

772 

773 func_id, args_id = self._get_output_identifiers(*args, **kwargs) 

774 self.store_backend.store_metadata([func_id, args_id], metadata) 

775 

776 this_duration = time.time() - start_time 

777 if this_duration > this_duration_limit: 

778 # This persistence should be fast. It will not be if repr() takes 

779 # time and its output is large, because json.dump will have to 

780 # write a large file. This should not be an issue with numpy arrays 

781 # for which repr() always output a short representation, but can 

782 # be with complex dictionaries. Fixing the problem should be a 

783 # matter of replacing repr() above by something smarter. 

784 warnings.warn("Persisting input arguments took %.2fs to run.\n" 

785 "If this happens often in your code, it can cause " 

786 "performance problems \n" 

787 "(results will be correct in all cases). \n" 

788 "The reason for this is probably some large input " 

789 "arguments for a wrapped\n" 

790 " function (e.g. large strings).\n" 

791 "THIS IS A JOBLIB ISSUE. If you can, kindly provide " 

792 "the joblib's team with an\n" 

793 " example so that they can fix the problem." 

794 % this_duration, stacklevel=5) 

795 return metadata

813class Memory(Logger): 

814 """ A context object for caching a function's return value each time it 

815 is called with the same input arguments. 

816 

817 All values are cached on the filesystem, in a deep directory 

818 structure. 

819 

820 Read more in the :ref:`User Guide <memory>`. 

821 

822 Parameters 

823 ---------- 

824 location: str or None 

825 The path of the base directory to use as a data store 

826 or None. If None is given, no caching is done and 

827 the Memory object is completely transparent. This option 

828 replaces cachedir since version 0.12. 

829 

830 backend: str, optional 

831 Type of store backend for reading/writing cache files. 

832 Default: 'local'. 

833 The 'local' backend is using regular filesystem operations to 

834 manipulate data (open, mv, etc) in the backend. 

835 

836 cachedir: str or None, optional 

837 

838 .. deprecated: 0.12 

839 'cachedir' has been deprecated in 0.12 and will be 

840 removed in 0.14. Use the 'location' parameter instead. 

841 

842 mmap_mode: {None, 'r+', 'r', 'w+', 'c'}, optional 

843 The memmapping mode used when loading from cache 

844 numpy arrays. See numpy.load for the meaning of the 

845 arguments. 

846 

847 compress: boolean, or integer, optional 

848 Whether to zip the stored data on disk. If an integer is 

849 given, it should be between 1 and 9, and sets the amount 

850 of compression. Note that compressed arrays cannot be 

851 read by memmapping. 

852 

853 verbose: int, optional 

854 Verbosity flag, controls the debug messages that are issued 

855 as functions are evaluated. 

856 

857 bytes_limit: int, optional 

858 Limit in bytes of the size of the cache. 

859 

860 backend_options: dict, optional 

861 Contains a dictionnary of named parameters used to configure 

862 the store backend. 

863 """ 

864 # ------------------------------------------------------------------------ 

865 # Public interface 

866 # ------------------------------------------------------------------------ 

867

869 mmap_mode=None, compress=False, verbose=1, bytes_limit=None,

872 Logger.__init__(self)

874 self.mmap_mode = mmap_mode

877 self.backend = backend 

878 self.compress = compress 

879 if backend_options is None: 

880 backend_options = {} 

881 self.backend_options = backend_options 

882 

883 if compress and mmap_mode is not None: 

884 warnings.warn('Compressed results cannot be memmapped', 

885 stacklevel=2) 

886 if cachedir is not None: 

887 if location is not None: 

888 raise ValueError( 

889 'You set both "location={0!r} and "cachedir={1!r}". ' 

890 "'cachedir' has been deprecated in version " 

891 "0.12 and will be removed in version 0.14.\n" 

892 'Please only set "location={0!r}"'.format( 

893 location, cachedir)) 

894 

895 warnings.warn(

904 if isinstance(location, _basestring):

923 def cache(self, func=None, ignore=None, verbose=None, mmap_mode=False): 

924 """ Decorates the given function func to only compute its return 

925 value for input arguments not cached on disk. 

926 

927 Parameters 

928 ---------- 

929 func: callable, optional 

930 The function to be decorated 

931 ignore: list of strings 

932 A list of arguments name to ignore in the hashing 

933 verbose: integer, optional 

934 The verbosity mode of the function. By default that 

935 of the memory object is used. 

936 mmap_mode: {None, 'r+', 'r', 'w+', 'c'}, optional 

937 The memmapping mode used when loading from cache 

938 numpy arrays. See numpy.load for the meaning of the 

939 arguments. By default that of the memory object is used. 

940 

941 Returns 

942 ------- 

943 decorated_func: MemorizedFunc object 

944 The returned object is a MemorizedFunc object, that is 

945 callable (behaves like a function), but offers extra 

946 methods for cache lookup and management. See the 

947 documentation for :class:`joblib.memory.MemorizedFunc`. 

948 """ 

949 if func is None: 

950 # Partial application, to be able to specify extra keyword 

951 # arguments in decorators 

952 return functools.partial(self.cache, ignore=ignore, 

953 verbose=verbose, mmap_mode=mmap_mode) 

954 if self.store_backend is None: 

955 return NotMemorizedFunc(func) 

956 if verbose is None: 

957 verbose = self._verbose 

958 if mmap_mode is False: 

959 mmap_mode = self.mmap_mode 

960 if isinstance(func, MemorizedFunc): 

961 func = func.func 

962 return MemorizedFunc(func, location=self.store_backend, 

963 backend=self.backend, 

964 ignore=ignore, mmap_mode=mmap_mode, 

965 compress=self.compress, 

966 verbose=verbose, timestamp=self.timestamp)

970 """ 

971 if warn: 

972 self.warn('Flushing completely the cache') 

973 if self.store_backend is not None: 

974 self.store_backend.clear()
