24def concurrency_safe_write(object_to_write, filename, write_func): 

25 """Writes an object into a unique file in a concurrency-safe way.""" 

26 thread_id = id(threading.current_thread()) 

27 temporary_filename = '{}.thread-{}-pid-{}'.format( 

28 filename, thread_id, os.getpid()) 

29 write_func(object_to_write, temporary_filename) 

30 

31 return temporary_filename

139class StoreBackendMixin(object): 

140 """Class providing all logic for managing the store in a generic way. 

141 

142 The StoreBackend subclass has to implement 3 methods: create_location, 

143 clear_location and configure. The StoreBackend also has to provide 

144 a private _open_item, _item_exists and _move_item methods. The _open_item 

145 method has to have the same signature as the builtin open and return a 

146 file-like object.

149 def load_item(self, path, verbose=1, msg=None): 

150 """Load an item from the store given its path as a list of 

151 strings.""" 

152 full_path = os.path.join(self.location, *path) 

153 

154 if verbose > 1: 

155 if verbose < 10: 

156 print('{0}...'.format(msg)) 

157 else: 

158 print('{0} from {1}'.format(msg, full_path)) 

159 

160 mmap_mode = (None if not hasattr(self, 'mmap_mode') 

161 else self.mmap_mode) 

162 

163 filename = os.path.join(full_path, 'output.pkl') 

164 if not self._item_exists(filename): 

165 raise KeyError("Non-existing item (may have been " 

166 "cleared).\nFile %s does not exist" % filename) 

167 

168 # file-like object cannot be used when mmap_mode is set 

169 if mmap_mode is None: 

170 with self._open_item(filename, "rb") as f: 

171 item = numpy_pickle.load(f) 

172 else: 

173 item = numpy_pickle.load(filename, mmap_mode=mmap_mode) 

174 return item

280 def reduce_store_size(self, bytes_limit): 

281 """Reduce store size to keep it under the given bytes limit.""" 

282 items_to_delete = self._get_items_to_delete(bytes_limit) 

283 

284 for item in items_to_delete: 

285 if self.verbose > 10: 

286 print('Deleting item {0}'.format(item)) 

287 try: 

288 self.clear_location(item.path) 

289 except (OSError, IOError): 

290 # Even with ignore_errors=True shutil.rmtree 

291 # can raise OSError (IOError in python 2) with 

292 # [Errno 116] Stale file handle if another process 

293 # has deleted the folder already. 

294 pass

296 def _get_items_to_delete(self, bytes_limit): 

297 """Get items to delete to keep the store under a size limit.""" 

298 if isinstance(bytes_limit, _basestring): 

299 bytes_limit = memstr_to_bytes(bytes_limit) 

300 

301 items = self.get_items() 

302 size = sum(item.size for item in items) 

303 

304 to_delete_size = size - bytes_limit 

305 if to_delete_size < 0: 

306 return [] 

307 

308 # We want to delete first the cache items that were accessed a 

309 # long time ago 

310 items.sort(key=operator.attrgetter('last_access')) 

311 

312 items_to_delete = [] 

313 size_so_far = 0 

314 

315 for item in items: 

316 if size_so_far > to_delete_size: 

317 break 

318 

319 items_to_delete.append(item) 

320 size_so_far += item.size 

321 

322 return items_to_delete

336class FileSystemStoreBackend(StoreBackendBase, StoreBackendMixin): 

337 """A StoreBackend used with local or network file systems.""" 

338 

339 _open_item = staticmethod(open) 

340 _item_exists = staticmethod(os.path.exists) 

341 _move_item = staticmethod(concurrency_safe_rename)

343 def clear_location(self, location): 

344 """Delete location on store.""" 

345 if (location == self.location): 

346 rm_subdirs(location) 

347 else: 

348 shutil.rmtree(location, ignore_errors=True)

350 def create_location(self, location): 

351 """Create object location on store""" 

352 mkdirp(location)

354 def get_items(self): 

355 """Returns the whole list of items available in the store.""" 

356 items = [] 

357 

358 for dirpath, _, filenames in os.walk(self.location): 

359 is_cache_hash_dir = re.match('[a-f0-9]{32}', 

360 os.path.basename(dirpath)) 

361 

362 if is_cache_hash_dir: 

363 output_filename = os.path.join(dirpath, 'output.pkl') 

364 try: 

365 last_access = os.path.getatime(output_filename) 

366 except OSError: 

367 try: 

368 last_access = os.path.getatime(dirpath) 

369 except OSError: 

370 # The directory has already been deleted 

371 continue 

372 

373 last_access = datetime.datetime.fromtimestamp(last_access) 

374 try: 

375 full_filenames = [os.path.join(dirpath, fn) 

376 for fn in filenames] 

377 dirsize = sum(os.path.getsize(fn) 

378 for fn in full_filenames) 

379 except OSError: 

380 # Either output_filename or one of the files in 

381 # dirpath does not exist any more. We assume this 

382 # directory is being cleaned by another process already 

383 continue 

384 

385 items.append(CacheItemInfo(dirpath, dirsize, 

386 last_access)) 

387 

388 return items

390 def configure(self, location, verbose=1, backend_options=None): 

391 """Configure the store backend. 

392 

393 For this backend, valid store options are 'compress' and 'mmap_mode' 

394 """ 

395 if backend_options is None: 

396 backend_options = {} 

397 

398 # setup location directory 

399 self.location = location 

400 if not os.path.exists(self.location): 

401 mkdirp(self.location) 

402 

403 # item can be stored compressed for faster I/O 

404 self.compress = backend_options.get('compress', False) 

405 

406 # FileSystemStoreBackend can be used with mmap_mode options under 

407 # certain conditions. 

408 mmap_mode = backend_options.get('mmap_mode') 

409 if self.compress and mmap_mode is not None: 

410 warnings.warn('Compressed items cannot be memmapped in a ' 

411 'filesystem store. Option will be ignored.', 

412 stacklevel=2) 

413 

414 self.mmap_mode = mmap_mode 

415 self.verbose = verbose
