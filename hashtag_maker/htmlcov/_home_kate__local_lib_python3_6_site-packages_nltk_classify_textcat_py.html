10""" 

11A module for language identification using the TextCat algorithm. 

12An implementation of the text categorization algorithm 

13presented in Cavnar, W. B. and J. M. Trenkle, 

14"N-Gram-Based Text Categorization". 

15 

16The algorithm takes advantage of Zipf's law and uses 

17n-gram frequencies to profile languages and text-yet to 

18be identified-then compares using a distance measure. 

19 

20Language n-grams are provided by the "An Crubadan" 

21project. A corpus reader was created separately to read 

22those files. 

23 

24For details regarding the algorithm, see: 

25http://www.let.rug.nl/~vannoord/TextCat/textcat.pdf 

26 

27For details about An Crubadan, see: 

28http://borel.slu.edu/crubadan/index.html 

29""" 

30

55class TextCat(object): 

56 

57 _corpus = None 

58 fingerprints = {} 

59 _START_CHAR = "<" 

60 _END_CHAR = ">" 

61 

62 last_distances = {} 

63 

64 def __init__(self): 

65 if not re: 

66 raise EnvironmentError( 

67 "classify.textcat requires the regex module that " 

68 "supports unicode. Try '$ pip install regex' and " 

69 "see https://pypi.python.org/pypi/regex for " 

70 "further details." 

71 ) 

72 

73 from nltk.corpus import crubadan 

74 

75 self._corpus = crubadan 

76 # Load all language ngrams into cache 

77 for lang in self._corpus.langs(): 

78 self._corpus.lang_freq(lang) 

79 

80 def remove_punctuation(self, text): 

81 ''' Get rid of punctuation except apostrophes ''' 

82 return re.sub(r"[^\P{P}\']+", "", text) 

83 

84 def profile(self, text): 

85 ''' Create FreqDist of trigrams within text ''' 

86 from nltk import word_tokenize, FreqDist 

87 

88 clean_text = self.remove_punctuation(text) 

89 tokens = word_tokenize(clean_text) 

90 

91 fingerprint = FreqDist() 

92 for t in tokens: 

93 token_trigram_tuples = trigrams(self._START_CHAR + t + self._END_CHAR) 

94 token_trigrams = [''.join(tri) for tri in token_trigram_tuples] 

95 

96 for cur_trigram in token_trigrams: 

97 if cur_trigram in fingerprint: 

98 fingerprint[cur_trigram] += 1 

99 else: 

100 fingerprint[cur_trigram] = 1 

101 

102 return fingerprint 

103 

104 def calc_dist(self, lang, trigram, text_profile): 

105 ''' Calculate the "out-of-place" measure between the 

106 text and language profile for a single trigram ''' 

107 

108 lang_fd = self._corpus.lang_freq(lang) 

109 dist = 0 

110 

111 if trigram in lang_fd: 

112 idx_lang_profile = list(lang_fd.keys()).index(trigram) 

113 idx_text = list(text_profile.keys()).index(trigram) 

114 

115 # print(idx_lang_profile, ", ", idx_text) 

116 dist = abs(idx_lang_profile - idx_text) 

117 else: 

118 # Arbitrary but should be larger than 

119 # any possible trigram file length 

120 # in terms of total lines 

121 if PY3: 

122 dist = maxsize 

123 else: 

124 dist = maxint 

125 

126 return dist 

127 

128 def lang_dists(self, text): 

129 ''' Calculate the "out-of-place" measure between 

130 the text and all languages ''' 

131 

132 distances = {} 

133 profile = self.profile(text) 

134 # For all the languages 

135 for lang in self._corpus._all_lang_freq.keys(): 

136 # Calculate distance metric for every trigram in 

137 # input text to be identified 

138 lang_dist = 0 

139 for trigram in profile: 

140 lang_dist += self.calc_dist(lang, trigram, profile) 

141 

142 distances[lang] = lang_dist 

143 

144 return distances 

145 

146 def guess_language(self, text): 

147 ''' Find the language with the min distance 

148 to the text and return its ISO 639-3 code ''' 

149 self.last_distances = self.lang_dists(text) 

150 

151 return min(self.last_distances, key=self.last_distances.get) 

152 #################################################')

156 from nltk.corpus import udhr

158 langs = [ 

159 'Kurdish-UTF8', 

160 'Abkhaz-UTF8', 

161 'Farsi_Persian-UTF8', 

162 'Hindi-UTF8', 

163 'Hawaiian-UTF8', 

164 'Russian-UTF8', 

165 'Vietnamese-UTF8', 

166 'Serbian_Srpski-UTF8', 

167 'Esperanto-UTF8', 

168 ] 

169 

170 friendly = { 

171 'kmr': 'Northern Kurdish', 

172 'abk': 'Abkhazian', 

173 'pes': 'Iranian Persian', 

174 'hin': 'Hindi', 

175 'haw': 'Hawaiian', 

176 'rus': 'Russian', 

177 'vie': 'Vietnamese', 

178 'srp': 'Serbian', 

179 'epo': 'Esperanto', 

180 } 

181 

182 tc = TextCat() 

183 

184 for cur_lang in langs: 

185 # Get raw data from UDHR corpus 

186 raw_sentences = udhr.sents(cur_lang) 

187 rows = len(raw_sentences) - 1 

188 cols = list(map(len, raw_sentences)) 

189 

190 sample = '' 

191 

192 # Generate a sample text of the language 

193 for i in range(0, rows): 

194 cur_sent = '' 

195 for j in range(0, cols[i]): 

196 cur_sent += ' ' + raw_sentences[i][j] 

197 

198 sample += cur_sent 

199 

200 # Try to detect what it is 

201 print('Language snippet: ' + sample[0:140] + '...') 

202 guess = tc.guess_language(sample) 

203 print('Language detection: %s (%s)' % (guess, friendly[guess])) 

204 print('#' * 140)
