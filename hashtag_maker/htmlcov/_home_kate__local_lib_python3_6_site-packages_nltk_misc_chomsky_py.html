119import textwrap, random 

120from itertools import chain, islice 

121 

122from six.moves import zip 

123 

124 

125def generate_chomsky(times=5, line_length=72): 

126 parts = [] 

127 for part in (leadins, subjects, verbs, objects): 

128 phraselist = list(map(str.strip, part.splitlines())) 

129 random.shuffle(phraselist) 

130 parts.append(phraselist) 

131 output = chain(*islice(zip(*parts), 0, times)) 

132 print(textwrap.fill(" ".join(output), line_length))
