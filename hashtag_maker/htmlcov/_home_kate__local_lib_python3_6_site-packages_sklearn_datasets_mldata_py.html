26@deprecated('mldata_filename was deprecated in version 0.20 and will be '

28def mldata_filename(dataname): 

29 """Convert a raw name for a data set in a mldata.org filename. 

30 

31 .. deprecated:: 0.20 

32 Will be removed in version 0.22 

33 

34 Parameters 

35 ---------- 

36 dataname : str 

37 Name of dataset 

38 

39 Returns 

40 ------- 

41 fname : str 

42 The converted dataname. 

43 """ 

44 dataname = dataname.lower().replace(' ', '-') 

45 return re.sub(r'[().]', '', dataname)

48@deprecated('fetch_mldata was deprecated in version 0.20 and will be removed ' 

49 'in version 0.22. Please use fetch_openml.') 

50def fetch_mldata(dataname, target_name='label', data_name='data', 

51 transpose_data=True, data_home=None): 

52 """Fetch an mldata.org data set 

53 

54 mldata.org is no longer operational. 

55 

56 If the file does not exist yet, it is downloaded from mldata.org . 

57 

58 mldata.org does not have an enforced convention for storing data or 

59 naming the columns in a data set. The default behavior of this function 

60 works well with the most common cases: 

61 

62 1) data values are stored in the column 'data', and target values in the 

63 column 'label' 

64 2) alternatively, the first column stores target values, and the second 

65 data values 

66 3) the data array is stored as `n_features x n_samples` , and thus needs 

67 to be transposed to match the `sklearn` standard 

68 

69 Keyword arguments allow to adapt these defaults to specific data sets 

70 (see parameters `target_name`, `data_name`, `transpose_data`, and 

71 the examples below). 

72 

73 mldata.org data sets may have multiple columns, which are stored in the 

74 Bunch object with their original name. 

75 

76 .. deprecated:: 0.20 

77 Will be removed in version 0.22 

78 

79 Parameters 

80 ---------- 

81 

82 dataname : str 

83 Name of the data set on mldata.org, 

84 e.g.: "leukemia", "Whistler Daily Snowfall", etc. 

85 The raw name is automatically converted to a mldata.org URL . 

86 

87 target_name : optional, default: 'label' 

88 Name or index of the column containing the target values. 

89 

90 data_name : optional, default: 'data' 

91 Name or index of the column containing the data. 

92 

93 transpose_data : optional, default: True 

94 If True, transpose the downloaded data array. 

95 

96 data_home : optional, default: None 

97 Specify another download and cache folder for the data sets. By default 

98 all scikit-learn data is stored in '~/scikit_learn_data' subfolders. 

99 

100 Returns 

101 ------- 

102 

103 data : Bunch 

104 Dictionary-like object, the interesting attributes are: 

105 'data', the data to learn, 'target', the classification labels, 

106 'DESCR', the full description of the dataset, and 

107 'COL_NAMES', the original names of the dataset columns. 

108 """ 

109 

110 # normalize dataset name 

111 dataname = mldata_filename(dataname) 

112 

113 # check if this data set has been already downloaded 

114 data_home = get_data_home(data_home=data_home) 

115 data_home = join(data_home, 'mldata') 

116 if not exists(data_home): 

117 os.makedirs(data_home) 

118 

119 matlab_name = dataname + '.mat' 

120 filename = join(data_home, matlab_name) 

121 

122 # if the file does not exist, download it 

123 if not exists(filename): 

124 urlname = MLDATA_BASE_URL % quote(dataname) 

125 try: 

126 mldata_url = urlopen(urlname) 

127 except HTTPError as e: 

128 if e.code == 404: 

129 e.msg = "Dataset '%s' not found on mldata.org." % dataname 

130 raise 

131 # store Matlab file 

132 try: 

133 with open(filename, 'w+b') as matlab_file: 

134 copyfileobj(mldata_url, matlab_file) 

135 except: 

136 os.remove(filename) 

137 raise 

138 mldata_url.close() 

139 

140 # load dataset matlab file 

141 with open(filename, 'rb') as matlab_file: 

142 matlab_dict = io.loadmat(matlab_file, struct_as_record=True) 

143 

144 # -- extract data from matlab_dict 

145 

146 # flatten column names 

147 col_names = [str(descr[0]) 

148 for descr in matlab_dict['mldata_descr_ordering'][0]] 

149 

150 # if target or data names are indices, transform then into names 

151 if isinstance(target_name, numbers.Integral): 

152 target_name = col_names[target_name] 

153 if isinstance(data_name, numbers.Integral): 

154 data_name = col_names[data_name] 

155 

156 # rules for making sense of the mldata.org data format 

157 # (earlier ones have priority): 

158 # 1) there is only one array => it is "data" 

159 # 2) there are multiple arrays 

160 # a) copy all columns in the bunch, using their column name 

161 # b) if there is a column called `target_name`, set "target" to it, 

162 # otherwise set "target" to first column 

163 # c) if there is a column called `data_name`, set "data" to it, 

164 # otherwise set "data" to second column 

165 

166 dataset = {'DESCR': 'mldata.org dataset: %s' % dataname, 

167 'COL_NAMES': col_names} 

168 

169 # 1) there is only one array => it is considered data 

170 if len(col_names) == 1: 

171 data_name = col_names[0] 

172 dataset['data'] = matlab_dict[data_name] 

173 # 2) there are multiple arrays 

174 else: 

175 for name in col_names: 

176 dataset[name] = matlab_dict[name] 

177 

178 if target_name in col_names: 

179 del dataset[target_name] 

180 dataset['target'] = matlab_dict[target_name] 

181 else: 

182 del dataset[col_names[0]] 

183 dataset['target'] = matlab_dict[col_names[0]] 

184 

185 if data_name in col_names: 

186 del dataset[data_name] 

187 dataset['data'] = matlab_dict[data_name] 

188 else: 

189 del dataset[col_names[1]] 

190 dataset['data'] = matlab_dict[col_names[1]] 

191 

192 # set axes to scikit-learn conventions 

193 if transpose_data: 

194 dataset['data'] = dataset['data'].T 

195 if 'target' in dataset: 

196 if not sp.sparse.issparse(dataset['target']): 

197 dataset['target'] = dataset['target'].squeeze() 

198 

199 return Bunch(**dataset)
