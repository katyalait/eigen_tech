48def row_norms(X, squared=False): 

49 """Row-wise (squared) Euclidean norm of X. 

50 

51 Equivalent to np.sqrt((X * X).sum(axis=1)), but also supports sparse 

52 matrices and does not create an X.shape-sized temporary. 

53 

54 Performs no input validation. 

55 

56 Parameters 

57 ---------- 

58 X : array_like 

59 The input array 

60 squared : bool, optional (default = False) 

61 If True, return squared norms. 

62 

63 Returns 

64 ------- 

65 array_like 

66 The row-wise (squared) Euclidean norm of X. 

67 """ 

68 if sparse.issparse(X): 

69 if not isinstance(X, sparse.csr_matrix): 

70 X = sparse.csr_matrix(X) 

71 norms = csr_row_norms(X) 

72 else: 

73 norms = np.einsum('ij,ij->i', X, X) 

74 

75 if not squared: 

76 np.sqrt(norms, norms) 

77 return norms

145def randomized_range_finder(A, size, n_iter, 

146 power_iteration_normalizer='auto', 

147 random_state=None): 

148 """Computes an orthonormal matrix whose range approximates the range of A. 

149 

150 Parameters 

151 ---------- 

152 A : 2D array 

153 The input data matrix 

154 

155 size : integer 

156 Size of the return array 

157 

158 n_iter : integer 

159 Number of power iterations used to stabilize the result 

160 

161 power_iteration_normalizer : 'auto' (default), 'QR', 'LU', 'none' 

162 Whether the power iterations are normalized with step-by-step 

163 QR factorization (the slowest but most accurate), 'none' 

164 (the fastest but numerically unstable when `n_iter` is large, e.g. 

165 typically 5 or larger), or 'LU' factorization (numerically stable 

166 but can lose slightly in accuracy). The 'auto' mode applies no 

167 normalization if `n_iter` <= 2 and switches to LU otherwise. 

168 

169 .. versionadded:: 0.18 

170 

171 random_state : int, RandomState instance or None, optional (default=None) 

172 The seed of the pseudo random number generator to use when shuffling 

173 the data. If int, random_state is the seed used by the random number 

174 generator; If RandomState instance, random_state is the random number 

175 generator; If None, the random number generator is the RandomState 

176 instance used by `np.random`. 

177 

178 Returns 

179 ------- 

180 Q : 2D array 

181 A (size x size) projection matrix, the range of which 

182 approximates well the range of the input matrix A. 

183 

184 Notes 

185 ----- 

186 

187 Follows Algorithm 4.3 of 

188 Finding structure with randomness: Stochastic algorithms for constructing 

189 approximate matrix decompositions 

190 Halko, et al., 2009 (arXiv:909) https://arxiv.org/pdf/0909.4061.pdf 

191 

192 An implementation of a randomized algorithm for principal component 

193 analysis 

194 A. Szlam et al. 2014 

195 """ 

196 random_state = check_random_state(random_state) 

197 

198 # Generating normal random vectors with shape: (A.shape[1], size) 

199 Q = random_state.normal(size=(A.shape[1], size)) 

200 if A.dtype.kind == 'f': 

201 # Ensure f32 is preserved as f32 

202 Q = Q.astype(A.dtype, copy=False) 

203 

204 # Deal with "auto" mode 

205 if power_iteration_normalizer == 'auto': 

206 if n_iter <= 2: 

207 power_iteration_normalizer = 'none' 

208 else: 

209 power_iteration_normalizer = 'LU' 

210 

211 # Perform power iterations with Q to further 'imprint' the top 

212 # singular vectors of A in Q 

213 for i in range(n_iter): 

214 if power_iteration_normalizer == 'none': 

215 Q = safe_sparse_dot(A, Q) 

216 Q = safe_sparse_dot(A.T, Q) 

217 elif power_iteration_normalizer == 'LU': 

218 Q, _ = linalg.lu(safe_sparse_dot(A, Q), permute_l=True) 

219 Q, _ = linalg.lu(safe_sparse_dot(A.T, Q), permute_l=True) 

220 elif power_iteration_normalizer == 'QR': 

221 Q, _ = linalg.qr(safe_sparse_dot(A, Q), mode='economic') 

222 Q, _ = linalg.qr(safe_sparse_dot(A.T, Q), mode='economic') 

223 

224 # Sample the range of A using by linear projection of Q 

225 # Extract an orthonormal basis 

226 Q, _ = linalg.qr(safe_sparse_dot(A, Q), mode='economic') 

227 return Q

230def randomized_svd(M, n_components, n_oversamples=10, n_iter='auto', 

231 power_iteration_normalizer='auto', transpose='auto', 

232 flip_sign=True, random_state=0): 

233 """Computes a truncated randomized SVD 

234 

235 Parameters 

236 ---------- 

237 M : ndarray or sparse matrix 

238 Matrix to decompose 

239 

240 n_components : int 

241 Number of singular values and vectors to extract. 

242 

243 n_oversamples : int (default is 10) 

244 Additional number of random vectors to sample the range of M so as 

245 to ensure proper conditioning. The total number of random vectors 

246 used to find the range of M is n_components + n_oversamples. Smaller 

247 number can improve speed but can negatively impact the quality of 

248 approximation of singular vectors and singular values. 

249 

250 n_iter : int or 'auto' (default is 'auto') 

251 Number of power iterations. It can be used to deal with very noisy 

252 problems. When 'auto', it is set to 4, unless `n_components` is small 

253 (< .1 * min(X.shape)) `n_iter` in which case is set to 7. 

254 This improves precision with few components. 

255 

256 .. versionchanged:: 0.18 

257 

258 power_iteration_normalizer : 'auto' (default), 'QR', 'LU', 'none' 

259 Whether the power iterations are normalized with step-by-step 

260 QR factorization (the slowest but most accurate), 'none' 

261 (the fastest but numerically unstable when `n_iter` is large, e.g. 

262 typically 5 or larger), or 'LU' factorization (numerically stable 

263 but can lose slightly in accuracy). The 'auto' mode applies no 

264 normalization if `n_iter` <= 2 and switches to LU otherwise. 

265 

266 .. versionadded:: 0.18 

267 

268 transpose : True, False or 'auto' (default) 

269 Whether the algorithm should be applied to M.T instead of M. The 

270 result should approximately be the same. The 'auto' mode will 

271 trigger the transposition if M.shape[1] > M.shape[0] since this 

272 implementation of randomized SVD tend to be a little faster in that 

273 case. 

274 

275 .. versionchanged:: 0.18 

276 

277 flip_sign : boolean, (True by default) 

278 The output of a singular value decomposition is only unique up to a 

279 permutation of the signs of the singular vectors. If `flip_sign` is 

280 set to `True`, the sign ambiguity is resolved by making the largest 

281 loadings for each component in the left singular vectors positive. 

282 

283 random_state : int, RandomState instance or None, optional (default=None) 

284 The seed of the pseudo random number generator to use when shuffling 

285 the data. If int, random_state is the seed used by the random number 

286 generator; If RandomState instance, random_state is the random number 

287 generator; If None, the random number generator is the RandomState 

288 instance used by `np.random`. 

289 

290 Notes 

291 ----- 

292 This algorithm finds a (usually very good) approximate truncated 

293 singular value decomposition using randomization to speed up the 

294 computations. It is particularly fast on large matrices on which 

295 you wish to extract only a small number of components. In order to 

296 obtain further speed up, `n_iter` can be set <=2 (at the cost of 

297 loss of precision). 

298 

299 References 

300 ---------- 

301 * Finding structure with randomness: Stochastic algorithms for constructing 

302 approximate matrix decompositions 

303 Halko, et al., 2009 https://arxiv.org/abs/0909.4061 

304 

305 * A randomized algorithm for the decomposition of matrices 

306 Per-Gunnar Martinsson, Vladimir Rokhlin and Mark Tygert 

307 

308 * An implementation of a randomized algorithm for principal component 

309 analysis 

310 A. Szlam et al. 2014 

311 """ 

312 if isinstance(M, (sparse.lil_matrix, sparse.dok_matrix)): 

313 warnings.warn("Calculating SVD of a {} is expensive. " 

314 "csr_matrix is more efficient.".format( 

315 type(M).__name__), 

316 sparse.SparseEfficiencyWarning) 

317 

318 random_state = check_random_state(random_state) 

319 n_random = n_components + n_oversamples 

320 n_samples, n_features = M.shape 

321 

322 if n_iter == 'auto': 

323 # Checks if the number of iterations is explicitly specified 

324 # Adjust n_iter. 7 was found a good compromise for PCA. See #5299 

325 n_iter = 7 if n_components < .1 * min(M.shape) else 4 

326 

327 if transpose == 'auto': 

328 transpose = n_samples < n_features 

329 if transpose: 

330 # this implementation is a bit faster with smaller shape[1] 

331 M = M.T 

332 

333 Q = randomized_range_finder(M, n_random, n_iter, 

334 power_iteration_normalizer, random_state) 

335 

336 # project M to the (k + p) dimensional space using the basis vectors 

337 B = safe_sparse_dot(Q.T, M) 

338 

339 # compute the SVD on the thin matrix: (k + p) wide 

340 Uhat, s, V = linalg.svd(B, full_matrices=False) 

341 

342 del B 

343 U = np.dot(Q, Uhat) 

344 

345 if flip_sign: 

346 if not transpose: 

347 U, V = svd_flip(U, V) 

348 else: 

349 # In case of transpose u_based_decision=false 

350 # to actually flip based on u and not v. 

351 U, V = svd_flip(U, V, u_based_decision=False) 

352 

353 if transpose: 

354 # transpose back the results according to the input convention 

355 return V[:n_components, :].T, s[:n_components], U[:, :n_components].T 

356 else: 

357 return U[:, :n_components], s[:n_components], V[:n_components, :]

360def weighted_mode(a, w, axis=0): 

361 """Returns an array of the weighted modal (most common) value in a 

362 

363 If there is more than one such value, only the first is returned. 

364 The bin-count for the modal bins is also returned. 

365 

366 This is an extension of the algorithm in scipy.stats.mode. 

367 

368 Parameters 

369 ---------- 

370 a : array_like 

371 n-dimensional array of which to find mode(s). 

372 w : array_like 

373 n-dimensional array of weights for each value 

374 axis : int, optional 

375 Axis along which to operate. Default is 0, i.e. the first axis. 

376 

377 Returns 

378 ------- 

379 vals : ndarray 

380 Array of modal values. 

381 score : ndarray 

382 Array of weighted counts for each mode. 

383 

384 Examples 

385 -------- 

386 >>> from sklearn.utils.extmath import weighted_mode 

387 >>> x = [4, 1, 4, 2, 4, 2] 

388 >>> weights = [1, 1, 1, 1, 1, 1] 

389 >>> weighted_mode(x, weights) 

390 (array([4.]), array([3.])) 

391 

392 The value 4 appears three times: with uniform weights, the result is 

393 simply the mode of the distribution. 

394 

395 >>> weights = [1, 3, 0.5, 1.5, 1, 2] # deweight the 4's 

396 >>> weighted_mode(x, weights) 

397 (array([2.]), array([3.5])) 

398 

399 The value 2 has the highest score: it appears twice with weights of 

400 1.5 and 2: the sum of these is 3.5. 

401 

402 See Also 

403 -------- 

404 scipy.stats.mode 

405 """ 

406 if axis is None: 

407 a = np.ravel(a) 

408 w = np.ravel(w) 

409 axis = 0 

410 else: 

411 a = np.asarray(a) 

412 w = np.asarray(w) 

413 

414 if a.shape != w.shape: 

415 w = np.full(a.shape, w, dtype=w.dtype) 

416 

417 scores = np.unique(np.ravel(a)) # get ALL unique values 

418 testshape = list(a.shape) 

419 testshape[axis] = 1 

420 oldmostfreq = np.zeros(testshape) 

421 oldcounts = np.zeros(testshape) 

422 for score in scores: 

423 template = np.zeros(a.shape) 

424 ind = (a == score) 

425 template[ind] = w[ind] 

426 counts = np.expand_dims(np.sum(template, axis), axis) 

427 mostfrequent = np.where(counts > oldcounts, score, oldmostfreq) 

428 oldcounts = np.maximum(counts, oldcounts) 

429 oldmostfreq = mostfrequent 

430 return mostfrequent, oldcounts

482def svd_flip(u, v, u_based_decision=True): 

483 """Sign correction to ensure deterministic output from SVD. 

484 

485 Adjusts the columns of u and the rows of v such that the loadings in the 

486 columns in u that are largest in absolute value are always positive. 

487 

488 Parameters 

489 ---------- 

490 u : ndarray 

491 u and v are the output of `linalg.svd` or 

492 `sklearn.utils.extmath.randomized_svd`, with matching inner dimensions 

493 so one can compute `np.dot(u * s, v)`. 

494 

495 v : ndarray 

496 u and v are the output of `linalg.svd` or 

497 `sklearn.utils.extmath.randomized_svd`, with matching inner dimensions 

498 so one can compute `np.dot(u * s, v)`. 

499 

500 u_based_decision : boolean, (default=True) 

501 If True, use the columns of u as the basis for sign flipping. 

502 Otherwise, use the rows of v. The choice of which variable to base the 

503 decision on is generally algorithm dependent. 

504 

505 

506 Returns 

507 ------- 

508 u_adjusted, v_adjusted : arrays with the same dimensions as the input. 

509 

510 """ 

511 if u_based_decision: 

512 # columns of u, rows of v 

513 max_abs_cols = np.argmax(np.abs(u), axis=0) 

514 signs = np.sign(u[max_abs_cols, range(u.shape[1])]) 

515 u *= signs 

516 v *= signs[:, np.newaxis] 

517 else: 

518 # rows of v, columns of u 

519 max_abs_rows = np.argmax(np.abs(v), axis=1) 

520 signs = np.sign(v[range(v.shape[0]), max_abs_rows]) 

521 u *= signs 

522 v *= signs[:, np.newaxis] 

523 return u, v

571def softmax(X, copy=True): 

572 """ 

573 Calculate the softmax function. 

574 

575 The softmax function is calculated by 

576 np.exp(X) / np.sum(np.exp(X), axis=1) 

577 

578 This will cause overflow when large values are exponentiated. 

579 Hence the largest value in each row is subtracted from each data 

580 point to prevent this. 

581 

582 Parameters 

583 ---------- 

584 X : array-like of floats, shape (M, N) 

585 Argument to the logistic function 

586 

587 copy : bool, optional 

588 Copy X or not. 

589 

590 Returns 

591 ------- 

592 out : array, shape (M, N) 

593 Softmax function evaluated at every point in x 

594 """ 

595 if copy: 

596 X = np.copy(X) 

597 max_prob = np.max(X, axis=1).reshape((-1, 1)) 

598 X -= max_prob 

599 np.exp(X, X) 

600 sum_prob = np.sum(X, axis=1).reshape((-1, 1)) 

601 X /= sum_prob 

602 return X

692def _incremental_mean_and_var(X, last_mean, last_variance, last_sample_count): 

693 """Calculate mean update and a Youngs and Cramer variance update. 

694 

695 last_mean and last_variance are statistics computed at the last step by the 

696 function. Both must be initialized to 0.0. In case no scaling is required 

697 last_variance can be None. The mean is always required and returned because 

698 necessary for the calculation of the variance. last_n_samples_seen is the 

699 number of samples encountered until now. 

700 

701 From the paper "Algorithms for computing the sample variance: analysis and 

702 recommendations", by Chan, Golub, and LeVeque. 

703 

704 Parameters 

705 ---------- 

706 X : array-like, shape (n_samples, n_features) 

707 Data to use for variance update 

708 

709 last_mean : array-like, shape: (n_features,) 

710 

711 last_variance : array-like, shape: (n_features,) 

712 

713 last_sample_count : array-like, shape (n_features,) 

714 

715 Returns 

716 ------- 

717 updated_mean : array, shape (n_features,) 

718 

719 updated_variance : array, shape (n_features,) 

720 If None, only mean is computed 

721 

722 updated_sample_count : array, shape (n_features,) 

723 

724 Notes 

725 ----- 

726 NaNs are ignored during the algorithm. 

727 

728 References 

729 ---------- 

730 T. Chan, G. Golub, R. LeVeque. Algorithms for computing the sample 

731 variance: recommendations, The American Statistician, Vol. 37, No. 3, 

732 pp. 242-247 

733 

734 Also, see the sparse implementation of this in 

735 `utils.sparsefuncs.incr_mean_variance_axis` and 

736 `utils.sparsefuncs_fast.incr_mean_variance_axis0` 

737 """ 

738 # old = stats until now 

739 # new = the current increment 

740 # updated = the aggregated stats 

741 last_sum = last_mean * last_sample_count 

742 new_sum = _safe_accumulator_op(np.nansum, X, axis=0) 

743 

744 new_sample_count = np.sum(~np.isnan(X), axis=0) 

745 updated_sample_count = last_sample_count + new_sample_count 

746 

747 updated_mean = (last_sum + new_sum) / updated_sample_count 

748 

749 if last_variance is None: 

750 updated_variance = None 

751 else: 

752 new_unnormalized_variance = ( 

753 _safe_accumulator_op(np.nanvar, X, axis=0) * new_sample_count) 

754 last_unnormalized_variance = last_variance * last_sample_count 

755 

756 with np.errstate(divide='ignore', invalid='ignore'): 

757 last_over_new_count = last_sample_count / new_sample_count 

758 updated_unnormalized_variance = ( 

759 last_unnormalized_variance + new_unnormalized_variance + 

760 last_over_new_count / updated_sample_count * 

761 (last_sum / last_over_new_count - new_sum) ** 2) 

762 

763 zeros = last_sample_count == 0 

764 updated_unnormalized_variance[zeros] = new_unnormalized_variance[zeros] 

765 updated_variance = updated_unnormalized_variance / updated_sample_count 

766 

767 return updated_mean, updated_variance, updated_sample_count

770def _deterministic_vector_sign_flip(u): 

771 """Modify the sign of vectors for reproducibility 

772 

773 Flips the sign of elements of all the vectors (rows of u) such that 

774 the absolute maximum element of each vector is positive. 

775 

776 Parameters 

777 ---------- 

778 u : ndarray 

779 Array with vectors as its rows. 

780 

781 Returns 

782 ------- 

783 u_flipped : ndarray with same shape as u 

784 Array with the sign flipped vectors as its rows. 

785 """ 

786 max_abs_rows = np.argmax(np.abs(u), axis=1) 

787 signs = np.sign(u[range(u.shape[0]), max_abs_rows]) 

788 u *= signs[:, np.newaxis] 

789 return u

792def stable_cumsum(arr, axis=None, rtol=1e-05, atol=1e-08): 

793 """Use high precision for cumsum and check that final value matches sum 

794 

795 Parameters 

796 ---------- 

797 arr : array-like 

798 To be cumulatively summed as flat 

799 axis : int, optional 

800 Axis along which the cumulative sum is computed. 

801 The default (None) is to compute the cumsum over the flattened array. 

802 rtol : float 

803 Relative tolerance, see ``np.allclose`` 

804 atol : float 

805 Absolute tolerance, see ``np.allclose`` 

806 """ 

807 out = np.cumsum(arr, axis=axis, dtype=np.float64) 

808 expected = np.sum(arr, axis=axis, dtype=np.float64) 

809 if not np.all(np.isclose(out.take(-1, axis=axis), expected, rtol=rtol, 

810 atol=atol, equal_nan=True)): 

811 warnings.warn('cumsum was found to be unstable: ' 

812 'its last element does not correspond to sum', 

813 RuntimeWarning) 

814 return out
