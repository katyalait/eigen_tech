73def _encode(values, uniques=None, encode=False): 

74 """Helper function to factorize (find uniques) and encode values. 

75 

76 Uses pure python method for object dtype, and numpy method for 

77 all other dtypes. 

78 The numpy method has the limitation that the `uniques` need to 

79 be sorted. Importantly, this is not checked but assumed to already be 

80 the case. The calling method needs to ensure this for all non-object 

81 values. 

82 

83 Parameters 

84 ---------- 

85 values : array 

86 Values to factorize or encode. 

87 uniques : array, optional 

88 If passed, uniques are not determined from passed values (this 

89 can be because the user specified categories, or because they 

90 already have been determined in fit). 

91 encode : bool, default False 

92 If True, also encode the values into integer codes based on `uniques`. 

93 

94 Returns 

95 ------- 

96 uniques 

97 If ``encode=False``. The unique values are sorted if the `uniques` 

98 parameter was None (and thus inferred from the data). 

99 (uniques, encoded) 

100 If ``encode=True``. 

101 

102 """ 

103 if values.dtype == object: 

104 try: 

105 res = _encode_python(values, uniques, encode) 

106 except TypeError: 

107 raise TypeError("argument must be a string or number") 

108 return res 

109 else: 

110 return _encode_numpy(values, uniques, encode)

113def _encode_check_unknown(values, uniques, return_mask=False): 

114 """ 

115 Helper function to check for unknowns in values to be encoded. 

116 

117 Uses pure python method for object dtype, and numpy method for 

118 all other dtypes. 

119 

120 Parameters 

121 ---------- 

122 values : array 

123 Values to check for unknowns. 

124 uniques : array 

125 Allowed uniques values. 

126 return_mask : bool, default False 

127 If True, return a mask of the same shape as `values` indicating 

128 the valid values. 

129 

130 Returns 

131 ------- 

132 diff : list 

133 The unique values present in `values` and not in `uniques` (the 

134 unknown values). 

135 valid_mask : boolean array 

136 Additionally returned if ``return_mask=True``. 

137 

138 """ 

139 if values.dtype == object: 

140 uniques_set = set(uniques) 

141 diff = list(set(values) - uniques_set) 

142 if return_mask: 

143 if diff: 

144 valid_mask = np.array([val in uniques_set for val in values]) 

145 else: 

146 valid_mask = np.ones(len(values), dtype=bool) 

147 return diff, valid_mask 

148 else: 

149 return diff 

150 else: 

151 unique_values = np.unique(values) 

152 diff = list(np.setdiff1d(unique_values, uniques, assume_unique=True)) 

153 if return_mask: 

154 if diff: 

155 valid_mask = np.in1d(values, uniques) 

156 else: 

157 valid_mask = np.ones(len(values), dtype=bool) 

158 return diff, valid_mask 

159 else: 

160 return diff

289class LabelBinarizer(BaseEstimator, TransformerMixin): 

290 """Binarize labels in a one-vs-all fashion 

291 

292 Several regression and binary classification algorithms are 

293 available in scikit-learn. A simple way to extend these algorithms 

294 to the multi-class classification case is to use the so-called 

295 one-vs-all scheme. 

296 

297 At learning time, this simply consists in learning one regressor 

298 or binary classifier per class. In doing so, one needs to convert 

299 multi-class labels to binary labels (belong or does not belong 

300 to the class). LabelBinarizer makes this process easy with the 

301 transform method. 

302 

303 At prediction time, one assigns the class for which the corresponding 

304 model gave the greatest confidence. LabelBinarizer makes this easy 

305 with the inverse_transform method. 

306 

307 Read more in the :ref:`User Guide <preprocessing_targets>`. 

308 

309 Parameters 

310 ---------- 

311 

312 neg_label : int (default: 0) 

313 Value with which negative labels must be encoded. 

314 

315 pos_label : int (default: 1) 

316 Value with which positive labels must be encoded. 

317 

318 sparse_output : boolean (default: False) 

319 True if the returned array from transform is desired to be in sparse 

320 CSR format. 

321 

322 Attributes 

323 ---------- 

324 

325 classes_ : array of shape [n_class] 

326 Holds the label for each class. 

327 

328 y_type_ : str, 

329 Represents the type of the target data as evaluated by 

330 utils.multiclass.type_of_target. Possible type are 'continuous', 

331 'continuous-multioutput', 'binary', 'multiclass', 

332 'multiclass-multioutput', 'multilabel-indicator', and 'unknown'. 

333 

334 sparse_input_ : boolean, 

335 True if the input data to transform is given as a sparse matrix, False 

336 otherwise. 

337 

338 Examples 

339 -------- 

340 >>> from sklearn import preprocessing 

341 >>> lb = preprocessing.LabelBinarizer() 

342 >>> lb.fit([1, 2, 6, 4, 2]) 

343 LabelBinarizer(neg_label=0, pos_label=1, sparse_output=False) 

344 >>> lb.classes_ 

345 array([1, 2, 4, 6]) 

346 >>> lb.transform([1, 6]) 

347 array([[1, 0, 0, 0], 

348 [0, 0, 0, 1]]) 

349 

350 Binary targets transform to a column vector 

351 

352 >>> lb = preprocessing.LabelBinarizer() 

353 >>> lb.fit_transform(['yes', 'no', 'no', 'yes']) 

354 array([[1], 

355 [0], 

356 [0], 

357 [1]]) 

358 

359 Passing a 2D matrix for multilabel classification 

360 

361 >>> import numpy as np 

362 >>> lb.fit(np.array([[0, 1, 1], [1, 0, 0]])) 

363 LabelBinarizer(neg_label=0, pos_label=1, sparse_output=False) 

364 >>> lb.classes_ 

365 array([0, 1, 2]) 

366 >>> lb.transform([0, 1, 2, 1]) 

367 array([[1, 0, 0], 

368 [0, 1, 0], 

369 [0, 0, 1], 

370 [0, 1, 0]]) 

371 

372 See also 

373 -------- 

374 label_binarize : function to perform the transform operation of 

375 LabelBinarizer with fixed classes. 

376 sklearn.preprocessing.OneHotEncoder : encode categorical features 

377 using a one-hot aka one-of-K scheme. 

378 """

525def label_binarize(y, classes, neg_label=0, pos_label=1, sparse_output=False): 

526 """Binarize labels in a one-vs-all fashion 

527 

528 Several regression and binary classification algorithms are 

529 available in scikit-learn. A simple way to extend these algorithms 

530 to the multi-class classification case is to use the so-called 

531 one-vs-all scheme. 

532 

533 This function makes it possible to compute this transformation for a 

534 fixed set of class labels known ahead of time. 

535 

536 Parameters 

537 ---------- 

538 y : array-like 

539 Sequence of integer labels or multilabel data to encode. 

540 

541 classes : array-like of shape [n_classes] 

542 Uniquely holds the label for each class. 

543 

544 neg_label : int (default: 0) 

545 Value with which negative labels must be encoded. 

546 

547 pos_label : int (default: 1) 

548 Value with which positive labels must be encoded. 

549 

550 sparse_output : boolean (default: False), 

551 Set to true if output binary array is desired in CSR sparse format 

552 

553 Returns 

554 ------- 

555 Y : numpy array or CSR matrix of shape [n_samples, n_classes] 

556 Shape will be [n_samples, 1] for binary problems. 

557 

558 Examples 

559 -------- 

560 >>> from sklearn.preprocessing import label_binarize 

561 >>> label_binarize([1, 6], classes=[1, 2, 4, 6]) 

562 array([[1, 0, 0, 0], 

563 [0, 0, 0, 1]]) 

564 

565 The class ordering is preserved: 

566 

567 >>> label_binarize([1, 6], classes=[1, 6, 4, 2]) 

568 array([[1, 0, 0, 0], 

569 [0, 1, 0, 0]]) 

570 

571 Binary targets transform to a column vector 

572 

573 >>> label_binarize(['yes', 'no', 'no', 'yes'], classes=['no', 'yes']) 

574 array([[1], 

575 [0], 

576 [0], 

577 [1]]) 

578 

579 See also 

580 -------- 

581 LabelBinarizer : class used to wrap the functionality of label_binarize and 

582 allow for fitting to classes independently of the transform operation 

583 """ 

584 if not isinstance(y, list): 

585 # XXX Workaround that will be removed when list of list format is 

586 # dropped 

587 y = check_array(y, accept_sparse='csr', ensure_2d=False, dtype=None) 

588 else: 

589 if _num_samples(y) == 0: 

590 raise ValueError('y has 0 samples: %r' % y) 

591 if neg_label >= pos_label: 

592 raise ValueError("neg_label={0} must be strictly less than " 

593 "pos_label={1}.".format(neg_label, pos_label)) 

594 

595 if (sparse_output and (pos_label == 0 or neg_label != 0)): 

596 raise ValueError("Sparse binarization is only supported with non " 

597 "zero pos_label and zero neg_label, got " 

598 "pos_label={0} and neg_label={1}" 

599 "".format(pos_label, neg_label)) 

600 

601 # To account for pos_label == 0 in the dense case 

602 pos_switch = pos_label == 0 

603 if pos_switch: 

604 pos_label = -neg_label 

605 

606 y_type = type_of_target(y) 

607 if 'multioutput' in y_type: 

608 raise ValueError("Multioutput target data is not supported with label " 

609 "binarization") 

610 if y_type == 'unknown': 

611 raise ValueError("The type of target data is not known") 

612 

613 n_samples = y.shape[0] if sp.issparse(y) else len(y) 

614 n_classes = len(classes) 

615 classes = np.asarray(classes) 

616 

617 if y_type == "binary": 

618 if n_classes == 1: 

619 if sparse_output: 

620 return sp.csr_matrix((n_samples, 1), dtype=int) 

621 else: 

622 Y = np.zeros((len(y), 1), dtype=np.int) 

623 Y += neg_label 

624 return Y 

625 elif len(classes) >= 3: 

626 y_type = "multiclass" 

627 

628 sorted_class = np.sort(classes) 

629 if (y_type == "multilabel-indicator" and classes.size != y.shape[1]): 

630 raise ValueError("classes {0} missmatch with the labels {1}" 

631 "found in the data".format(classes, unique_labels(y))) 

632 

633 if y_type in ("binary", "multiclass"): 

634 y = column_or_1d(y) 

635 

636 # pick out the known labels from y 

637 y_in_classes = np.in1d(y, classes) 

638 y_seen = y[y_in_classes] 

639 indices = np.searchsorted(sorted_class, y_seen) 

640 indptr = np.hstack((0, np.cumsum(y_in_classes))) 

641 

642 data = np.empty_like(indices) 

643 data.fill(pos_label) 

644 Y = sp.csr_matrix((data, indices, indptr), 

645 shape=(n_samples, n_classes)) 

646 elif y_type == "multilabel-indicator": 

647 Y = sp.csr_matrix(y) 

648 if pos_label != 1: 

649 data = np.empty_like(Y.data) 

650 data.fill(pos_label) 

651 Y.data = data 

652 else: 

653 raise ValueError("%s target data is not supported with label " 

654 "binarization" % y_type) 

655 

656 if not sparse_output: 

657 Y = Y.toarray() 

658 Y = Y.astype(int, copy=False) 

659 

660 if neg_label != 0: 

661 Y[Y == 0] = neg_label 

662 

663 if pos_switch: 

664 Y[Y == pos_label] = 0 

665 else: 

666 Y.data = Y.data.astype(int, copy=False) 

667 

668 # preserve label ordering 

669 if np.any(classes != sorted_class): 

670 indices = np.searchsorted(sorted_class, classes) 

671 Y = Y[:, indices] 

672 

673 if y_type == "binary": 

674 if sparse_output: 

675 Y = Y.getcol(-1) 

676 else: 

677 Y = Y[:, -1].reshape((-1, 1)) 

678 

679 return Y

682def _inverse_binarize_multiclass(y, classes): 

683 """Inverse label binarization transformation for multiclass. 

684 

685 Multiclass uses the maximal score instead of a threshold. 

686 """ 

687 classes = np.asarray(classes) 

688 

689 if sp.issparse(y): 

690 # Find the argmax for each row in y where y is a CSR matrix 

691 

692 y = y.tocsr() 

693 n_samples, n_outputs = y.shape 

694 outputs = np.arange(n_outputs) 

695 row_max = min_max_axis(y, 1)[1] 

696 row_nnz = np.diff(y.indptr) 

697 

698 y_data_repeated_max = np.repeat(row_max, row_nnz) 

699 # picks out all indices obtaining the maximum per row 

700 y_i_all_argmax = np.flatnonzero(y_data_repeated_max == y.data) 

701 

702 # For corner case where last row has a max of 0 

703 if row_max[-1] == 0: 

704 y_i_all_argmax = np.append(y_i_all_argmax, [len(y.data)]) 

705 

706 # Gets the index of the first argmax in each row from y_i_all_argmax 

707 index_first_argmax = np.searchsorted(y_i_all_argmax, y.indptr[:-1]) 

708 # first argmax of each row 

709 y_ind_ext = np.append(y.indices, [0]) 

710 y_i_argmax = y_ind_ext[y_i_all_argmax[index_first_argmax]] 

711 # Handle rows of all 0 

712 y_i_argmax[np.where(row_nnz == 0)[0]] = 0 

713 

714 # Handles rows with max of 0 that contain negative numbers 

715 samples = np.arange(n_samples)[(row_nnz > 0) & 

716 (row_max.ravel() == 0)] 

717 for i in samples: 

718 ind = y.indices[y.indptr[i]:y.indptr[i + 1]] 

719 y_i_argmax[i] = classes[np.setdiff1d(outputs, ind)][0] 

720 

721 return classes[y_i_argmax] 

722 else: 

723 return classes.take(y.argmax(axis=1), mode="clip")

726def _inverse_binarize_thresholding(y, output_type, classes, threshold): 

727 """Inverse label binarization transformation using thresholding.""" 

728 

729 if output_type == "binary" and y.ndim == 2 and y.shape[1] > 2: 

730 raise ValueError("output_type='binary', but y.shape = {0}". 

731 format(y.shape)) 

732 

733 if output_type != "binary" and y.shape[1] != len(classes): 

734 raise ValueError("The number of class is not equal to the number of " 

735 "dimension of y.") 

736 

737 classes = np.asarray(classes) 

738 

739 # Perform thresholding 

740 if sp.issparse(y): 

741 if threshold > 0: 

742 if y.format not in ('csr', 'csc'): 

743 y = y.tocsr() 

744 y.data = np.array(y.data > threshold, dtype=np.int) 

745 y.eliminate_zeros() 

746 else: 

747 y = np.array(y.toarray() > threshold, dtype=np.int) 

748 else: 

749 y = np.array(y > threshold, dtype=np.int) 

750 

751 # Inverse transform data 

752 if output_type == "binary": 

753 if sp.issparse(y): 

754 y = y.toarray() 

755 if y.ndim == 2 and y.shape[1] == 2: 

756 return classes[y[:, 1]] 

757 else: 

758 if len(classes) == 1: 

759 return np.repeat(classes[0], len(y)) 

760 else: 

761 return classes[y.ravel()] 

762 

763 elif output_type == "multilabel-indicator": 

764 return y 

765 

766 else: 

767 raise ValueError("{0} format is not supported".format(output_type))

770class MultiLabelBinarizer(BaseEstimator, TransformerMixin): 

771 """Transform between iterable of iterables and a multilabel format 

772 

773 Although a list of sets or tuples is a very intuitive format for multilabel 

774 data, it is unwieldy to process. This transformer converts between this 

775 intuitive format and the supported multilabel format: a (samples x classes) 

776 binary matrix indicating the presence of a class label. 

777 

778 Parameters 

779 ---------- 

780 classes : array-like of shape [n_classes] (optional) 

781 Indicates an ordering for the class labels. 

782 All entries should be unique (cannot contain duplicate classes). 

783 

784 sparse_output : boolean (default: False), 

785 Set to true if output binary array is desired in CSR sparse format 

786 

787 Attributes 

788 ---------- 

789 classes_ : array of labels 

790 A copy of the `classes` parameter where provided, 

791 or otherwise, the sorted set of classes found when fitting. 

792 

793 Examples 

794 -------- 

795 >>> from sklearn.preprocessing import MultiLabelBinarizer 

796 >>> mlb = MultiLabelBinarizer() 

797 >>> mlb.fit_transform([(1, 2), (3,)]) 

798 array([[1, 1, 0], 

799 [0, 0, 1]]) 

800 >>> mlb.classes_ 

801 array([1, 2, 3]) 

802 

803 >>> mlb.fit_transform([{'sci-fi', 'thriller'}, {'comedy'}]) 

804 array([[0, 1, 1], 

805 [1, 0, 0]]) 

806 >>> list(mlb.classes_) 

807 ['comedy', 'sci-fi', 'thriller'] 

808 

809 See also 

810 -------- 

811 sklearn.preprocessing.OneHotEncoder : encode categorical features 

812 using a one-hot aka one-of-K scheme. 

813 """ 

814 

815 def __init__(self, classes=None, sparse_output=False): 

816 self.classes = classes 

817 self.sparse_output = sparse_output 

818 

819 def fit(self, y): 

820 """Fit the label sets binarizer, storing `classes_` 

821 

822 Parameters 

823 ---------- 

824 y : iterable of iterables 

825 A set of labels (any orderable and hashable object) for each 

826 sample. If the `classes` parameter is set, `y` will not be 

827 iterated. 

828 

829 Returns 

830 ------- 

831 self : returns this MultiLabelBinarizer instance 

832 """ 

833 self._cached_dict = None 

834 if self.classes is None: 

835 classes = sorted(set(itertools.chain.from_iterable(y))) 

836 elif len(set(self.classes)) < len(self.classes): 

837 raise ValueError("The classes argument contains duplicate " 

838 "classes. Remove these duplicates before passing " 

839 "them to MultiLabelBinarizer.") 

840 else: 

841 classes = self.classes 

842 dtype = np.int if all(isinstance(c, int) for c in classes) else object 

843 self.classes_ = np.empty(len(classes), dtype=dtype) 

844 self.classes_[:] = classes 

845 return self 

846 

847 def fit_transform(self, y): 

848 """Fit the label sets binarizer and transform the given label sets 

849 

850 Parameters 

851 ---------- 

852 y : iterable of iterables 

853 A set of labels (any orderable and hashable object) for each 

854 sample. If the `classes` parameter is set, `y` will not be 

855 iterated. 

856 

857 Returns 

858 ------- 

859 y_indicator : array or CSR matrix, shape (n_samples, n_classes) 

860 A matrix such that `y_indicator[i, j] = 1` iff `classes_[j]` is in 

861 `y[i]`, and 0 otherwise. 

862 """ 

863 self._cached_dict = None 

864 

865 if self.classes is not None: 

866 return self.fit(y).transform(y) 

867 

868 # Automatically increment on new class 

869 class_mapping = defaultdict(int) 

870 class_mapping.default_factory = class_mapping.__len__ 

871 yt = self._transform(y, class_mapping) 

872 

873 # sort classes and reorder columns 

874 tmp = sorted(class_mapping, key=class_mapping.get) 

875 

876 # (make safe for tuples) 

877 dtype = np.int if all(isinstance(c, int) for c in tmp) else object 

878 class_mapping = np.empty(len(tmp), dtype=dtype) 

879 class_mapping[:] = tmp 

880 self.classes_, inverse = np.unique(class_mapping, return_inverse=True) 

881 # ensure yt.indices keeps its current dtype 

882 yt.indices = np.array(inverse[yt.indices], dtype=yt.indices.dtype, 

883 copy=False) 

884 

885 if not self.sparse_output: 

886 yt = yt.toarray() 

887 

888 return yt 

889 

890 def transform(self, y): 

891 """Transform the given label sets 

892 

893 Parameters 

894 ---------- 

895 y : iterable of iterables 

896 A set of labels (any orderable and hashable object) for each 

897 sample. If the `classes` parameter is set, `y` will not be 

898 iterated. 

899 

900 Returns 

901 ------- 

902 y_indicator : array or CSR matrix, shape (n_samples, n_classes) 

903 A matrix such that `y_indicator[i, j] = 1` iff `classes_[j]` is in 

904 `y[i]`, and 0 otherwise. 

905 """ 

906 check_is_fitted(self, 'classes_') 

907 

908 class_to_index = self._build_cache() 

909 yt = self._transform(y, class_to_index) 

910 

911 if not self.sparse_output: 

912 yt = yt.toarray() 

913 

914 return yt 

915 

916 def _build_cache(self): 

917 if self._cached_dict is None: 

918 self._cached_dict = dict(zip(self.classes_, 

919 range(len(self.classes_)))) 

920 

921 return self._cached_dict 

922 

923 def _transform(self, y, class_mapping): 

924 """Transforms the label sets with a given mapping 

925 

926 Parameters 

927 ---------- 

928 y : iterable of iterables 

929 class_mapping : Mapping 

930 Maps from label to column index in label indicator matrix 

931 

932 Returns 

933 ------- 

934 y_indicator : sparse CSR matrix, shape (n_samples, n_classes) 

935 Label indicator matrix 

936 """ 

937 indices = array.array('i') 

938 indptr = array.array('i', [0]) 

939 unknown = set() 

940 for labels in y: 

941 index = set() 

942 for label in labels: 

943 try: 

944 index.add(class_mapping[label]) 

945 except KeyError: 

946 unknown.add(label) 

947 indices.extend(index) 

948 indptr.append(len(indices)) 

949 if unknown: 

950 warnings.warn('unknown class(es) {0} will be ignored' 

951 .format(sorted(unknown, key=str))) 

952 data = np.ones(len(indices), dtype=int) 

953 

954 return sp.csr_matrix((data, indices, indptr), 

955 shape=(len(indptr) - 1, len(class_mapping)))

957 def inverse_transform(self, yt): 

958 """Transform the given indicator matrix into label sets 

959 

960 Parameters 

961 ---------- 

962 yt : array or sparse matrix of shape (n_samples, n_classes) 

963 A matrix containing only 1s ands 0s. 

964 

965 Returns 

966 ------- 

967 y : list of tuples 

968 The set of labels for each sample such that `y[i]` consists of 

969 `classes_[j]` for each `yt[i, j] == 1`. 

970 """ 

971 check_is_fitted(self, 'classes_') 

972 

973 if yt.shape[1] != len(self.classes_): 

974 raise ValueError('Expected indicator for {0} classes, but got {1}' 

975 .format(len(self.classes_), yt.shape[1])) 

976 

977 if sp.issparse(yt): 

978 yt = yt.tocsr() 

979 if len(yt.data) != 0 and len(np.setdiff1d(yt.data, [0, 1])) > 0: 

980 raise ValueError('Expected only 0s and 1s in label indicator.') 

981 return [tuple(self.classes_.take(yt.indices[start:end])) 

982 for start, end in zip(yt.indptr[:-1], yt.indptr[1:])] 

983 else: 

984 unexpected = np.setdiff1d(yt, [0, 1]) 

985 if len(unexpected) > 0: 

986 raise ValueError('Expected only 0s and 1s in label indicator. ' 

987 'Also got {0}'.format(unexpected)) 

988 return [tuple(self.classes_.compress(indicators)) for indicators 

989 in yt]
