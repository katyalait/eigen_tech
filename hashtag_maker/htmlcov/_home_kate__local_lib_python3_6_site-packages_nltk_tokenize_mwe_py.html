8""" 

9Multi-Word Expression Tokenizer 

10 

11A ``MWETokenizer`` takes a string which has already been divided into tokens and 

12retokenizes it, merging multi-word expressions into single tokens, using a lexicon 

13of MWEs: 

14 

15 

16 >>> from nltk.tokenize import MWETokenizer 

17 

18 >>> tokenizer = MWETokenizer([('a', 'little'), ('a', 'little', 'bit'), ('a', 'lot')]) 

19 >>> tokenizer.add_mwe(('in', 'spite', 'of')) 

20 

21 >>> tokenizer.tokenize('Testing testing testing one two three'.split()) 

22 ['Testing', 'testing', 'testing', 'one', 'two', 'three'] 

23 

24 >>> tokenizer.tokenize('This is a test in spite'.split()) 

25 ['This', 'is', 'a', 'test', 'in', 'spite'] 

26 

27 >>> tokenizer.tokenize('In a little or a little bit or a lot in spite of'.split()) 

28 ['In', 'a_little', 'or', 'a_little_bit', 'or', 'a_lot', 'in_spite_of'] 

29 

30""" 

31from nltk.util import Trie 

32 

33from nltk.tokenize.api import TokenizerI 

34 

35 

36class MWETokenizer(TokenizerI): 

37 """A tokenizer that processes tokenized text and merges multi-word expressions 

38 into single tokens. 

39 """ 

40 

41 def __init__(self, mwes=None, separator='_'): 

42 """Initialize the multi-word tokenizer with a list of expressions and a 

43 separator 

44 

45 :type mwes: list(list(str)) 

46 :param mwes: A sequence of multi-word expressions to be merged, where 

47 each MWE is a sequence of strings. 

48 :type separator: str 

49 :param separator: String that should be inserted between words in a multi-word 

50 expression token. (Default is '_') 

51 

52 """ 

53 if not mwes: 

54 mwes = [] 

55 self._mwes = Trie(mwes) 

56 self._separator = separator 

57 

58 def add_mwe(self, mwe): 

59 """Add a multi-word expression to the lexicon (stored as a word trie) 

60 

61 We use ``util.Trie`` to represent the trie. Its form is a dict of dicts.  

62 The key True marks the end of a valid MWE. 

63 

64 :param mwe: The multi-word expression we're adding into the word trie 

65 :type mwe: tuple(str) or list(str) 

66 

67 :Example: 

68 

69 >>> tokenizer = MWETokenizer() 

70 >>> tokenizer.add_mwe(('a', 'b')) 

71 >>> tokenizer.add_mwe(('a', 'b', 'c')) 

72 >>> tokenizer.add_mwe(('a', 'x')) 

73 >>> expected = {'a': {'x': {True: None}, 'b': {True: None, 'c': {True: None}}}} 

74 >>> tokenizer._mwes == expected 

75 True 

76 

77 """ 

78 self._mwes.insert(mwe) 

79 

80 def tokenize(self, text): 

81 """ 

82 

83 :param text: A list containing tokenized text 

84 :type text: list(str) 

85 :return: A list of the tokenized text with multi-words merged together 

86 :rtype: list(str) 

87 

88 :Example: 

89 

90 >>> tokenizer = MWETokenizer([('hors', "d'oeuvre")], separator='+') 

91 >>> tokenizer.tokenize("An hors d'oeuvre tonight, sir?".split()) 

92 ['An', "hors+d'oeuvre", 'tonight,', 'sir?'] 

93  

94 """ 

95 i = 0 

96 n = len(text) 

97 result = [] 

98 

99 while i < n: 

100 if text[i] in self._mwes: 

101 # possible MWE match 

102 j = i 

103 trie = self._mwes 

104 while j < n and text[j] in trie: 

105 trie = trie[text[j]] 

106 j = j + 1 

107 else: 

108 if Trie.LEAF in trie: 

109 # success! 

110 result.append(self._separator.join(text[i:j])) 

111 i = j 

112 else: 

113 # no match, so backtrack 

114 result.append(text[i]) 

115 i += 1 

116 else: 

117 result.append(text[i]) 

118 i += 1 

119 

120 return result
