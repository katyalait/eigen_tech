79def fetch_rcv1(data_home=None, subset='all', download_if_missing=True, 

80 random_state=None, shuffle=False, return_X_y=False): 

81 """Load the RCV1 multilabel dataset (classification). 

82 

83 Download it if necessary. 

84 

85 Version: RCV1-v2, vectors, full sets, topics multilabels. 

86 

87 ================= ===================== 

88 Classes 103 

89 Samples total 804414 

90 Dimensionality 47236 

91 Features real, between 0 and 1 

92 ================= ===================== 

93 

94 Read more in the :ref:`User Guide <rcv1_dataset>`. 

95 

96 .. versionadded:: 0.17 

97 

98 Parameters 

99 ---------- 

100 data_home : string, optional 

101 Specify another download and cache folder for the datasets. By default 

102 all scikit-learn data is stored in '~/scikit_learn_data' subfolders. 

103 

104 subset : string, 'train', 'test', or 'all', default='all' 

105 Select the dataset to load: 'train' for the training set 

106 (23149 samples), 'test' for the test set (781265 samples), 

107 'all' for both, with the training samples first if shuffle is False. 

108 This follows the official LYRL2004 chronological split. 

109 

110 download_if_missing : boolean, default=True 

111 If False, raise a IOError if the data is not locally available 

112 instead of trying to download the data from the source site. 

113 

114 random_state : int, RandomState instance or None (default) 

115 Determines random number generation for dataset shuffling. Pass an int 

116 for reproducible output across multiple function calls. 

117 See :term:`Glossary <random_state>`. 

118 

119 shuffle : bool, default=False 

120 Whether to shuffle dataset. 

121 

122 return_X_y : boolean, default=False. 

123 If True, returns ``(dataset.data, dataset.target)`` instead of a Bunch 

124 object. See below for more information about the `dataset.data` and 

125 `dataset.target` object. 

126 

127 .. versionadded:: 0.20 

128 

129 Returns 

130 ------- 

131 dataset : dict-like object with the following attributes: 

132 

133 dataset.data : scipy csr array, dtype np.float64, shape (804414, 47236) 

134 The array has 0.16% of non zero values. 

135 

136 dataset.target : scipy csr array, dtype np.uint8, shape (804414, 103) 

137 Each sample has a value of 1 in its categories, and 0 in others. 

138 The array has 3.15% of non zero values. 

139 

140 dataset.sample_id : numpy array, dtype np.uint32, shape (804414,) 

141 Identification number of each sample, as ordered in dataset.data. 

142 

143 dataset.target_names : numpy array, dtype object, length (103) 

144 Names of each target (RCV1 topics), as ordered in dataset.target. 

145 

146 dataset.DESCR : string 

147 Description of the RCV1 dataset. 

148 

149 (data, target) : tuple if ``return_X_y`` is True 

150 

151 .. versionadded:: 0.20 

152 """ 

153 N_SAMPLES = 804414 

154 N_FEATURES = 47236 

155 N_CATEGORIES = 103 

156 N_TRAIN = 23149 

157 

158 data_home = get_data_home(data_home=data_home) 

159 rcv1_dir = join(data_home, "RCV1") 

160 if download_if_missing: 

161 if not exists(rcv1_dir): 

162 makedirs(rcv1_dir) 

163 

164 samples_path = _pkl_filepath(rcv1_dir, "samples.pkl") 

165 sample_id_path = _pkl_filepath(rcv1_dir, "sample_id.pkl") 

166 sample_topics_path = _pkl_filepath(rcv1_dir, "sample_topics.pkl") 

167 topics_path = _pkl_filepath(rcv1_dir, "topics_names.pkl") 

168 

169 # load data (X) and sample_id 

170 if download_if_missing and (not exists(samples_path) or 

171 not exists(sample_id_path)): 

172 files = [] 

173 for each in XY_METADATA: 

174 logger.info("Downloading %s" % each.url) 

175 file_path = _fetch_remote(each, dirname=rcv1_dir) 

176 files.append(GzipFile(filename=file_path)) 

177 

178 Xy = load_svmlight_files(files, n_features=N_FEATURES) 

179 

180 # Training data is before testing data 

181 X = sp.vstack([Xy[8], Xy[0], Xy[2], Xy[4], Xy[6]]).tocsr() 

182 sample_id = np.hstack((Xy[9], Xy[1], Xy[3], Xy[5], Xy[7])) 

183 sample_id = sample_id.astype(np.uint32, copy=False) 

184 

185 _joblib.dump(X, samples_path, compress=9) 

186 _joblib.dump(sample_id, sample_id_path, compress=9) 

187 

188 # delete archives 

189 for f in files: 

190 f.close() 

191 remove(f.name) 

192 else: 

193 X, sample_id = _refresh_cache([samples_path, sample_id_path], 9) 

194 # TODO: Revert to the following two lines in v0.23 

195 # X = joblib.load(samples_path) 

196 # sample_id = joblib.load(sample_id_path) 

197 

198 # load target (y), categories, and sample_id_bis 

199 if download_if_missing and (not exists(sample_topics_path) or 

200 not exists(topics_path)): 

201 logger.info("Downloading %s" % TOPICS_METADATA.url) 

202 topics_archive_path = _fetch_remote(TOPICS_METADATA, 

203 dirname=rcv1_dir) 

204 

205 # parse the target file 

206 n_cat = -1 

207 n_doc = -1 

208 doc_previous = -1 

209 y = np.zeros((N_SAMPLES, N_CATEGORIES), dtype=np.uint8) 

210 sample_id_bis = np.zeros(N_SAMPLES, dtype=np.int32) 

211 category_names = {} 

212 with GzipFile(filename=topics_archive_path, mode='rb') as f: 

213 for line in f: 

214 line_components = line.decode("ascii").split(" ") 

215 if len(line_components) == 3: 

216 cat, doc, _ = line_components 

217 if cat not in category_names: 

218 n_cat += 1 

219 category_names[cat] = n_cat 

220 

221 doc = int(doc) 

222 if doc != doc_previous: 

223 doc_previous = doc 

224 n_doc += 1 

225 sample_id_bis[n_doc] = doc 

226 y[n_doc, category_names[cat]] = 1 

227 

228 # delete archive 

229 remove(topics_archive_path) 

230 

231 # Samples in X are ordered with sample_id, 

232 # whereas in y, they are ordered with sample_id_bis. 

233 permutation = _find_permutation(sample_id_bis, sample_id) 

234 y = y[permutation, :] 

235 

236 # save category names in a list, with same order than y 

237 categories = np.empty(N_CATEGORIES, dtype=object) 

238 for k in category_names.keys(): 

239 categories[category_names[k]] = k 

240 

241 # reorder categories in lexicographic order 

242 order = np.argsort(categories) 

243 categories = categories[order] 

244 y = sp.csr_matrix(y[:, order]) 

245 

246 _joblib.dump(y, sample_topics_path, compress=9) 

247 _joblib.dump(categories, topics_path, compress=9) 

248 else: 

249 y, categories = _refresh_cache([sample_topics_path, topics_path], 9) 

250 # TODO: Revert to the following two lines in v0.23 

251 # y = joblib.load(sample_topics_path) 

252 # categories = joblib.load(topics_path) 

253 

254 if subset == 'all': 

255 pass 

256 elif subset == 'train': 

257 X = X[:N_TRAIN, :] 

258 y = y[:N_TRAIN, :] 

259 sample_id = sample_id[:N_TRAIN] 

260 elif subset == 'test': 

261 X = X[N_TRAIN:, :] 

262 y = y[N_TRAIN:, :] 

263 sample_id = sample_id[N_TRAIN:] 

264 else: 

265 raise ValueError("Unknown subset parameter. Got '%s' instead of one" 

266 " of ('all', 'train', test')" % subset) 

267 

268 if shuffle: 

269 X, y, sample_id = shuffle_(X, y, sample_id, random_state=random_state) 

270 

271 module_path = dirname(__file__) 

272 with open(join(module_path, 'descr', 'rcv1.rst')) as rst_file: 

273 fdescr = rst_file.read() 

274 

275 if return_X_y: 

276 return X, y 

277 

278 return Bunch(data=X, target=y, sample_id=sample_id, 

279 target_names=categories, DESCR=fdescr)

282def _inverse_permutation(p): 

283 """inverse permutation p""" 

284 n = p.size 

285 s = np.zeros(n, dtype=np.int32) 

286 i = np.arange(n, dtype=np.int32) 

287 np.put(s, p, i) # s[p] = i 

288 return s

291def _find_permutation(a, b): 

292 """find the permutation from a to b""" 

293 t = np.argsort(a) 

294 u = np.argsort(b) 

295 u_ = _inverse_permutation(u) 

296 return t[u_]
