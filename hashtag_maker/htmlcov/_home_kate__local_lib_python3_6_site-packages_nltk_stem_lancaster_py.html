175 def __init__(self, rule_tuple=None, strip_prefix_flag=False): 

176 """Create an instance of the Lancaster stemmer. 

177 """ 

178 # Setup an empty rule dictionary - this will be filled in later 

179 self.rule_dictionary = {} 

180 # Check if a user wants to strip prefix 

181 self._strip_prefix = strip_prefix_flag 

182 # Check if a user wants to use his/her own rule tuples. 

183 self._rule_tuple = rule_tuple if rule_tuple else self.default_rule_tuple 

184 

185 def parseRules(self, rule_tuple=None): 

186 """Validate the set of rules used in this stemmer. 

187 

188 If this function is called as an individual method, without using stem 

189 method, rule_tuple argument will be compiled into self.rule_dictionary. 

190 If this function is called within stem, self._rule_tuple will be used. 

191 

192 """ 

193 # If there is no argument for the function, use class' own rule tuple. 

194 rule_tuple = rule_tuple if rule_tuple else self._rule_tuple 

195 valid_rule = re.compile("^[a-z]+\*?\d[a-z]*[>\.]?$") 

196 # Empty any old rules from the rule set before adding new ones 

197 self.rule_dictionary = {} 

198 

199 for rule in rule_tuple: 

200 if not valid_rule.match(rule): 

201 raise ValueError("The rule {0} is invalid".format(rule)) 

202 first_letter = rule[0:1] 

203 if first_letter in self.rule_dictionary: 

204 self.rule_dictionary[first_letter].append(rule) 

205 else: 

206 self.rule_dictionary[first_letter] = [rule]

208 def stem(self, word): 

209 """Stem a word using the Lancaster stemmer. 

210 """ 

211 # Lower-case the word, since all the rules are lower-cased 

212 word = word.lower() 

213 word = self.__stripPrefix(word) if self._strip_prefix else word 

214 

215 # Save a copy of the original word 

216 intact_word = word 

217 

218 # If rule dictionary is empty, parse rule tuple. 

219 if not self.rule_dictionary: 

220 self.parseRules() 

221 

222 return self.__doStemming(word, intact_word)

224 def __doStemming(self, word, intact_word): 

225 """Perform the actual word stemming 

226 """ 

227 

228 valid_rule = re.compile("^([a-z]+)(\*?)(\d)([a-z]*)([>\.]?)$") 

229 

230 proceed = True 

231 

232 while proceed: 

233 

234 # Find the position of the last letter of the word to be stemmed 

235 last_letter_position = self.__getLastLetter(word) 

236 

237 # Only stem the word if it has a last letter and a rule matching that last letter 

238 if ( 

239 last_letter_position < 0 

240 or word[last_letter_position] not in self.rule_dictionary 

241 ): 

242 proceed = False 

243 

244 else: 

245 rule_was_applied = False 

246 

247 # Go through each rule that matches the word's final letter 

248 for rule in self.rule_dictionary[word[last_letter_position]]: 

249 rule_match = valid_rule.match(rule) 

250 if rule_match: 

251 ( 

252 ending_string, 

253 intact_flag, 

254 remove_total, 

255 append_string, 

256 cont_flag, 

257 ) = rule_match.groups() 

258 

259 # Convert the number of chars to remove when stemming 

260 # from a string to an integer 

261 remove_total = int(remove_total) 

262 

263 # Proceed if word's ending matches rule's word ending 

264 if word.endswith(ending_string[::-1]): 

265 if intact_flag: 

266 if word == intact_word and self.__isAcceptable( 

267 word, remove_total 

268 ): 

269 word = self.__applyRule( 

270 word, remove_total, append_string 

271 ) 

272 rule_was_applied = True 

273 if cont_flag == '.': 

274 proceed = False 

275 break 

276 elif self.__isAcceptable(word, remove_total): 

277 word = self.__applyRule( 

278 word, remove_total, append_string 

279 ) 

280 rule_was_applied = True 

281 if cont_flag == '.': 

282 proceed = False 

283 break 

284 # If no rules apply, the word doesn't need any more stemming 

285 if rule_was_applied == False: 

286 proceed = False 

287 return word

289 def __getLastLetter(self, word): 

290 """Get the zero-based index of the last alphabetic character in this string 

291 """ 

292 last_letter = -1 

293 for position in range(len(word)): 

294 if word[position].isalpha(): 

295 last_letter = position 

296 else: 

297 break 

298 return last_letter

300 def __isAcceptable(self, word, remove_total): 

301 """Determine if the word is acceptable for stemming. 

302 """ 

303 word_is_acceptable = False 

304 # If the word starts with a vowel, it must be at least 2 

305 # characters long to be stemmed 

306 if word[0] in "aeiouy": 

307 if len(word) - remove_total >= 2: 

308 word_is_acceptable = True 

309 # If the word starts with a consonant, it must be at least 3 

310 # characters long (including one vowel) to be stemmed 

311 elif len(word) - remove_total >= 3: 

312 if word[1] in "aeiouy": 

313 word_is_acceptable = True 

314 elif word[2] in "aeiouy": 

315 word_is_acceptable = True 

316 return word_is_acceptable

318 def __applyRule(self, word, remove_total, append_string): 

319 """Apply the stemming rule to the word 

320 """ 

321 # Remove letters from the end of the word 

322 new_word_length = len(word) - remove_total 

323 word = word[0:new_word_length] 

324 

325 # And add new letters to the end of the truncated word 

326 if append_string: 

327 word += append_string 

328 return word

331 """Remove prefix from a word. 

332 

333 This function originally taken from Whoosh. 

334 

335 """ 

336 for prefix in ( 

337 "kilo", 

338 "micro", 

339 "milli", 

340 "intra", 

341 "ultra",

343 "nano", 

344 "pico", 

345 "pseudo", 

346 ): 

347 if word.startswith(prefix): 

348 return word[len(prefix) :]
