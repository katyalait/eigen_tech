10""" 

11Register JSON tags, so the nltk data loader knows what module and class to look for. 

12 

13NLTK uses simple '!' tags to mark the types of objects, but the fully-qualified 

14"tag:nltk.org,2011:" prefix is also accepted in case anyone ends up 

15using it. 

16""" 

17 

18import json 

19 

20json_tags = {} 

21 

22TAG_PREFIX = '!' 

23 

24 

25def register_tag(cls): 

26 """ 

27 Decorates a class to register it's json tag. 

28 """ 

29 json_tags[TAG_PREFIX + getattr(cls, 'json_tag')] = cls 

30 return cls 

31 

32 

33class JSONTaggedEncoder(json.JSONEncoder): 

34 def default(self, obj): 

35 obj_tag = getattr(obj, 'json_tag', None) 

36 if obj_tag is None: 

37 return super(JSONTaggedEncoder, self).default(obj) 

38 obj_tag = TAG_PREFIX + obj_tag 

39 obj = obj.encode_json_obj() 

40 return {obj_tag: obj} 

41 

42 

43class JSONTaggedDecoder(json.JSONDecoder): 

44 def decode(self, s): 

45 return self.decode_obj(super(JSONTaggedDecoder, self).decode(s)) 

46 

47 @classmethod 

48 def decode_obj(cls, obj): 

49 # Decode nested objects first. 

50 if isinstance(obj, dict): 

51 obj = dict((key, cls.decode_obj(val)) for (key, val) in obj.items()) 

52 elif isinstance(obj, list): 

53 obj = list(cls.decode_obj(val) for val in obj) 

54 # Check if we have a tagged object. 

55 if not isinstance(obj, dict) or len(obj) != 1: 

56 return obj 

57 obj_tag = next(iter(obj.keys())) 

58 if not obj_tag.startswith('!'): 

59 return obj 

60 if obj_tag not in json_tags: 

61 raise ValueError('Unknown tag', obj_tag) 

62 obj_cls = json_tags[obj_tag] 

63 return obj_cls.decode_json_obj(obj[obj_tag]) 

64
