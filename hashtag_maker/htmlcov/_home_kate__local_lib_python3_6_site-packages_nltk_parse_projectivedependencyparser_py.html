29@total_ordering 

30@python_2_unicode_compatible 

31class DependencySpan(object): 

32 """ 

33 A contiguous span over some part of the input string representing 

34 dependency (head -> modifier) relationships amongst words. An atomic 

35 span corresponds to only one word so it isn't a 'span' in the conventional 

36 sense, as its _start_index = _end_index = _head_index for concatenation 

37 purposes. All other spans are assumed to have arcs between all nodes 

38 within the start and end indexes of the span, and one head index corresponding 

39 to the head word for the entire span. This is the same as the root node if 

40 the dependency structure were depicted as a graph. 

41 """ 

42 

43 def __init__(self, start_index, end_index, head_index, arcs, tags): 

44 self._start_index = start_index 

45 self._end_index = end_index 

46 self._head_index = head_index 

47 self._arcs = arcs 

48 self._tags = tags 

49 self._comparison_key = (start_index, end_index, head_index, tuple(arcs)) 

50 self._hash = hash(self._comparison_key) 

51 

52 def head_index(self): 

53 """ 

54 :return: An value indexing the head of the entire ``DependencySpan``. 

55 :rtype: int 

56 """ 

57 return self._head_index 

58 

59 def __repr__(self): 

60 """ 

61 :return: A concise string representatino of the ``DependencySpan``. 

62 :rtype: str. 

63 """ 

64 return 'Span %d-%d; Head Index: %d' % ( 

65 self._start_index, 

66 self._end_index, 

67 self._head_index, 

68 ) 

69 

70 def __str__(self): 

71 """ 

72 :return: A verbose string representation of the ``DependencySpan``. 

73 :rtype: str 

74 """ 

75 str = 'Span %d-%d; Head Index: %d' % ( 

76 self._start_index, 

77 self._end_index, 

78 self._head_index, 

79 ) 

80 for i in range(len(self._arcs)): 

81 str += '\n%d <- %d, %s' % (i, self._arcs[i], self._tags[i]) 

82 return str 

83 

84 def __eq__(self, other): 

85 return ( 

86 type(self) == type(other) and self._comparison_key == other._comparison_key 

87 ) 

88 

89 def __ne__(self, other): 

90 return not self == other 

91 

92 def __lt__(self, other): 

93 if not isinstance(other, DependencySpan): 

94 raise_unorderable_types("<", self, other) 

95 return self._comparison_key < other._comparison_key 

96 

97 def __hash__(self): 

98 """ 

99 :return: The hash value of this ``DependencySpan``. 

100 """ 

101 return self._hash

443 def train(self, graphs): 

444 """ 

445 Trains a ProbabilisticDependencyGrammar based on the list of input 

446 DependencyGraphs. This model is an implementation of Eisner's (1996) 

447 Model C, which derives its statistics from head-word, head-tag, 

448 child-word, and child-tag relationships. 

449 

450 :param graphs: A list of dependency graphs to train from. 

451 :type: list(DependencyGraph) 

452 """ 

453 productions = [] 

454 events = defaultdict(int) 

455 tags = {} 

456 for dg in graphs: 

457 for node_index in range(1, len(dg.nodes)): 

458 # children = dg.nodes[node_index]['deps'] 

459 children = list(chain(*dg.nodes[node_index]['deps'].values())) 

460 

461 nr_left_children = dg.left_children(node_index) 

462 nr_right_children = dg.right_children(node_index) 

463 nr_children = nr_left_children + nr_right_children 

464 for child_index in range( 

465 0 - (nr_left_children + 1), nr_right_children + 2 

466 ): 

467 head_word = dg.nodes[node_index]['word'] 

468 head_tag = dg.nodes[node_index]['tag'] 

469 if head_word in tags: 

470 tags[head_word].add(head_tag) 

471 else: 

472 tags[head_word] = set([head_tag]) 

473 child = 'STOP' 

474 child_tag = 'STOP' 

475 prev_word = 'START' 

476 prev_tag = 'START' 

477 if child_index < 0: 

478 array_index = child_index + nr_left_children 

479 if array_index >= 0: 

480 child = dg.nodes[children[array_index]]['word'] 

481 child_tag = dg.nodes[children[array_index]]['tag'] 

482 if child_index != -1: 

483 prev_word = dg.nodes[children[array_index + 1]]['word'] 

484 prev_tag = dg.nodes[children[array_index + 1]]['tag'] 

485 if child != 'STOP': 

486 productions.append(DependencyProduction(head_word, [child])) 

487 head_event = '(head (%s %s) (mods (%s, %s, %s) left))' % ( 

488 child, 

489 child_tag, 

490 prev_tag, 

491 head_word, 

492 head_tag, 

493 ) 

494 mod_event = '(mods (%s, %s, %s) left))' % ( 

495 prev_tag, 

496 head_word, 

497 head_tag, 

498 ) 

499 events[head_event] += 1 

500 events[mod_event] += 1 

501 elif child_index > 0: 

502 array_index = child_index + nr_left_children - 1 

503 if array_index < nr_children: 

504 child = dg.nodes[children[array_index]]['word'] 

505 child_tag = dg.nodes[children[array_index]]['tag'] 

506 if child_index != 1: 

507 prev_word = dg.nodes[children[array_index - 1]]['word'] 

508 prev_tag = dg.nodes[children[array_index - 1]]['tag'] 

509 if child != 'STOP': 

510 productions.append(DependencyProduction(head_word, [child])) 

511 head_event = '(head (%s %s) (mods (%s, %s, %s) right))' % ( 

512 child, 

513 child_tag, 

514 prev_tag, 

515 head_word, 

516 head_tag, 

517 ) 

518 mod_event = '(mods (%s, %s, %s) right))' % ( 

519 prev_tag, 

520 head_word, 

521 head_tag, 

522 ) 

523 events[head_event] += 1 

524 events[mod_event] += 1 

525 self._grammar = ProbabilisticDependencyGrammar(productions, events, tags)
