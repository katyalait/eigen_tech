1""" 

2The :mod:`sklearn.metrics.scorer` submodule implements a flexible 

3interface for model selection and evaluation using 

4arbitrary score functions. 

5 

6A scorer object is a callable that can be passed to 

7:class:`sklearn.model_selection.GridSearchCV` or 

8:func:`sklearn.model_selection.cross_val_score` as the ``scoring`` 

9parameter, to specify how a model should be evaluated. 

10 

11The signature of the call is ``(estimator, X, y)`` where ``estimator`` 

12is the model to be evaluated, ``X`` is the test data and ``y`` is the 

13ground truth labeling (or ``None`` in the case of unsupervised models).

100class _ProbaScorer(_BaseScorer): 

101 def __call__(self, clf, X, y, sample_weight=None): 

102 """Evaluate predicted probabilities for X relative to y_true. 

103 

104 Parameters 

105 ---------- 

106 clf : object 

107 Trained classifier to use for scoring. Must have a predict_proba 

108 method; the output of that is used to compute the score. 

109 

110 X : array-like or sparse matrix 

111 Test data that will be fed to clf.predict_proba. 

112 

113 y : array-like 

114 Gold standard target values for X. These must be class labels, 

115 not probabilities. 

116 

117 sample_weight : array-like, optional (default=None) 

118 Sample weights. 

119 

120 Returns 

121 ------- 

122 score : float 

123 Score function applied to prediction of estimator on X. 

124 """ 

125 y_type = type_of_target(y) 

126 y_pred = clf.predict_proba(X) 

127 if y_type == "binary": 

128 if y_pred.shape[1] == 2: 

129 y_pred = y_pred[:, 1] 

130 else: 

131 raise ValueError('got predict_proba of shape {},' 

132 ' but need classifier with two' 

133 ' classes for {} scoring'.format( 

134 y_pred.shape, self._score_func.__name__)) 

135 if sample_weight is not None: 

136 return self._sign * self._score_func(y, y_pred, 

137 sample_weight=sample_weight, 

138 **self._kwargs) 

139 else: 

140 return self._sign * self._score_func(y, y_pred, **self._kwargs)

146class _ThresholdScorer(_BaseScorer): 

147 def __call__(self, clf, X, y, sample_weight=None): 

148 """Evaluate decision function output for X relative to y_true. 

149 

150 Parameters 

151 ---------- 

152 clf : object 

153 Trained classifier to use for scoring. Must have either a 

154 decision_function method or a predict_proba method; the output of 

155 that is used to compute the score. 

156 

157 X : array-like or sparse matrix 

158 Test data that will be fed to clf.decision_function or 

159 clf.predict_proba. 

160 

161 y : array-like 

162 Gold standard target values for X. These must be class labels, 

163 not decision function values. 

164 

165 sample_weight : array-like, optional (default=None) 

166 Sample weights. 

167 

168 Returns 

169 ------- 

170 score : float 

171 Score function applied to prediction of estimator on X. 

172 """ 

173 y_type = type_of_target(y) 

174 if y_type not in ("binary", "multilabel-indicator"): 

175 raise ValueError("{0} format is not supported".format(y_type)) 

176 

177 if is_regressor(clf): 

178 y_pred = clf.predict(X) 

179 else: 

180 try: 

181 y_pred = clf.decision_function(X) 

182 

183 # For multi-output multi-class estimator 

184 if isinstance(y_pred, list): 

185 y_pred = np.vstack([p for p in y_pred]).T 

186 

187 except (NotImplementedError, AttributeError): 

188 y_pred = clf.predict_proba(X) 

189 

190 if y_type == "binary": 

191 if y_pred.shape[1] == 2: 

192 y_pred = y_pred[:, 1] 

193 else: 

194 raise ValueError('got predict_proba of shape {},' 

195 ' but need classifier with two' 

196 ' classes for {} scoring'.format( 

197 y_pred.shape, 

198 self._score_func.__name__)) 

199 elif isinstance(y_pred, list): 

200 y_pred = np.vstack([p[:, -1] for p in y_pred]).T 

201 

202 if sample_weight is not None: 

203 return self._sign * self._score_func(y, y_pred, 

204 sample_weight=sample_weight, 

205 **self._kwargs) 

206 else: 

207 return self._sign * self._score_func(y, y_pred, **self._kwargs)

243def check_scoring(estimator, scoring=None, allow_none=False): 

244 """Determine scorer from user options. 

245 

246 A TypeError will be thrown if the estimator cannot be scored. 

247 

248 Parameters 

249 ---------- 

250 estimator : estimator object implementing 'fit' 

251 The object to use to fit the data. 

252 

253 scoring : string, callable or None, optional, default: None 

254 A string (see model evaluation documentation) or 

255 a scorer callable object / function with signature 

256 ``scorer(estimator, X, y)``. 

257 

258 allow_none : boolean, optional, default: False 

259 If no scoring is specified and the estimator has no score function, we 

260 can either return None or raise an exception. 

261 

262 Returns 

263 ------- 

264 scoring : callable 

265 A scorer callable object / function with signature 

266 ``scorer(estimator, X, y)``. 

267 """ 

268 if not hasattr(estimator, 'fit'): 

269 raise TypeError("estimator should be an estimator implementing " 

270 "'fit' method, %r was passed" % estimator) 

271 if isinstance(scoring, str): 

272 return get_scorer(scoring) 

273 elif callable(scoring): 

274 # Heuristic to ensure user has not passed a metric 

275 module = getattr(scoring, '__module__', None) 

276 if hasattr(module, 'startswith') and \ 

277 module.startswith('sklearn.metrics.') and \ 

278 not module.startswith('sklearn.metrics.scorer') and \ 

279 not module.startswith('sklearn.metrics.tests.'): 

280 raise ValueError('scoring value %r looks like it is a metric ' 

281 'function rather than a scorer. A scorer should ' 

282 'require an estimator as its first parameter. ' 

283 'Please use `make_scorer` to convert a metric ' 

284 'to a scorer.' % scoring) 

285 return get_scorer(scoring) 

286 elif scoring is None: 

287 if hasattr(estimator, 'score'): 

288 return _passthrough_scorer 

289 elif allow_none: 

290 return None 

291 else: 

292 raise TypeError( 

293 "If no scoring is specified, the estimator passed should " 

294 "have a 'score' method. The estimator %r does not." 

295 % estimator) 

296 elif isinstance(scoring, Iterable): 

297 raise ValueError("For evaluating multiple scores, use " 

298 "sklearn.model_selection.cross_validate instead. " 

299 "{0} was passed.".format(scoring)) 

300 else: 

301 raise ValueError("scoring value should either be a callable, string or" 

302 " None. %r was passed" % scoring)

305def _check_multimetric_scoring(estimator, scoring=None): 

306 """Check the scoring parameter in cases when multiple metrics are allowed 

307 

308 Parameters 

309 ---------- 

310 estimator : sklearn estimator instance 

311 The estimator for which the scoring will be applied. 

312 

313 scoring : string, callable, list/tuple, dict or None, default: None 

314 A single string (see :ref:`scoring_parameter`) or a callable 

315 (see :ref:`scoring`) to evaluate the predictions on the test set. 

316 

317 For evaluating multiple metrics, either give a list of (unique) strings 

318 or a dict with names as keys and callables as values. 

319 

320 NOTE that when using custom scorers, each scorer should return a single 

321 value. Metric functions returning a list/array of values can be wrapped 

322 into multiple scorers that return one value each. 

323 

324 See :ref:`multimetric_grid_search` for an example. 

325 

326 If None the estimator's score method is used. 

327 The return value in that case will be ``{'score': <default_scorer>}``. 

328 If the estimator's score method is not available, a ``TypeError`` 

329 is raised. 

330 

331 Returns 

332 ------- 

333 scorers_dict : dict 

334 A dict mapping each scorer name to its validated scorer. 

335 

336 is_multimetric : bool 

337 True if scorer is a list/tuple or dict of callables 

338 False if scorer is None/str/callable 

339 """ 

340 if callable(scoring) or scoring is None or isinstance(scoring, 

341 str): 

342 scorers = {"score": check_scoring(estimator, scoring=scoring)} 

343 return scorers, False 

344 else: 

345 err_msg_generic = ("scoring should either be a single string or " 

346 "callable for single metric evaluation or a " 

347 "list/tuple of strings or a dict of scorer name " 

348 "mapped to the callable for multiple metric " 

349 "evaluation. Got %s of type %s" 

350 % (repr(scoring), type(scoring))) 

351 

352 if isinstance(scoring, (list, tuple, set)): 

353 err_msg = ("The list/tuple elements must be unique " 

354 "strings of predefined scorers. ") 

355 invalid = False 

356 try: 

357 keys = set(scoring) 

358 except TypeError: 

359 invalid = True 

360 if invalid: 

361 raise ValueError(err_msg) 

362 

363 if len(keys) != len(scoring): 

364 raise ValueError(err_msg + "Duplicate elements were found in" 

365 " the given list. %r" % repr(scoring)) 

366 elif len(keys) > 0: 

367 if not all(isinstance(k, str) for k in keys): 

368 if any(callable(k) for k in keys): 

369 raise ValueError(err_msg + 

370 "One or more of the elements were " 

371 "callables. Use a dict of score name " 

372 "mapped to the scorer callable. " 

373 "Got %r" % repr(scoring)) 

374 else: 

375 raise ValueError(err_msg + 

376 "Non-string types were found in " 

377 "the given list. Got %r" 

378 % repr(scoring)) 

379 scorers = {scorer: check_scoring(estimator, scoring=scorer) 

380 for scorer in scoring} 

381 else: 

382 raise ValueError(err_msg + 

383 "Empty list was given. %r" % repr(scoring)) 

384 

385 elif isinstance(scoring, dict): 

386 keys = set(scoring) 

387 if not all(isinstance(k, str) for k in keys): 

388 raise ValueError("Non-string types were found in the keys of " 

389 "the given dict. scoring=%r" % repr(scoring)) 

390 if len(keys) == 0: 

391 raise ValueError("An empty dict was passed. %r" 

392 % repr(scoring)) 

393 scorers = {key: check_scoring(estimator, scoring=scorer) 

394 for key, scorer in scoring.items()} 

395 else: 

396 raise ValueError(err_msg_generic) 

397 return scorers, True

400def make_scorer(score_func, greater_is_better=True, needs_proba=False, 

401 needs_threshold=False, **kwargs): 

402 """Make a scorer from a performance metric or loss function. 

403 

404 This factory function wraps scoring functions for use in GridSearchCV 

405 and cross_val_score. It takes a score function, such as ``accuracy_score``, 

406 ``mean_squared_error``, ``adjusted_rand_index`` or ``average_precision`` 

407 and returns a callable that scores an estimator's output. 

408 

409 Read more in the :ref:`User Guide <scoring>`. 

410 

411 Parameters 

412 ---------- 

413 score_func : callable, 

414 Score function (or loss function) with signature 

415 ``score_func(y, y_pred, **kwargs)``. 

416 

417 greater_is_better : boolean, default=True 

418 Whether score_func is a score function (default), meaning high is good, 

419 or a loss function, meaning low is good. In the latter case, the 

420 scorer object will sign-flip the outcome of the score_func. 

421 

422 needs_proba : boolean, default=False 

423 Whether score_func requires predict_proba to get probability estimates 

424 out of a classifier. 

425 

426 If True, for binary `y_true`, the score function is supposed to accept 

427 a 1D `y_pred` (i.e., probability of the positive class, shape 

428 `(n_samples,)`). 

429 

430 needs_threshold : boolean, default=False 

431 Whether score_func takes a continuous decision certainty. 

432 This only works for binary classification using estimators that 

433 have either a decision_function or predict_proba method. 

434 

435 If True, for binary `y_true`, the score function is supposed to accept 

436 a 1D `y_pred` (i.e., probability of the positive class or the decision 

437 function, shape `(n_samples,)`). 

438 

439 For example ``average_precision`` or the area under the roc curve 

440 can not be computed using discrete predictions alone. 

441 

442 **kwargs : additional arguments 

443 Additional parameters to be passed to score_func. 

444 

445 Returns 

446 ------- 

447 scorer : callable 

448 Callable object that returns a scalar score; greater is better. 

449 

450 Examples 

451 -------- 

452 >>> from sklearn.metrics import fbeta_score, make_scorer 

453 >>> ftwo_scorer = make_scorer(fbeta_score, beta=2) 

454 >>> ftwo_scorer 

455 make_scorer(fbeta_score, beta=2) 

456 >>> from sklearn.model_selection import GridSearchCV 

457 >>> from sklearn.svm import LinearSVC 

458 >>> grid = GridSearchCV(LinearSVC(), param_grid={'C': [1, 10]}, 

459 ... scoring=ftwo_scorer) 

460 """ 

461 sign = 1 if greater_is_better else -1 

462 if needs_proba and needs_threshold: 

463 raise ValueError("Set either needs_proba or needs_threshold to True," 

464 " but not both.") 

465 if needs_proba: 

466 cls = _ProbaScorer 

467 elif needs_threshold: 

468 cls = _ThresholdScorer 

469 else: 

470 cls = _PredictScorer 

471 return cls(score_func, sign, kwargs)
