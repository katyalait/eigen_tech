14def _generate_enums(hypothesis, reference, preprocess=str.lower): 

15 """ 

16 Takes in string inputs for hypothesis and reference and returns  

17 enumerated word lists for each of them 

18 

19 :param hypothesis: hypothesis string 

20 :type hypothesis: str 

21 :param reference: reference string 

22 :type reference: str 

23 :preprocess: preprocessing method (default str.lower) 

24 :type preprocess: method 

25 :return: enumerated words list 

26 :rtype: list of 2D tuples, list of 2D tuples 

27 """ 

28 hypothesis_list = list(enumerate(preprocess(hypothesis).split())) 

29 reference_list = list(enumerate(preprocess(reference).split())) 

30 return hypothesis_list, reference_list

49def _match_enums(enum_hypothesis_list, enum_reference_list): 

50 """ 

51 matches exact words in hypothesis and reference and returns 

52 a word mapping between enum_hypothesis_list and enum_reference_list 

53 based on the enumerated word id. 

54 

55 :param enum_hypothesis_list: enumerated hypothesis list 

56 :type enum_hypothesis_list: list of tuples 

57 :param enum_reference_list: enumerated reference list 

58 :type enum_reference_list: list of 2D tuples 

59 :return: enumerated matched tuples, enumerated unmatched hypothesis tuples,  

60 enumerated unmatched reference tuples 

61 :rtype: list of 2D tuples, list of 2D tuples, list of 2D tuples 

62 """ 

63 word_match = [] 

64 for i in range(len(enum_hypothesis_list))[::-1]: 

65 for j in range(len(enum_reference_list))[::-1]: 

66 if enum_hypothesis_list[i][1] == enum_reference_list[j][1]: 

67 word_match.append((enum_hypothesis_list[i][0],enum_reference_list[j][0])) 

68 (enum_hypothesis_list.pop(i)[1],enum_reference_list.pop(j)[1]) 

69 break 

70 return word_match, enum_hypothesis_list, enum_reference_list

72def _enum_stem_match(enum_hypothesis_list, enum_reference_list, stemmer = PorterStemmer()): 

73 """ 

74 Stems each word and matches them in hypothesis and reference  

75 and returns a word mapping between enum_hypothesis_list and  

76 enum_reference_list based on the enumerated word id. The function also  

77 returns a enumerated list of unmatched words for hypothesis and reference. 

78 

79 :param enum_hypothesis_list: 

80 :type enum_hypothesis_list: 

81 :param enum_reference_list: 

82 :type enum_reference_list: 

83 :param stemmer: nltk.stem.api.StemmerI object (default PorterStemmer()) 

84 :type stemmer: nltk.stem.api.StemmerI or any class that implements a stem method 

85 :return: enumerated matched tuples, enumerated unmatched hypothesis tuples,  

86 enumerated unmatched reference tuples 

87 :rtype: list of 2D tuples, list of 2D tuples, list of 2D tuples 

88 """ 

89 stemmed_enum_list1 = [(word_pair[0],stemmer.stem(word_pair[1])) \ 

90 for word_pair in enum_hypothesis_list] 

91 

92 stemmed_enum_list2 = [(word_pair[0],stemmer.stem(word_pair[1])) \ 

93 for word_pair in enum_reference_list] 

94 

95 word_match, enum_unmat_hypo_list, enum_unmat_ref_list = \ 

96 _match_enums(stemmed_enum_list1, stemmed_enum_list2) 

97 

98 enum_unmat_hypo_list = list(zip(*enum_unmat_hypo_list)) if len(enum_unmat_hypo_list)>0 else [] 

99 

100 enum_unmat_ref_list = list(zip(*enum_unmat_ref_list)) if len(enum_unmat_ref_list)>0 else [] 

101 

102 enum_hypothesis_list = list(filter(lambda x:x[0] not in enum_unmat_hypo_list, 

103 enum_hypothesis_list)) 

104 

105 enum_reference_list = list(filter(lambda x:x[0] not in enum_unmat_ref_list, 

106 enum_reference_list)) 

107 

108 return word_match, enum_hypothesis_list, enum_reference_list

129def _enum_wordnetsyn_match(enum_hypothesis_list, enum_reference_list, wordnet = wordnet): 

130 """ 

131 Matches each word in reference to a word in hypothesis  

132 if any synonym of a hypothesis word is the exact match  

133 to the reference word. 

134 

135 :param enum_hypothesis_list: enumerated hypothesis list 

136 :param enum_reference_list: enumerated reference list 

137 :param wordnet: a wordnet corpus reader object (default nltk.corpus.wordnet) 

138 :type wordnet: WordNetCorpusReader 

139 :return: list of matched tuples, unmatched hypothesis list, unmatched reference list 

140 :rtype: list of tuples, list of tuples, list of tuples 

141 

142 """ 

143 word_match = [] 

144 for i in range(len(enum_hypothesis_list))[::-1]: 

145 hypothesis_syns = set(chain(*[[lemma.name() for lemma in synset.lemmas() 

146 if lemma.name().find('_')<0] 

147 for synset in \ 

148 wordnet.synsets( 

149 enum_hypothesis_list[i][1])] 

150 )).union({enum_hypothesis_list[i][1]}) 

151 for j in range(len(enum_reference_list))[::-1]: 

152 if enum_reference_list[j][1] in hypothesis_syns: 

153 word_match.append((enum_hypothesis_list[i][0],enum_reference_list[j][0])) 

154 enum_hypothesis_list.pop(i),enum_reference_list.pop(j) 

155 break 

156 return word_match, enum_hypothesis_list, enum_reference_list

173def _enum_allign_words(enum_hypothesis_list, enum_reference_list, 

174 stemmer=PorterStemmer(), wordnet = wordnet): 

175 """ 

176 Aligns/matches words in the hypothesis to reference by sequentially  

177 applying exact match, stemmed match and wordnet based synonym match.  

178 in case there are multiple matches the match which has the least number 

179 of crossing is chosen. Takes enumerated list as input instead of  

180 string input 

181 

182 :param enum_hypothesis_list: enumerated hypothesis list 

183 :param enum_reference_list: enumerated reference list 

184 :param stemmer: nltk.stem.api.StemmerI object (default PorterStemmer()) 

185 :type stemmer: nltk.stem.api.StemmerI or any class that implements a stem method 

186 :param wordnet: a wordnet corpus reader object (default nltk.corpus.wordnet) 

187 :type wordnet: WordNetCorpusReader 

188 :return: sorted list of matched tuples, unmatched hypothesis list,  

189 unmatched reference list 

190 :rtype: list of tuples, list of tuples, list of tuples 

191 """ 

192 exact_matches, enum_hypothesis_list, enum_reference_list = \ 

193 _match_enums(enum_hypothesis_list, enum_reference_list) 

194 

195 stem_matches, enum_hypothesis_list, enum_reference_list = \ 

196 _enum_stem_match(enum_hypothesis_list, enum_reference_list, 

197 stemmer = stemmer) 

198 

199 wns_matches, enum_hypothesis_list, enum_reference_list = \ 

200 _enum_wordnetsyn_match(enum_hypothesis_list, enum_reference_list, 

201 wordnet = wordnet) 

202 

203 return (sorted(exact_matches + stem_matches + wns_matches, key=lambda wordpair:wordpair[0]), 

204 enum_hypothesis_list, enum_reference_list)

226def _count_chunks(matches): 

227 """ 

228 Counts the fewest possible number of chunks such that matched unigrams 

229 of each chunk are adjacent to each other. This is used to caluclate the 

230 fragmentation part of the metric. 

231 

232 :param matches: list containing a mapping of matched words (output of allign_words) 

233 :return: Number of chunks a sentence is divided into post allignment 

234 :rtype: int 

235 """ 

236 i=0 

237 chunks = 1 

238 while(i<len(matches)-1): 

239 if (matches[i+1][0]==matches[i][0]+1) and (matches[i+1][1]==matches[i][1]+1): 

240 i+=1 

241 continue 

242 i+=1 

243 chunks += 1 

244 return chunks

246def single_meteor_score(reference, 

247 hypothesis, 

248 preprocess = str.lower, 

249 stemmer = PorterStemmer(), 

250 wordnet = wordnet, 

251 alpha=0.9, 

252 beta=3, 

253 gamma=0.5): 

254 """ 

255 Calculates METEOR score for single hypothesis and reference as per  

256 "Meteor: An Automatic Metric for MT Evaluation with HighLevels of 

257 Correlation with Human Judgments" by Alon Lavie and Abhaya Agarwal,  

258 in Proceedings of ACL.  

259 http://www.cs.cmu.edu/~alavie/METEOR/pdf/Lavie-Agarwal-2007-METEOR.pdf 

260 

261 

262 >>> hypothesis1 = 'It is a guide to action which ensures that the military always obeys the commands of the party' 

263 

264 >>> reference1 = 'It is a guide to action that ensures that the military will forever heed Party commands' 

265 

266 

267 >>> round(single_meteor_score(reference1, hypothesis1),4) 

268 0.7398 

269 

270 If there is no words match during the alignment the method returns the  

271 score as 0. We can safely return a zero instead of raising a  

272 division by zero error as no match usually implies a bad translation.  

273 

274 >>> round(meteor_score('this is a cat', 'non matching hypothesis'),4)  

275 0.0 

276 

277 :param references: reference sentences 

278 :type references: list(str) 

279 :param hypothesis: a hypothesis sentence 

280 :type hypothesis: str 

281 :param preprocess: preprocessing function (default str.lower) 

282 :type preprocess: method 

283 :param stemmer: nltk.stem.api.StemmerI object (default PorterStemmer()) 

284 :type stemmer: nltk.stem.api.StemmerI or any class that implements a stem method 

285 :param wordnet: a wordnet corpus reader object (default nltk.corpus.wordnet) 

286 :type wordnet: WordNetCorpusReader 

287 :param alpha: parameter for controlling relative weights of precision and recall. 

288 :type alpha: float 

289 :param beta: parameter for controlling shape of penalty as a  

290 function of as a function of fragmentation. 

291 :type beta: float 

292 :param gamma: relative weight assigned to fragmentation penality. 

293 :type gamma: float 

294 :return: The sentence-level METEOR score. 

295 :rtype: float 

296 """ 

297 enum_hypothesis, enum_reference = _generate_enums(hypothesis, 

298 reference, 

299 preprocess = preprocess) 

300 translation_length = len(enum_hypothesis) 

301 reference_length = len(enum_reference) 

302 matches, _, _ = _enum_allign_words(enum_hypothesis, enum_reference) 

303 matches_count = len(matches) 

304 try: 

305 precision = float(matches_count)/translation_length 

306 recall = float(matches_count)/reference_length 

307 fmean = (precision*recall)/(alpha*precision+(1-alpha)*recall) 

308 chunk_count = float(_count_chunks(matches)) 

309 frag_frac = chunk_count/matches_count 

310 except ZeroDivisionError: 

311 return 0.0 

312 penalty = gamma*frag_frac**beta 

313 return (1-penalty)*fmean

315def meteor_score(references, 

316 hypothesis, 

317 preprocess = str.lower, 

318 stemmer = PorterStemmer(), 

319 wordnet = wordnet, 

320 alpha=0.9, 

321 beta=3, 

322 gamma=0.5): 

323 """ 

324 Calculates METEOR score for hypothesis with multiple references as  

325 described in "Meteor: An Automatic Metric for MT Evaluation with  

326 HighLevels of Correlation with Human Judgments" by Alon Lavie and  

327 Abhaya Agarwal, in Proceedings of ACL.  

328 http://www.cs.cmu.edu/~alavie/METEOR/pdf/Lavie-Agarwal-2007-METEOR.pdf 

329 

330 

331 In case of multiple references the best score is chosen. This method  

332 iterates over single_meteor_score and picks the best pair among all  

333 the references for a given hypothesis 

334 

335 >>> hypothesis1 = 'It is a guide to action which ensures that the military always obeys the commands of the party' 

336 >>> hypothesis2 = 'It is to insure the troops forever hearing the activity guidebook that party direct' 

337 

338 >>> reference1 = 'It is a guide to action that ensures that the military will forever heed Party commands' 

339 >>> reference2 = 'It is the guiding principle which guarantees the military forces always being under the command of the Party' 

340 >>> reference3 = 'It is the practical guide for the army always to heed the directions of the party' 

341 

342 >>> round(meteor_score([reference1, reference2, reference3], hypothesis1),4) 

343 0.7398 

344 

345 If there is no words match during the alignment the method returns the  

346 score as 0. We can safely return a zero instead of raising a  

347 division by zero error as no match usually implies a bad translation.  

348 

349 >>> round(meteor_score(['this is a cat'], 'non matching hypothesis'),4)  

350 0.0 

351 

352 :param references: reference sentences 

353 :type references: list(str) 

354 :param hypothesis: a hypothesis sentence 

355 :type hypothesis: str 

356 :param preprocess: preprocessing function (default str.lower) 

357 :type preprocess: method 

358 :param stemmer: nltk.stem.api.StemmerI object (default PorterStemmer()) 

359 :type stemmer: nltk.stem.api.StemmerI or any class that implements a stem method 

360 :param wordnet: a wordnet corpus reader object (default nltk.corpus.wordnet) 

361 :type wordnet: WordNetCorpusReader 

362 :param alpha: parameter for controlling relative weights of precision and recall. 

363 :type alpha: float 

364 :param beta: parameter for controlling shape of penalty as a function  

365 of as a function of fragmentation. 

366 :type beta: float 

367 :param gamma: relative weight assigned to fragmentation penality. 

368 :type gamma: float 

369 :return: The sentence-level METEOR score. 

370 :rtype: float 

371 """ 

372 return max([single_meteor_score(reference, 

373 hypothesis, 

374 stemmer = stemmer, 

375 wordnet = wordnet, 

376 alpha = alpha, 

377 beta = beta, 

378 gamma = gamma) for reference in references])
