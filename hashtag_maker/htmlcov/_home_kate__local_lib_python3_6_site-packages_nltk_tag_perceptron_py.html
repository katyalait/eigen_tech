32class AveragedPerceptron(object): 

33 

34 '''An averaged perceptron, as implemented by Matthew Honnibal. 

35 

36 See more implementation details here: 

37 https://explosion.ai/blog/part-of-speech-pos-tagger-in-python 

38 ''' 

39 

40 def __init__(self): 

41 # Each feature gets its own weight vector, so weights is a dict-of-dicts 

42 self.weights = {} 

43 self.classes = set() 

44 # The accumulated values, for the averaging. These will be keyed by 

45 # feature/clas tuples 

46 self._totals = defaultdict(int) 

47 # The last time the feature was changed, for the averaging. Also 

48 # keyed by feature/clas tuples 

49 # (tstamps is short for timestamps) 

50 self._tstamps = defaultdict(int) 

51 # Number of instances seen 

52 self.i = 0 

53 

54 

55 def _softmax(self, scores): 

56 s = np.fromiter(scores.values(), dtype=float) 

57 exps = np.exp(s) 

58 return exps / np.sum(exps) 

59 

60 

61 def predict(self, features, return_conf=False): 

62 '''Dot-product the features and current weights and return the best label.''' 

63 scores = defaultdict(float) 

64 for feat, value in features.items(): 

65 if feat not in self.weights or value == 0: 

66 continue 

67 weights = self.weights[feat] 

68 for label, weight in weights.items(): 

69 scores[label] += value * weight 

70 

71 # Do a secondary alphabetic sort, for stability 

72 best_label = max(self.classes, key=lambda label: (scores[label], label)) 

73 # compute the confidence 

74 conf = max(self._softmax(scores)) if return_conf == True else None 

75 

76 return best_label, conf 

77 

78 

79 def update(self, truth, guess, features): 

80 '''Update the feature weights.''' 

81 

82 def upd_feat(c, f, w, v): 

83 param = (f, c) 

84 self._totals[param] += (self.i - self._tstamps[param]) * w 

85 self._tstamps[param] = self.i 

86 self.weights[f][c] = w + v 

87 

88 self.i += 1 

89 if truth == guess: 

90 return None 

91 for f in features: 

92 weights = self.weights.setdefault(f, {}) 

93 upd_feat(truth, f, weights.get(truth, 0.0), 1.0) 

94 upd_feat(guess, f, weights.get(guess, 0.0), -1.0) 

95 

96 def average_weights(self): 

97 '''Average weights from all iterations.''' 

98 for feat, weights in self.weights.items(): 

99 new_feat_weights = {} 

100 for clas, weight in weights.items(): 

101 param = (feat, clas) 

102 total = self._totals[param] 

103 total += (self.i - self._tstamps[param]) * weight 

104 averaged = round(total / self.i, 3) 

105 if averaged: 

106 new_feat_weights[clas] = averaged 

107 self.weights[feat] = new_feat_weights 

108 

109 def save(self, path): 

110 '''Save the pickled model weights.''' 

111 with open(path, 'wb') as fout: 

112 return pickle.dump(dict(self.weights), fout) 

113 

114 def load(self, path): 

115 '''Load the pickled model weights.''' 

116 self.weights = load(path)

119@python_2_unicode_compatible 

120class PerceptronTagger(TaggerI): 

121 

122 ''' 

123 Greedy Averaged Perceptron tagger, as implemented by Matthew Honnibal. 

124 See more implementation details here: 

125 https://explosion.ai/blog/part-of-speech-pos-tagger-in-python 

126 

127 >>> from nltk.tag.perceptron import PerceptronTagger 

128 

129 Train the model 

130 

131 >>> tagger = PerceptronTagger(load=False) 

132 

133 >>> tagger.train([[('today','NN'),('is','VBZ'),('good','JJ'),('day','NN')], 

134 ... [('yes','NNS'),('it','PRP'),('beautiful','JJ')]]) 

135 

136 >>> tagger.tag(['today','is','a','beautiful','day']) 

137 [('today', 'NN'), ('is', 'PRP'), ('a', 'PRP'), ('beautiful', 'JJ'), ('day', 'NN')] 

138 

139 Use the pretrain model (the default constructor) 

140 

141 >>> pretrain = PerceptronTagger() 

142 

143 >>> pretrain.tag('The quick brown fox jumps over the lazy dog'.split()) 

144 [('The', 'DT'), ('quick', 'JJ'), ('brown', 'NN'), ('fox', 'NN'), ('jumps', 'VBZ'), ('over', 'IN'), ('the', 'DT'), ('lazy', 'JJ'), ('dog', 'NN')] 

145 

146 >>> pretrain.tag("The red cat".split()) 

147 [('The', 'DT'), ('red', 'JJ'), ('cat', 'NN')] 

148 ''' 

149 

150 START = ['-START-', '-START2-'] 

151 END = ['-END-', '-END2-'] 

152 

153 def __init__(self, load=True): 

154 ''' 

155 :param load: Load the pickled model upon instantiation. 

156 ''' 

157 self.model = AveragedPerceptron() 

158 self.tagdict = {} 

159 self.classes = set() 

160 if load: 

161 AP_MODEL_LOC = 'file:' + str( 

162 find('taggers/averaged_perceptron_tagger/' + PICKLE) 

163 ) 

164 self.load(AP_MODEL_LOC) 

165 

166 def tag(self, tokens, return_conf=False, use_tagdict=True): 

167 ''' 

168 Tag tokenized sentences. 

169 :params tokens: list of word 

170 :type tokens: list(str) 

171 ''' 

172 prev, prev2 = self.START 

173 output = [] 

174 

175 context = self.START + [self.normalize(w) for w in tokens] + self.END 

176 for i, word in enumerate(tokens): 

177 tag, conf = (self.tagdict.get(word), 1.0) if use_tagdict == True else (None, None) 

178 if not tag: 

179 features = self._get_features(i, word, context, prev, prev2) 

180 tag, conf = self.model.predict(features, return_conf) 

181 output.append((word, tag, conf) if return_conf == True else (word, tag)) 

182 

183 prev2 = prev 

184 prev = tag 

185 

186 return output

188 def train(self, sentences, save_loc=None, nr_iter=5): 

189 '''Train a model from sentences, and save it at ``save_loc``. ``nr_iter`` 

190 controls the number of Perceptron training iterations. 

191 

192 :param sentences: A list or iterator of sentences, where each sentence 

193 is a list of (words, tags) tuples. 

194 :param save_loc: If not ``None``, saves a pickled model in this location. 

195 :param nr_iter: Number of training iterations. 

196 ''' 

197 # We'd like to allow ``sentences`` to be either a list or an iterator, 

198 # the latter being especially important for a large training dataset. 

199 # Because ``self._make_tagdict(sentences)`` runs regardless, we make 

200 # it populate ``self._sentences`` (a list) with all the sentences. 

201 # This saves the overheard of just iterating through ``sentences`` to 

202 # get the list by ``sentences = list(sentences)``. 

203 

204 self._sentences = list() # to be populated by self._make_tagdict... 

205 self._make_tagdict(sentences) 

206 self.model.classes = self.classes 

207 for iter_ in range(nr_iter): 

208 c = 0 

209 n = 0 

210 for sentence in self._sentences: 

211 words, tags = zip(*sentence) 

212 

213 prev, prev2 = self.START 

214 context = self.START + [self.normalize(w) for w in words] + self.END 

215 for i, word in enumerate(words): 

216 guess = self.tagdict.get(word) 

217 if not guess: 

218 feats = self._get_features(i, word, context, prev, prev2) 

219 guess,_ = self.model.predict(feats) 

220 self.model.update(tags[i], guess, feats) 

221 prev2 = prev 

222 prev = guess 

223 c += guess == tags[i] 

224 n += 1 

225 random.shuffle(self._sentences) 

226 logging.info("Iter {0}: {1}/{2}={3}".format(iter_, c, n, _pc(c, n))) 

227 

228 # We don't need the training sentences anymore, and we don't want to 

229 # waste space on them when we pickle the trained tagger. 

230 self._sentences = None 

231 

232 self.model.average_weights() 

233 # Pickle as a binary file 

234 if save_loc is not None: 

235 with open(save_loc, 'wb') as fout: 

236 # changed protocol from -1 to 2 to make pickling Python 2 compatible 

237 pickle.dump((self.model.weights, self.tagdict, self.classes), fout, 2)

248 def normalize(self, word): 

249 ''' 

250 Normalization used in pre-processing. 

251 - All words are lower cased 

252 - Groups of digits of length 4 are represented as !YEAR; 

253 - Other digits are represented as !DIGITS 

254 

255 :rtype: str 

256 ''' 

257 if '-' in word and word[0] != '-': 

258 return '!HYPHEN' 

259 elif word.isdigit() and len(word) == 4: 

260 return '!YEAR' 

261 elif word[0].isdigit(): 

262 return '!DIGITS' 

263 else: 

264 return word.lower()

266 def _get_features(self, i, word, context, prev, prev2): 

267 '''Map tokens into a feature representation, implemented as a 

268 {hashable: int} dict. If the features change, a new model must be 

269 trained. 

270 ''' 

271 

272 def add(name, *args): 

273 features[' '.join((name,) + tuple(args))] += 1 

274 

275 i += len(self.START) 

276 features = defaultdict(int) 

277 # It's useful to have a constant feature, which acts sort of like a prior 

278 add('bias') 

279 add('i suffix', word[-3:]) 

280 add('i pref1', word[0]) 

281 add('i-1 tag', prev) 

282 add('i-2 tag', prev2) 

283 add('i tag+i-2 tag', prev, prev2) 

284 add('i word', context[i]) 

285 add('i-1 tag+i word', prev, context[i]) 

286 add('i-1 word', context[i - 1]) 

287 add('i-1 suffix', context[i - 1][-3:]) 

288 add('i-2 word', context[i - 2]) 

289 add('i+1 word', context[i + 1]) 

290 add('i+1 suffix', context[i + 1][-3:]) 

291 add('i+2 word', context[i + 2]) 

292 return features

294 def _make_tagdict(self, sentences): 

295 ''' 

296 Make a tag dictionary for single-tag words. 

297 :param sentences: A list of list of (word, tag) tuples. 

298 ''' 

299 counts = defaultdict(lambda: defaultdict(int)) 

300 for sentence in sentences: 

301 self._sentences.append(sentence) 

302 for word, tag in sentence: 

303 counts[word][tag] += 1 

304 self.classes.add(tag) 

305 freq_thresh = 20 

306 ambiguity_thresh = 0.97 

307 for word, tag_freqs in counts.items(): 

308 tag, mode = max(tag_freqs.items(), key=lambda item: item[1]) 

309 n = sum(tag_freqs.values()) 

310 # Don't add rare words to the tag dictionary 

311 # Only add quite unambiguous words 

312 if n >= freq_thresh and (mode / n) >= ambiguity_thresh: 

313 self.tagdict[word] = tag

320def _load_data_conll_format(filename): 

321 print('Read from file: ', filename) 

322 with open(filename, 'rb') as fin: 

323 sentences = [] 

324 sentence = [] 

325 for line in fin.readlines(): 

326 line = line.strip() 

327 # print line 

328 if len(line) == 0: 

329 sentences.append(sentence) 

330 sentence = [] 

331 continue 

332 tokens = line.split('\t') 

333 word = tokens[1] 

334 tag = tokens[4] 

335 sentence.append((word, tag)) 

336 return sentences
